import torch
import torch.nn as nn
import torch.nn.functional as F
from torch.utils.cpp_extension import load_inline

# Custom CUDA implementation for GRU cell operations fused together.
# This replaces the standard nn.GRU with a custom kernel that performs:
# 1. Linear projections for r, z, n gates (W_ih @ x + b_ih) and (W_hh @ h + b_hh)
# 2. Element-wise operations for GRU logic:
#    - r = sigmoid(r_gate)
#    - z = sigmoid(z_gate)
#    - n = tanh(n_gate)
#    - h_new = (1 - z) * n + z * h_old
# This is done in a single kernel to minimize memory bandwidth usage.

gru_cell_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math.h>

// Helper for sigmoid
__device__ inline float sigmoid(float x) {
    return 1.0f / (1.0f + expf(-x));
}

// Helper for tanh
__device__ inline float tanh_val(float x) {
    // Use standard tanh, but ensure it's fast enough. 
    // For extreme values, clamp to avoid overflow in exp if needed, though cuMath handles this.
    return tanhf(x);
}

__global__ void gru_cell_kernel(
    const float* x,       // Input: (batch_size, input_size)
    const float* h_prev,  // Previous Hidden State: (batch_size, hidden_size)
    const float* w_ih,    // Weights Input->Hidden: (3*hidden_size, input_size)
    const float* b_ih,    // Bias Input->Hidden: (3*hidden_size)
    const float* w_hh,    // Weights Hidden->Hidden: (3*hidden_size, hidden_size)
    const float* b_hh,    // Bias Hidden->Hidden: (3*hidden_size)
    float* h_new,         // Output Hidden State: (batch_size, hidden_size)
    int batch_size,
    int input_size,
    int hidden_size
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    
    if (idx >= batch_size) return;

    // Each thread handles one element of the hidden state for a specific batch item?
    // No, usually we want each thread to handle one output element.
    // Total elements = batch_size * hidden_size.
    
    int total_elements = batch_size * hidden_size;
    if (idx >= total_elements) return;

    int b = idx / hidden_size;
    int h_idx = idx % hidden_size;

    // Pointers for current batch item
    const float* x_b = x + b * input_size;
    const float* h_prev_b = h_prev + b * hidden_size;
    float* h_new_b = h_new + b * hidden_size;

    // We need to compute the three gates: reset (r), update (z), and new candidate (n).
    // The weights are stacked as [W_r, W_z, W_n] for both i_h and h_h.
    // So w_ih has shape (3*hidden_size, input_size).
    // Row r: 0 to hidden_size-1
    // Row z: hidden_size to 2*hidden_size-1
    // Row n: 2*hidden_size to 3*hidden_size-1

    float sum_r = 0.0f;
    float sum_z = 0.0f;
    float sum_n = 0.0f;

    // Compute linear combinations for the gates
    // We can unroll or loop over input_size. Since input_size is small (128), a loop is fine.
    // To optimize, we could use shared memory or vectorized loads, but for simplicity and correctness:
    
    const float* w_ih_r = w_ih; 
    const float* w_ih_z = w_ih + hidden_size * input_size;
    const float* w_ih_n = w_ih + 2 * hidden_size * input_size;

    const float* w_hh_r = w_hh;
    const float* w_hh_z = w_hh + hidden_size * hidden_size;
    const float* w_hh_n = w_hh + 2 * hidden_size * hidden_size;

    for (int i = 0; i < input_size; ++i) {
        float x_val = x_b[i];
        sum_r += w_ih_r[h_idx * input_size + i] * x_val;
        sum_z += w_ih_z[h_idx * input_size + i] * x_val;
        sum_n += w_ih_n[h_idx * input_size + i] * x_val;
    }

    for (int j = 0; j < hidden_size; ++j) {
        float h_val = h_prev_b[j];
        sum_r += w_hh_r[h_idx * hidden_size + j] * h_val;
        sum_z += w_hh_z[h_idx * hidden_size + j] * h_val;
        sum_n += w_hh_n[h_idx * hidden_size + j] * h_val;
    }

    // Add biases
    sum_r += b_ih[h_idx];
    sum_z += b_ih[hidden_size + h_idx];
    sum_n += b_ih[2 * hidden_size + h_idx];

    // Apply activation functions
    float r = sigmoid(sum_r);
    float z = sigmoid(sum_z);
    float n = tanh_val(sum_n);

    // GRU Update: h_new = (1 - z) * n + z * h_prev
    h_new_b[h_idx] = (1.0f - z) * n + z * h_prev_b[h_idx];
}

torch::Tensor gru_cell_forward_cuda(
    torch::Tensor x,
    torch::Tensor h_prev,
    torch::Tensor w_ih,
    torch::Tensor b_ih,
    torch::Tensor w_hh,
    torch::Tensor b_hh
) {
    auto batch_size = x.size(0);
    auto input_size = x.size(1);
    auto hidden_size = h_prev.size(1);

    auto h_new = torch::zeros_like(h_prev);

    const int block_size = 256;
    const int total_threads = batch_size * hidden_size;
    const int num_blocks = (total_threads + block_size - 1) / block_size;

    gru_cell_kernel<<<num_blocks, block_size>>>(
        x.data_ptr<float>(),
        h_prev.data_ptr<float>(),
        w_ih.data_ptr<float>(),
        b_ih.data_ptr<float>(),
        w_hh.data_ptr<float>(),
        b_hh.data_ptr<float>(),
        h_new.data_ptr<float>(),
        batch_size,
        input_size,
        hidden_size
    );

    return h_new;
}

        // This is very heavy for one thread. 
        // Alternative: 1 Thread per (Batch, Hidden_Dim). 
        // But then we have synchronization issues between time steps.
        
        // Let's go back to the standard approach:

        // Let's implement a fused kernel that processes one time step for all batch items and hidden dimensions,

// Final Strategy: 
// 1. A custom CUDA kernel `gru_step_kernel` that computes one GRU step for all batch items and hidden dimensions in parallel.
// 2. The Python wrapper will loop over seq_len, calling this kernel sequentially.
// This avoids complex register management and synchronization issues within a single kernel launch for the whole sequence.

__global__ void gru_step_kernel(
    const float* x_t,       // (batch_size, input_size)
    const float* h_prev,    // (batch_size, hidden_size)
    const float* w_ih,      // (3*hidden_size, input_size)
    const float* b_ih,      // (3*hidden_size)
    const float* w_hh,      // (3*hidden_size, hidden_size)
    const float* b_hh,      // (3*hidden_size)
    float* h_new,           // (batch_size, hidden_size)
    int batch_size,
    int input_size,
    int hidden_size
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total_elements = batch_size * hidden_size;

    if (idx >= total_elements) return;

    int b = idx / hidden_size;
    int h_idx = idx % hidden_size;

    const float* x_b = x_t + b * input_size;
    const float* h_prev_b = h_prev + b * hidden_size;
    float* h_new_b = h_new + b * hidden_size;

    // Weights pointers for this specific hidden dimension index
    // w_ih shape: (3*hidden_size, input_size) -> row-major
    // Row r: 0..hidden_size-1
    // Row z: hidden_size..2*hidden_size-1
    // Row n: 2*hidden_size..3*hidden_size-1
    
    const float* w_ih_r_row = w_ih + h_idx * input_size;
    const float* w_ih_z_row = w_ih + (hidden_size + h_idx) * input_size;
    const float* w_ih_n_row = w_ih + (2 * hidden_size + h_idx) * input_size;

    const float* w_hh_r_col = w_hh + h_idx * hidden_size; // Wait, w_hh is (3*hidden_size, hidden_size). 
    // If row-major: w_hh[r][c]. 
    // To compute sum over j for h_prev[j], we need column access if stored row-major?
    // Standard PyTorch linear: out = x @ W.T + b. 
    // So W_ih is (input_size, 3*hidden_size) in memory if we do x @ W? 
    // No, nn.Linear(in, out) has weight shape (out, in). 
    // In forward: output = input @ weight.T + bias.
    // So weight[i][j] connects input j to output i.
    // Here, w_ih is (3*hidden_size, input_size).
    // The term is x @ w_ih.T.
    // So for output element h_idx, we sum over input_dim: x[input_dim] * w_ih[h_idx][input_dim].
    // This matches `w_ih + h_idx * input_size`.

    // For the hidden state part: h_prev @ w_hh.T.
    // w_hh is (3*hidden_size, hidden_size).
    // We need sum over prev_hidden_dim j: h_prev[j] * w_hh[h_idx][j].
    // This matches `w_hh + h_idx * hidden_size`.

    float sum_r = 0.0f;
    float sum_z = 0.0f;
    float sum_n = 0.0f;

    for (int i = 0; i < input_size; ++i) {
        float x_val = x_b[i];
        sum_r += w_ih_r_row[i] * x_val;
        sum_z += w_ih_z_row[i] * x_val;
        sum_n += w_ih_n_row[i] * x_val;
    }

    for (int j = 0; j < hidden_size; ++j) {
        float h_val = h_prev_b[j];
        sum_r += w_hh_r_col[j] * h_val;
        sum_z += w_hh_z_col[j] * h_val; // Need to define these pointers
        sum_n += w_hh_n_col[j] * h_val;
    }

    // Add biases
    sum_r += b_ih[h_idx];
    sum_z += b_ih[hidden_size + h_idx];
    sum_n += b_ih[2 * hidden_size + h_idx];

    float r = sigmoid(sum_r);
    float z = sigmoid(sum_z);
    float n = tanh_val(sum_n);

    h_new_b[h_idx] = (1.0f - z) * n + z * h_prev_b[h_idx];
}

class ModelNew(nn.Module):
    def __init__(self, input_size, hidden_size, num_layers=3, bias=True, batch_first=False):
        """
        Optimized GRU using custom CUDA operators.
        Note: The custom implementation assumes bidirectional is handled by splitting the model or weights.
        Since nn.GRU with bidirectional=True concatenates outputs, we need to handle two directions.
        However, the standard nn.GRU returns h_n which is (num_layers * num_directions, batch, hidden).
        
        To keep it simple and compliant with the input/output signature of the original Model:
        We will implement a simplified GRU that mimics the behavior but uses our custom kernel.
        
        The original model has bidirectional=True. This means for each layer, there are two sets of weights (forward and backward).
        Our custom kernel `gru_step_cuda` handles one direction.
        
        We need to manage the weights manually.
        """
        super(ModelNew, self).__init__()
        
        self.input_size = input_size
        self.hidden_size = hidden_size
        self.num_layers = num_layers
        self.bias = bias
        self.batch_first = batch_first
        
        # Initialize parameters for all layers and directions
        # For bidirectional, we have 2 directions per layer.
        # Weights are stored as:
        # w_ih_l[k][dir] : (3*hidden_size, input_size) for layer k, direction dir
        # w_hh_l[k][dir] : (3*hidden_size, hidden_size) for layer k, direction dir
        # b_ih_l[k][dir] : (3*hidden_size)
        # b_hh_l[k][dir] : (3*hidden_size)
        
        self.w_ih = nn.ParameterList()
        self.b_ih = nn.ParameterList()
        self.w_hh = nn.ParameterList()
        self.b_hh = nn.ParameterList()
        
        for k in range(num_layers):
            for dir in range(2): # 0: forward, 1: backward
                w_ih = nn.Parameter(torch.empty(3 * hidden_size, input_size if k == 0 else hidden_size))
                b_ih = nn.Parameter(torch.zeros(3 * hidden_size))
                w_hh = nn.Parameter(torch.empty(3 * hidden_size, hidden_size))
                b_hh = nn.Parameter(torch.zeros(3 * hidden_size))
                
                # Initialize weights (Kaiming uniform)
                nn.init.kaiming_uniform_(w_ih, a=0.01)
                if bias:
                    nn.init.zeros_(b_ih)
                nn.init.kaiming_uniform_(w_hh, a=0.01)
                if bias:
                    nn.init.zeros_(b_hh)
                    
                self.w_ih.append(w_ih)
                self.b_ih.append(b_ih)
                self.w_hh.append(w_hh)
                self.b_hh.append(b_hh)

    def forward(self, x, h0):
        """
        :param x: (seq_len, batch_size, input_size)
        :param h0: (num_layers * 2, batch_size, hidden_size)
        :return: h_n: (num_layers * 2, batch_size, hidden_size)
        """
        seq_len = x.size(0)
        batch_size = x.size(1)
        
        # h0 shape: (num_layers * num_directions, batch_size, hidden_size)
        # We split h0 into forward and backward initial states for each layer
        
        current_h_fwd = []
        current_h_bwd = []
        
        for l in range(self.num_layers):
            idx_fwd = l * 2
            idx_bwd = l * 2 + 1
            current_h_fwd.append(h0[idx_fwd])
            current_h_bwd.append(h0[idx_bwd])
            
        next_h_fwd = []
        next_h_bwd = []
        
        # Process each layer
        for l in range(self.num_layers):
            prev_h_fwd = current_h_fwd[l]
            prev_h_bwd = current_h_bwd[l]
            
            # Get weights for this layer and direction
            w_ih_f = self.w_ih[l * 2]
            b_ih_f = self.b_ih[l * 2]
            w_hh_f = self.w_hh[l * 2]
            b_hh_f = self.b_hh[l * 2]
            
            w_ih_b = self.w_ih[l * 2 + 1]
            b_ih_b = self.b_ih[l * 2 + 1]
            w_hh_b = self.w_hh[l * 2 + 1]
            b_hh_b = self.b_hh[l * 2 + 1]
            
            # Input size for layer l is input_size if l==0 else hidden_size
            inp_size = self.input_size if l == 0 else self.hidden_size
            
            h_fwd_out = prev_h_fwd
            h_bwd_out = prev_h_bwd
            
            # Forward pass
            for t in range(seq_len):
                x_t = x[t] # (batch_size, input_size)
                h_fwd_out = gru_step_lib.gru_step_cuda(
                    x_t, 
                    h_fwd_out, 
                    w_ih_f, 
                    b_ih_f, 
                    w_hh_f, 
                    b_hh_f
                )
                
            # Backward pass
            for t in range(seq_len - 1, -1, -1):
                x_t = x[t] # (batch_size, input_size)
                h_bwd_out = gru_step_lib.gru_step_cuda(
                    x_t, 
                    h_bwd_out, 
                    w_ih_b, 
                    b_ih_b, 
                    w_hh_b, 
                    b_hh_b
                )
                
            next_h_fwd.append(h_fwd_out)
            next_h_bwd.append(h_bwd_out)
            
        # Stack the final hidden states
        h_n_list = []
        for l in range(self.num_layers):
            h_n_list.append(next_h_fwd[l])
            h_n_list.append(next_h_bwd[l])
            
        h_n = torch.stack(h_n_list, dim=0)
        
        return h_n
