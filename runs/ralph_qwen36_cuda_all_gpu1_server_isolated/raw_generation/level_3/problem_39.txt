__global__ void gru_step_kernel_fused_v2(
    const float* x,       // (batch, input_size)
    const float* h_prev,  // (batch, hidden_size)
    const float* w_ih,    // (3*hidden, input)
    const float* b_ih,    // (3*hidden)
    const float* w_hh,    // (3*hidden, hidden)
    const float* b_hh,    // (3*hidden)
    float* h_new,         // (batch, hidden_size)
    int batch_size,
    int input_size,
    int hidden_size
) {
    int batch_idx = blockIdx.x;
    if (batch_idx >= batch_size) return;

    int hid_idx = threadIdx.x + blockDim.x * blockIdx.y;
    if (hid_idx >= hidden_size) return;

    // Indices for the 3 gates
    int r_hid = hid_idx;
    int z_hid = hid_idx + hidden_size;
    int n_hid = hid_idx + 2 * hidden_size;

    // Compute Reset Gate
    float r_pre = 0.0f;
    for (int i = 0; i < input_size; ++i) {
        r_pre += x[batch_idx * input_size + i] * w_ih[r_hid * input_size + i];
    }
    for (int j = 0; j < hidden_size; ++j) {
        r_pre += h_prev[batch_idx * hidden_size + j] * w_hh[r_hid * hidden_size + j];
    }
    r_pre += b_ih[r_hid] + b_hh[r_hid];

    // Compute Update Gate

    for (int i = 0; i < input_size; ++i) {
        z_pre += x[batch_idx * input_size + i] * w_ih[z_hid * input_size + i];
    }
    for (int j = 0; j < hidden_size; ++j) {
        z_pre += h_prev[batch_idx * hidden_size + j] * w_hh[z_hid * hidden_size + j];
    }
    z_pre += b_ih[z_hid] + b_hh[z_hid];
    float z = sigmoid(z_pre);

    // Compute New Gate
    // n = tanh(W_in @ x + b_in + W_hn @ (r * h_prev) + b_n)
    float n_pre = 0.0f;
    for (int i = 0; i < input_size; ++i) {
        n_pre += x[batch_idx * input_size + i] * w_ih[n_hid * input_size + i];
    }
    for (int j = 0; j < hidden_size; ++j) {
        // r * h_prev: we need the reset gate value for each hidden unit.
        // We only computed r for hid_idx. We need r for all j.

// Shared Memory Approach
__global__ void gru_step_kernel_shared(
    const float* x,       // (batch, input_size)
    const float* h_prev,  // (batch, hidden_size)
    const float* w_ih,    // (3*hidden, input)
    const float* b_ih,    // (3*hidden)
    const float* w_hh,    // (3*hidden, hidden)
    const float* b_hh,    // (3*hidden)
    float* h_new,         // (batch, hidden_size)
    int batch_size,
    int input_size,
    int hidden_size
) {
    extern __shared__ float shared_mem[];
    
    int batch_idx = blockIdx.x;
    if (batch_idx >= batch_size) return;

    int hid_idx = threadIdx.x + blockDim.x * blockIdx.y;
    if (hid_idx >= hidden_size) return;

    // Shared memory layout:
    // [0 .. hidden_size-1]: r gates
    // [hidden_size .. 2*hidden_size-1]: z gates
    // [2*hidden_size .. 3*hidden_size-1]: n_pre (or just compute n on fly)
    
    float* r_shared = shared_mem;
    float* z_shared = shared_mem + hidden_size;
    float* h_prev_shared = shared_mem + 2 * hidden_size; // Copy h_prev to shared for fast access?

    // Load h_prev into shared memory
    h_prev_shared[hid_idx] = h_prev[batch_idx * hidden_size + hid_idx];
    __syncthreads();

    // Compute Reset Gate
    float r_pre = 0.0f;
    int r_hid = hid_idx;
    for (int i = 0; i < input_size; ++i) {
        r_pre += x[batch_idx * input_size + i] * w_ih[r_hid * input_size + i];
    }
    for (int j = 0; j < hidden_size; ++j) {
        r_pre += h_prev_shared[j] * w_hh[r_hid * hidden_size + j];
    }
    r_pre += b_ih[r_hid] + b_hh[r_hid];
    r_shared[hid_idx] = sigmoid(r_pre);

    // Compute Update Gate
    float z_pre = 0.0f;
    int z_hid = hid_idx + hidden_size;
    for (int i = 0; i < input_size; ++i) {
        z_pre += x[batch_idx * input_size + i] * w_ih[z_hid * input_size + i];
    }
    for (int j = 0; j < hidden_size; ++j) {
        z_pre += h_prev_shared[j] * w_hh[z_hid * hidden_size + j];
    }
    z_pre += b_ih[z_hid] + b_hh[z_hid];
    z_shared[hid_idx] = sigmoid(z_pre);

    __syncthreads();

    // Compute New Gate and Final Output
    float n_pre = 0.0f;
    int n_hid = hid_idx + 2 * hidden_size;
    
    // W_in @ x
    for (int i = 0; i < input_size; ++i) {
        n_pre += x[batch_idx * input_size + i] * w_ih[n_hid * input_size + i];
    }
    
    // W_hn @ (r * h_prev)
    for (int j = 0; j < hidden_size; ++j) {
        float r_j = r_shared[j];
        float h_j = h_prev_shared[j];
        n_pre += r_j * h_j * w_hh[n_hid * hidden_size + j];
    }
    n_pre += b_ih[n_hid] + b_hh[n_hid];
    
    float n = tanh_val(n_pre);
    
    // h_new = (1 - z) * n + z * h_prev
    float z_val = z_shared[hid_idx];
    float h_prev_val = h_prev_shared[hid_idx];
    h_new[batch_idx * hidden_size + hid_idx] = (1.0f - z_val) * n + z_val * h_prev_val;
}

    torch::Tensor x,       // (batch, input_size)

    torch::Tensor b_ih,    // (3*hidden)
    torch::Tensor w_hh,    // (3*hidden, hidden)
    torch::Tensor b_hh     // (3*hidden)

    auto input_size = x.size(1);
    auto hidden_size = h_prev.size(1);

    auto h_new = torch::zeros_like(h_prev);

    const int block_size = 256; // Or 512, depending on hidden_size
    // We need enough threads to cover hidden_size. 
    // If hidden_size > block_size * grid_y, we might need more blocks.
    // Let's use a 2D grid: blockIdx.x = batch, blockIdx.y = hid_idx / block_size
    
    const int num_blocks_x = batch_size;
    const int num_blocks_y = (hidden_size + block_size - 1) / block_size;
    
    dim3 grid(num_blocks_x, num_blocks_y);
    dim3 threads(block_size);
    
    size_t shared_mem_size = 3 * hidden_size * sizeof(float);

        x.data_ptr<float>(),

        b_hh.data_ptr<float>(),
        h_new.data_ptr<float>(),
        batch_size,
        input_size,
        hidden_size
    );

    return h_new;
}

            # If it's not the first layer, input is the output of the previous layer
            if layer > 0:
                x = output
            
            # Call custom CUDA kernel for this layer
            layer_output = gru_module.gru_forward_cuda(x, current_h, w_ih, b_ih, w_hh, b_hh)
            
            # Update h_n for this layer
            h_n.append(layer_output[-1]) # Last time step
            
            # If not the last layer, prepare input for next layer
            if layer < self.num_layers - 1:
                x = layer_output
            
            output = layer_output
        
        h_n = torch.stack(h_n)
        
        return output, h_n
