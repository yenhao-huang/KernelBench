// Kernel for Global Average Pooling + Linear Layer
__global__ void gap_fc_kernel(
    const float* input, 
    const float* weight, 
    const float* bias, 
    float* output, 
    int batch_size, 
    int in_features, 
    int out_features, 
    int height_in, 
    int width_in
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total_elements = batch_size * out_features;
    
    if (idx >= total_elements) return;

    int c_out = idx % out_features;
    int n = idx / out_features;

    float sum = 0.0f;
    int spatial_size = height_in * width_in;
    
    // Global Average Pooling: average over H and W for each channel
    for (int c_in = 0; c_in < in_features; ++c_in) {
        float channel_sum = 0.0f;
        for (int i = 0; i < spatial_size; ++i) {
            int input_idx = n * in_features * spatial_size + 
                            c_in * spatial_size + i;
            channel_sum += input[input_idx];
        }
        float avg = channel_sum / spatial_size;
        
        // Linear layer: y = xW^T + b
        int weight_idx = c_out * in_features + c_in;
        sum += avg * weight[weight_idx];
    }
    
    if (bias != nullptr) {
        sum += bias[c_out];
    }
    
    output[idx] = sum;
}

            conv_dw(int(32 * alpha), int(64 * alpha), 1),
            conv_dw(int(64 * alpha), int(128 * alpha), 2),
            conv_dw(int(128 * alpha), int(128 * alpha), 1),
            conv_dw(int(128 * alpha), int(256 * alpha), 2),
            conv_dw(int(256 * alpha), int(256 * alpha), 1),
            conv_dw(int(256 * alpha), int(512 * alpha), 2),
            conv_dw(int(512 * alpha), int(512 * alpha), 1),
            conv_dw(int(512 * alpha), int(512 * alpha), 1),
            conv_dw(int(512 * alpha), int(512 * alpha), 1),
            conv_dw(int(512 * alpha), int(512 * alpha), 1),
            conv_dw(int(512 * alpha), int(512 * alpha), 1),
            conv_dw(int(512 * alpha), int(1024 * alpha), 2),
            conv_dw(int(1024 * alpha), int(1024 * alpha), 1)
        )
        
        self.fc = nn.Linear(int(1024 * alpha), num_classes)

    def forward(self, x):
        """
        Optimized forward pass using custom CUDA kernels.
        
        :param x: The input tensor, shape (batch_size, input_channels, height, width)
        :return: The output tensor, shape (batch_size, num_classes)
        """
        
        # Helper to apply Conv+BN+ReLU fusion
        def apply_conv_bn_relu(inp, conv_layer, bn_layer, is_depthwise=False):
            w = conv_layer.weight
            b = conv_layer.bias if conv_layer.bias is not None else torch.zeros_like(bn_layer.running_mean)
            rm = bn_layer.running_mean
            rv = bn_layer.running_var
            
            stride = conv_layer.stride[0]
            pad = conv_layer.padding[0]
            
            if is_depthwise:
                return mobile_net_ops.depthwise_conv_bn_relu_cuda(inp, w, b, rm, rv, stride, stride, pad, pad)
            else:
                return mobile_net_ops.conv_bn_relu_cuda(inp, w, b, rm, rv, stride, stride, pad, pad)

        # Helper to apply Pointwise Conv+BN+ReLU fusion
        def apply_pointwise_conv_bn_relu(inp, conv_layer, bn_layer):
            w = conv_layer.weight
            b = conv_layer.bias if conv_layer.bias is not None else torch.zeros_like(bn_layer.running_mean)
            rm = bn_layer.running_mean
            rv = bn_layer.running_var
            
            return mobile_net_ops.pointwise_conv_bn_relu_cuda(inp, w, b, rm, rv)

        # Block 1: Standard Conv2d + BN + ReLU
        x = apply_conv_bn_relu(x, self.model[0][0], self.model[0][1])
        
        # Blocks 2-15: Depthwise Separable Convolutions
        for i in range(1, len(self.model)):
            block = self.model[i]
            # First part: Depthwise Conv + BN + ReLU
            x = apply_conv_bn_relu(x, block[0], block[1], is_depthwise=True)
            # Second part: Pointwise Conv + BN + ReLU
            x = apply_pointwise_conv_bn_relu(x, block[2], block[3])
            
        # Global Average Pooling
        x = F.avg_pool2d(x, 7)
        
        # Flatten
        x = x.view(x.size(0), -1)
        
        # FC Layer
        x = self.fc(x)
        
        return x
