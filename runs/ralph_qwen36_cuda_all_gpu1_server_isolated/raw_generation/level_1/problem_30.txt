import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Define the custom CUDA kernel for Softsign activation: x / (1 + |x|)
softsign_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void softsign_kernel(const float* input, float* output, int size) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < size) {
        float val = input[idx];
        // Compute absolute value manually to avoid potential library call overhead or branching issues
        float abs_val = val < 0 ? -val : val;
        output[idx] = val / (1.0f + abs_val);
    }
}

torch::Tensor softsign_cuda(torch::Tensor input) {
    auto size = input.numel();
    auto output = torch::empty_like(input);

    if (size == 0) {
        return output;
    }

    const int block_size = 256;
    const int num_blocks = (size + block_size - 1) / block_size;

    softsign_kernel<<<num_blocks, block_size>>>(input.data_ptr<float>(), output.data_ptr<float>(), size);

    // Check for kernel launch errors
    cudaError_t err = cudaGetLastError();
    if (err != cudaSuccess) {
        throw std::runtime_error(std::string("CUDA error in softsign: ") + cudaGetErrorString(err));
    }

    return output;
}
"""

softsign_cpp_source = (
    "torch::Tensor softsign_cuda(torch::Tensor input);"
)

# Compile the inline CUDA code for Softsign
softsign_module = load_inline(
    name="softsign_custom",
    cpp_sources=softsign_cpp_source,
    cuda_sources=softsign_source,
    functions=["softsign_cuda"],
    verbose=False,
    extra_cflags=["-O3"],
    extra_ldflags=["-lcudart"]
)


class ModelNew(nn.Module):
    """
    Optimized model that performs a Softsign activation using custom CUDA.
    """
    def __init__(self):
        super(ModelNew, self).__init__()
        self.softsign_op = softsign_module.softsign_cuda

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Applies Softsign activation to the input tensor using custom CUDA kernel.

        Args:
            x (torch.Tensor): Input tensor of any shape.

        Returns:
            torch.Tensor: Output tensor with Softsign applied, same shape as input.
        """
        return self.softsign_op(x)
