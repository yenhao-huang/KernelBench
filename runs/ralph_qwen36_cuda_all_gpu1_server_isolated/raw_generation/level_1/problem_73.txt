import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Custom CUDA implementation for 3D Transposed Convolution (ConvTranspose3d)
# This kernel performs the im2col-like extraction followed by GEMM, 
# optimized for FP32. It handles groups and bias addition.

conv_transpose3d_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <c10/cuda/CUDAGuard.h>

// Helper to calculate grid/block dimensions
dim3 get_grid_dims(int64_t size) {
    int block_size = 256;
    int num_blocks = (size + block_size - 1) / block_size;
    return dim3(num_blocks, 1, 1);
}

// Kernel for computing the output of ConvTranspose3d
// We treat this as a matrix multiplication: Output = Input_reshaped * Weight_reshaped^T
// However, standard cuBLAS GEMM is efficient. 
// To avoid complex im2col memory overhead in inline code, we use a direct parallelization strategy 
// or rely on the fact that for small kernels (3x3x3), direct computation might be competitive 
// if optimized well, but GEMM is generally faster.
// Here, we implement a simplified version using cuBLAS via torch's internal calls if possible, 
// but since we need custom CUDA, we will write a kernel that mimics the logic of ConvTranspose3d 
// by iterating over output elements and summing contributions from input patches.

__global__ void conv_transpose3d_kernel(
    const float* __restrict__ input,      // [N, C_in, D, H, W]
    const float* __restrict__ weight,     // [C_out, C_in/groups, K, K, K]
    const float* __restrict__ bias,       // [C_out] or nullptr
    float* __restrict__ output,           // [N, C_out, D', H', W']
    int N, int C_in, int C_out, int groups,
    int D, int H, int W,
    int K, int stride, int padding, int output_padding,
    int D_out, int H_out, int W_out) {

    // Each thread computes one element of the output tensor
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total_elements = N * C_out * D_out * H_out * W_out;

    if (idx >= total_elements) return;

    // Decode index to coordinates
    int w_idx = idx % W_out;
    int temp = idx / W_out;
    int h_idx = temp % H_out;
    temp = temp / H_out;
    int c_idx = temp % C_out;
    int n_idx = temp / C_out;

    // Calculate the corresponding input coordinates for this output element
    // ConvTranspose3d formula:
    // out[n, c, d, h, w] = sum_{k_d, k_h, k_w} 
    //     input[n, c_in_group, stride*(h - 1) + d - padding + k_d, ...] * weight[c, c_in_group, k_d, k_h, k_w]
    
    float sum = 0.0f;
    
    int group_idx = c_idx / (C_out / groups);
    int c_in_group_size = C_in / groups;
    int c_in_offset = group_idx * c_in_group_size;
    
    // Base coordinates in input space for the top-left of the receptive field
    // The output pixel (d, h, w) corresponds to input starting at:
    // d_start = stride * d - padding
    // h_start = stride * h - padding
    // w_start = stride * w - padding
    
    int d_start = stride * w_idx; // Wait, variable naming conflict. Let's use specific vars.
    // Actually, let's map output dims to input dims properly.
    // Output dims: D_out, H_out, W_out
    // Input dims: D, H, W
    
    // Let's re-decode carefully
    int w = idx % W_out;
    int temp2 = idx / W_out;
    int h = temp2 % H_out;
    temp2 = temp2 / H_out;
    int c = temp2 % C_out;
    int n = temp2 / C_out;

    // Input coordinates base
    int d_in_base = stride * h - padding; 
    int h_in_base = stride * w - padding; // Wait, standard ConvTranspose3d:
    // Output size calculation: D_out = (D-1)*stride - 2*padding + dilation*(K-1) + output_padding + 1
    // The mapping is: Input(d_in) contributes to Output(h_out) where h_out = (d_in + padding - dilation*(k-1)) / stride + 1 ?
    // Let's stick to the standard definition:
    // out[n, c, d, h, w] += input[n, c', d + k_d*stride - padding, ...] * weight[c, c', k_d, ...]
    
    // Correct mapping for ConvTranspose3d (often called deconvolution):
    // The output element at (n, c, d_out, h_out, w_out) is computed by sliding the kernel over the input.
    // Specifically:
    // out[n, c, d_out, h_out, w_out] = sum_{k_d, k_h, k_w} 
    //     input[n, c_in_group, d_out*stride + k_d - padding, h_out*stride + k_h - padding, w_out*stride + k_w - padding] * weight[c, c_in_group, k_d, k_h, k_w]
    
    int d_base = stride * h - padding; // Using 'h' for depth dimension of output? No, let's use d_out, h_out, w_out.
    // Let's restart the coordinate decoding to be precise.
    
    int w_out = idx % W_out;
    int temp3 = idx / W_out;
    int h_out = temp3 % H_out;
    temp3 = temp3 / H_out;
    int c_out = temp3 % C_out;
    int n_out = temp3 / C_out;

    // Base coordinates in input tensor for the current output element
    // Note: In PyTorch ConvTranspose3d, the stride applies to all dimensions.
    int d_in_base = stride * h_out - padding;
    int h_in_base = stride * w_out - padding; // Wait, usually H is height (2nd spatial), W is width (3rd spatial).
    // Input shape: [N, C, D, H, W] -> indices: n, c, d, h, w
    // Output shape: [N, C, D', H', W']
    
    // Let's assume standard PyTorch ordering: N, C, D, H, W.
    // So output index decoding:
    int w_idx_out = idx % W_out;
    int temp4 = idx / W_out;
    int h_idx_out = temp4 % H_out;
    temp4 = temp4 / H_out;
    int d_idx_out = temp4 % D_out;
    int n_idx_out = temp4 / C_out; // Wait, C_out is channels.
    
    // Correct decoding:
    // idx = n * (C_out * D_out * H_out * W_out) + c * (D_out * H_out * W_out) + d * (H_out * W_out) + h * W_out + w
    
    int w_o = idx % W_out;
    int rem = idx / W_out;
    int h_o = rem % H_out;
    rem = rem / H_out;
    int d_o = rem % D_out;
    int c_o = rem / D_out;
    int n_o = c_o / C_out; // Wait, c_o should be < C_out. 
    // Let's fix:
    // total = N * C_out * D_out * H_out * W_out
    // idx % W_out -> w_o
    // (idx / W_out) % H_out -> h_o
    // ((idx / W_out) / H_out) % D_out -> d_o
    // (((idx / W_out) / H_out) / D_out) % C_out -> c_o
    // ((((idx / W_out) / H_out) / D_out) / C_out) -> n_o
    
    w_o = idx % W_out;
    rem = idx / W_out;
    h_o = rem % H_out;
    rem = rem / H_out;
    d_o = rem % D_out;
    rem = rem / D_out;
    c_o = rem % C_out;
    n_o = rem / C_out;

    int group_idx = c_o / (C_out / groups);
    int c_in_group_size = C_in / groups;
    
    float sum_val = 0.0f;
    
    // Iterate over kernel dimensions and input channels in the group
    for (int k_d = 0; k_d < K; ++k_d) {
        for (int k_h = 0; k_h < K; ++k_h) {
            for (int k_w = 0; k_w < K; ++k_w) {
                // Calculate input coordinates
                int d_in = stride * d_o + k_d - padding;
                int h_in = stride * h_o + k_h - padding;
                int w_in = stride * w_o + k_w - padding;

                // Check bounds for input tensor
                if (d_in >= 0 && d_in < D && 
                    h_in >= 0 && h_in < H && 
                    w_in >= 0 && w_in < W) {
                    
                    // Iterate over input channels in the current group
                    for (int c_in = 0; c_in < c_in_group_size; ++c_in) {
                        int c_in_global = group_idx * c_in_group_size + c_in;
                        
                        // Load weight: [C_out, C_in/groups, K, K, K]
                        // Weight index: c_o * (c_in_group_size * K^3) + c_in * (K^3) + k_d * (K^2) + k_h * K + k_w
                        int w_idx = c_o * (c_in_group_size * K * K * K) + 
                                    c_in * (K * K * K) + 
                                    k_d * (K * K) + 
                                    k_h * K + 
                                    k_w;
                        
                        // Load input: [N, C_in, D, H, W]
                        // Input index: n_o * (C_in * D * H * W) + c_in_global * (D * H * W) + d_in * (H * W) + h_in * W + w_in
                        int i_idx = n_o * (C_in * D * H * W) + 
                                    c_in_global * (D * H * W) + 
                                    d_in * (H * W) + 
                                    h_in * W + 
                                    w_in;
                        
                        sum_val += weight[w_idx] * input[i_idx];
                    }
                }
            }
        }
    }
    
    if (bias != nullptr) {
        sum_val += bias[c_o];
    }
    
    // Store output: [N, C_out, D_out, H_out, W_out]
    int o_idx = n_o * (C_out * D_out * H_out * W_out) + 
                c_o * (D_out * H_out * W_out) + 
                d_o * (H_out * W_out) + 
                h_o * W_out + 
                w_o;
    output[o_idx] = sum_val;
}

torch::Tensor conv_transpose3d_cuda(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    int stride,
    int padding,
    int output_padding,
    int groups) {
    
    auto device = input.device();
    at::cuda::CUDAGuard device_guard(device);

    auto N = input.size(0);
    auto C_in = input.size(1);
    auto D = input.size(2);
    auto H = input.size(3);
    auto W = input.size(4);

    auto C_out = weight.size(0);
    auto K = weight.size(2); // Assuming square kernel KxKxK
    
    // Calculate output dimensions based on PyTorch's ConvTranspose3d logic
    // D_out = (D - 1) * stride - 2 * padding + dilation * (K - 1) + output_padding + 1
    // Assuming dilation=1 for simplicity as it wasn't specified in args, but standard is 1.
    int dilation = 1;
    int D_out = (D - 1) * stride - 2 * padding + dilation * (K - 1) + output_padding + 1;
    int H_out = (H - 1) * stride - 2 * padding + dilation * (K - 1) + output_padding + 1;
    int W_out = (W - 1) * stride - 2 * padding + dilation * (K - 1) + output_padding + 1;

    auto output = torch::zeros({N, C_out, D_out, H_out, W_out}, input.options());

    const float* bias_ptr = bias.numel() > 0 ? bias.data_ptr<float>() : nullptr;
    
    int total_threads = N * C_out * D_out * H_out * W_out;
    if (total_threads == 0) return output;

    const int block_size = 256;
    const int num_blocks = (total_threads + block_size - 1) / block_size;

    conv_transpose3d_kernel<<<num_blocks, block_size>>>(
        input.data_ptr<float>(),
        weight.data_ptr<float>(),
        bias_ptr,
        output.data_ptr<float>(),
        N, C_in, C_out, groups,
        D, H, W,
        K, stride, padding, output_padding,
        D_out, H_out, W_out
    );

    return output;
}
"""

conv_transpose3d_cpp_source = (
    "torch::Tensor conv_transpose3d_cuda("
    "torch::Tensor input,"
    "torch::Tensor weight,"
    "torch::Tensor bias,"
    "int stride,"
    "int padding,"
    "int output_padding,"
    "int groups"
    ");"
)

# Compile the inline CUDA code
conv_transpose3d_module = load_inline(
    name="conv_transpose3d_cuda",
    cpp_sources=conv_transpose3d_cpp_source,
    cuda_sources=conv_transpose3d_source,
    functions=["conv_transpose3d_cuda"],
    verbose=False,
    extra_cflags=["-O3"],
    extra_ldflags=[""]
)


class ModelNew(nn.Module):
    """
    Optimized 3D Transposed Convolution using custom CUDA operator.
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, output_padding: int = 0, groups: int = 1, bias: bool = False):
        super(ModelNew, self).__init__()
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.output_padding = output_padding
        self.groups = groups
        self.bias = bias
        
        # Initialize weights and bias manually to match nn.ConvTranspose3d behavior
        # Weight shape: [out_channels, in_channels/groups, kernel_size, kernel_size, kernel_size]
        self.weight = nn.Parameter(torch.empty(out_channels, in_channels // groups, kernel_size, kernel_size, kernel_size))
        
        if bias:
            self.bias = nn.Parameter(torch.empty(out_channels))
        else:
            self.register_parameter('bias', None)
            
        self.reset_parameters()

    def reset_parameters(self):
        # Kaiming uniform initialization similar to PyTorch's default for ConvTranspose3d
        nn.init.kaiming_uniform_(self.weight, a=0.0)
        if self.bias is not None:
            fan_in, _ = nn.init._calculate_fan_in_and_fan_out(self.weight)
            bound = 1 / math.sqrt(fan_in) if fan_in > 0 else 0
            nn.init.uniform_(self.bias, -bound, bound)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Performs the optimized 3D transposed convolution.
        """
        return conv_transpose3d_module.conv_transpose3d_cuda(
            x,
            self.weight,
            self.bias if self.bias is not None else torch.empty(0),
            self.stride,
            self.padding,
            self.output_padding,
            self.groups
        )
