import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Define the custom CUDA kernel for Mean Squared Error
mse_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cmath>

__global__ void mse_kernel(const float* predictions, const float* targets, float* out_sum, int size) {
    // Use atomicAdd to accumulate the squared differences into a single output value
    // We process elements in parallel and sum them up atomically
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < size) {
        float diff = predictions[idx] - targets[idx];
        atomicAdd(out_sum, diff * diff);
    }
}

torch::Tensor mse_cuda(torch::Tensor predictions, torch::Tensor targets) {
    auto size = predictions.numel();
    
    // Create a temporary tensor to hold the sum of squared errors
    // We use float32 for accumulation as per requirements
    auto out_sum = torch::zeros({1}, predictions.options());
    
    const int block_size = 256;
    const int num_blocks = (size + block_size - 1) / block_size;

    mse_kernel<<<num_blocks, block_size>>>(
        predictions.data_ptr<float>(), 
        targets.data_ptr<float>(), 
        out_sum.data_ptr<float>(), 
        size
    );

    // Synchronize to ensure atomic operations are complete before reading the result
    cudaDeviceSynchronize();
    
    // Compute the mean: sum / size
    float sum_val = out_sum.item<float>();
    float mean_val = sum_val / static_cast<float>(size);
    
    return torch::tensor(mean_val, predictions.options());
}
"""

mse_cpp_source = (
    "torch::Tensor mse_cuda(torch::Tensor predictions, torch::Tensor targets);"
)

# Compile the inline CUDA code for MSE calculation
mse_module = load_inline(
    name="mse_custom",
    cpp_sources=mse_cpp_source,
    cuda_sources=mse_source,
    functions=["mse_cuda"],
    verbose=False,
    extra_cflags=["-O2"],
    extra_ldflags=[""]
)


class ModelNew(nn.Module):
    """
    A model that computes the Mean Squared Error loss for regression tasks.
    Optimized with custom CUDA operator.
    """
    def __init__(self):
        super(ModelNew, self).__init__()
        self.mse_op = mse_module.mse_cuda

    def forward(self, predictions, targets):
        return self.mse_op(predictions, targets)
