import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Define the custom CUDA kernel for min reduction along a specific dimension
min_reduction_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <float.h>

__global__ void min_reduction_kernel(const float* input, float* output, int batch_size, int dim1, int dim2) {
    // We assume the reduction is over the last dimension (dim2) for simplicity in this 3D tensor case.
    // If the model allows arbitrary dims, we would need more complex indexing.
    // Given the example inputs: (batch_size, dim1, dim2), and typical usage, 
    // let's assume reduction is over the last dimension (dim=2) or handle general cases.
    // However, to keep it simple and robust for the provided shape (B, D1, D2), 
    // we will implement reduction over the last dimension (dim2).
    
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < batch_size * dim1) {
        int b = idx / dim1;
        int d1 = idx % dim1;
        
        float min_val = FLT_MAX;
        const float* row_ptr = input + (b * dim1 + d1) * dim2;
        
        for (int i = 0; i < dim2; ++i) {
            if (row_ptr[i] < min_val) {
                min_val = row_ptr[i];
            }
        }
        output[idx] = min_val;
    }
}

torch::Tensor min_reduction_cuda(torch::Tensor x, int dim) {
    // For this specific optimization, we assume the reduction is over the last dimension (dim == 2 or -1)
    // as it maps well to a simple kernel. If dim is different, we could transpose, but let's stick to the common case 
    // or handle the general 3D tensor by assuming dim=2 for the provided example structure.
    
    auto sizes = x.sizes();
    int batch_size = sizes[0];
    int dim1 = sizes[1];
    int dim2 = sizes[2];
    
    // If the reduction dimension is not the last one, we would need to permute or write a more complex kernel.
    // For this example, we assume dim=2 (last dimension) for optimal coalesced access if we structure it right,
    // but actually reducing over the last dimension with threads per row is efficient.
    
    auto output = torch::empty({batch_size, dim1}, x.options());
    
    const int block_size = 256;
    const int num_threads = batch_size * dim1;
    const int num_blocks = (num_threads + block_size - 1) / block_size;
    
    min_reduction_kernel<<<num_blocks, block_size>>>(x.data_ptr<float>(), output.data_ptr<float>(), batch_size, dim1, dim2);
    
    return output;
}
"""

min_reduction_cpp_source = (
    "torch::Tensor min_reduction_cuda(torch::Tensor x, int dim);"
)

# Compile the inline CUDA code for min reduction
min_reduction = load_inline(
    name="min_reduction",
    cpp_sources=min_reduction_cpp_source,
    cuda_sources=min_reduction_source,
    functions=["min_reduction_cuda"],
    verbose=False,
    extra_cflags=["-O2"],
    extra_ldflags=[""],
)


class ModelNew(nn.Module):
    """
    Optimized model that performs min reduction over the last dimension using a custom CUDA operator.
    Note: This implementation assumes reduction over the last dimension (dim=2) for the 3D tensor input.
    If the original model supports arbitrary dims, this specific kernel is optimized for dim=-1.
    """
    def __init__(self, dim: int) -> None:
        super().__init__()
        self.dim = dim
        self.min_reduction_op = min_reduction

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # The custom kernel is designed for reduction over the last dimension.
        # If the user specifies a different dimension, we might need to permute.
        # However, to strictly follow the "replace operator" pattern and given the example inputs,
        # we assume the primary use case or optimize for the most common efficient path.
        # For a general solution in a real scenario, we'd handle permutations. 
        # Here, we call the custom kernel which handles dim=2 (last dim).
        
        if self.dim == 2 or self.dim == -1:
            return self.min_reduction_op.min_reduction_cuda(x, self.dim)
        else:
            # Fallback to PyTorch for non-last dimensions if necessary,

            # If dim is 0 or 1, we could transpose, reduce, and transpose back, but that adds overhead.

            return torch.min(x, dim=self.dim)[0]
