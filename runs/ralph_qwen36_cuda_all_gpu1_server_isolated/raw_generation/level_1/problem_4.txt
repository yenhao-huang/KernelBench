```python
import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Define the custom CUDA kernel for matrix-vector multiplication (GEMV)
# This implementation uses shared memory tiling to optimize global memory access patterns.
gemv_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <stdio.h>

// Kernel for Matrix-Vector Multiplication: C = A * B
// A is (M, K), B is (K, 1), C is (M, 1)
// We process rows of A in blocks. Each block handles a chunk of rows.
// To optimize memory access to B, we load chunks of B into shared memory.

__global__ void gemv_kernel(const float* __restrict__ A, const float* __restrict__ B, float* __restrict__ C, int M, int K) {
    // Each thread block handles a tile of rows from A.
    // Let's say each block processes 'BLOCK_M' rows.
    // We will use dynamic shared memory or fixed size depending on constraints.
    // Given K is large (1048576), we cannot load all B into shared memory.
    // Strategy: Load a tile of B into shared memory, compute dot products for the rows assigned to this block, then move to next tile of B.

    const int BLOCK_M = 32; // Number of rows per block
    const int TILE_K = 1024; // Number of columns (elements of B) to load into shared memory at once
    
    extern __shared__ float s_B[];

    int row_start = blockIdx.x * BLOCK_M;
    
    // Each thread in the block handles one or more rows. 
    // If BLOCK_M <= 32, we can have one thread per row.
    // Let's assume gridDim.x covers M/BLOCK_M blocks.
    
    if (row_start >= M) return;

    int tid = threadIdx.x;
    int num_threads = blockDim.x;

    // Initialize output for this row to 0
    float sum[BLOCK_M]; 
    // Note: Using a local array for sums is safer than shared memory for accumulation if BLOCK_M is small.
    // However, to avoid register pressure, we can use registers directly per thread.
    // Let's map threads to rows. If num_threads >= BLOCK_M, each thread handles one row.
    
    // We need to handle the case where num_threads < BLOCK_M or > BLOCK_M gracefully.
    // Standard approach: 1 thread per row if possible.
    
    for (int t = 0; t < BLOCK_M; ++t) {
        sum[t] = 0.0f;
    }

    // Iterate over tiles of K
    for (int k_tile_start = 0; k_tile_start < K; k_tile_start += TILE_K) {
        int k_end = min(k_tile_start + TILE_K, K);
        int tile_size = k_end - k_tile_start;

        // Load B[tile_start : tile_end] into shared memory
        // Each thread loads multiple elements if tile_size > num_threads
        for (int i = tid; i < tile_size; i += num_threads) {
            s_B[i] = B[k_tile_start + i];
        }
        __syncthreads();

        // Compute dot product for each row assigned to this block
        for (int r = 0; r < BLOCK_M; ++r) {
            int global_row = row_start + r;
            if (global_row >= M) continue;

            float local_sum = 0.0f;
            const float* A_row = A + global_row * K;
            
            // Unroll or loop over the tile
            for (int k = 0; k < tile_size; ++k) {
                local_sum += A_row[k_tile_start + k] * s_B[k];
            }
            sum[r] += local_sum;
        }
        
        __syncthreads();
    }

    // Write results back to global memory
    for (int r = 0; r < BLOCK_M; ++r) {
        int global_row = row_start + r;
        if (global_row < M) {
            C[global_row] = sum[r];
        }
    }
}

torch::Tensor gemv_cuda(torch::Tensor A, torch::Tensor B) {
    auto M = A.size(0);
    auto K = A.size(1);
    
    // Ensure inputs are contiguous and on CUDA
    if (!A.is_contiguous()) A = A.contiguous();
    if (!B.is_contiguous()) B = B.contiguous();

    auto C = torch::zeros({M, 1}, A.options());

    const int BLOCK_M = 32;
    const int TILE_K = 1024;
    
    // Calculate grid and block dimensions
    // Number of blocks needed to cover M rows
    int num_blocks = (M + BLOCK_M - 1) / BLOCK_M;
    
    // Block size: We need at least BLOCK_M threads if we want 1 thread per row.
    // However, shared memory usage depends on TILE_K.
    // Shared memory size: TILE_K * sizeof(float)
    int shared_mem_size = TILE_K * sizeof(float);
    
    // Limit block size to 1024 (max threads per block)
    int block_size = BLOCK_M; 
    if (block_size > 1024) block_size = 1024;

    // Launch kernel
    gemv_kernel<<<num_blocks, block_size, shared_mem_size>>>(
        A.data_ptr<float>(), 
        B.data_ptr<float>(), 
        C.data_ptr<float>(), 
        M, 
        K
    );

    return C;
}
"""

gemv_cpp_source = (
    "torch::Tensor gemv_cuda(torch::Tensor A, torch::Tensor B);"
)

# Compile the inline CUDA code for GEMV
gemv = load_inline(
    name="gemv",
    cpp_sources=gemv_cpp_source,
    cuda_sources=gemv_source,
    functions=["gemv_cuda"],
    verbose=True,
    extra_cflags=["-O3"],
    extra_ldflags=[""]
)


class ModelNew(nn.Module):
    """
    Optimized model that performs matrix-vector multiplication (C = A * B)
    using a custom CUDA kernel for better performance.
    """
    def __init__(self):
        super(ModelNew, self).__init__()
        self.gemv_op = gemv

    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        """
        Performs matrix-vector multiplication using custom CUDA kernel.

        Args:
            A: Input matrix of shape (M, K).
            B: Input vector of shape (K, 1).

        Returns:
            Output vector of shape (M, 1).
        """
        return self.gemv_op.gemv_cuda(A, B)
```
