# Define custom CUDA kernels for optimized operations
# We will replace the core Linear (Matmul) and Softmax operations which are often bottlenecks.
# Specifically, we implement a fused Matmul + Bias Add + ReLU (if applicable) or just high-performance Matmul.
# Given the Electra architecture uses GELU, we will focus on optimizing the Linear layers and potentially the final projection.

cuda_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <stdio.h>

// Helper for atomic add if needed, but here we use standard grid-stride loops for matmul-like ops or simple element-wise

// Kernel 1: Optimized Matrix Multiplication (GEMM) using shared memory tiling
// This is a simplified but effective GEMM kernel for FP32
__global__ void gemm_kernel(const float* A, const float* B, float* C, int M, int N, int K) {
    // Block dimensions
    const int BLOCK_SIZE = 16;
    
    // Shared memory for tiles of A and B
    __shared__ float As[BLOCK_SIZE][BLOCK_SIZE];
    __shared__ float Bs[BLOCK_SIZE][BLOCK_SIZE];

    // Thread indices within the block
    int tx = threadIdx.x;
    int ty = threadIdx.y;

    // Global row and column indices for this thread
    int row = blockIdx.y * BLOCK_SIZE + ty;
    int col = blockIdx.x * BLOCK_SIZE + tx;

    float sum = 0.0f;

    // Loop over K dimension in tiles
    for (int i = 0; i < (K + BLOCK_SIZE - 1) / BLOCK_SIZE; ++i) {
        // Load tile from A into shared memory
        if (row < M && (i * BLOCK_SIZE + tx) < K) {
            As[ty][tx] = A[row * K + i * BLOCK_SIZE + tx];
        } else {
            As[ty][tx] = 0.0f;
        }

        // Load tile from B into shared memory
        if ((i * BLOCK_SIZE + ty) < K && col < N) {
            Bs[ty][tx] = B[(i * BLOCK_SIZE + ty) * N + col];
        } else {
            Bs[ty][tx] = 0.0f;
        }

        __syncthreads();

        // Compute partial dot product
        for (int j = 0; j < BLOCK_SIZE; ++j) {
            sum += As[ty][j] * Bs[j][tx];
        }

        __syncthreads();
    }

    // Write result to global memory
    if (row < M && col < N) {
        C[row * N + col] = sum;
    }
}

// Kernel 2: Bias Addition and Activation (GELU approximation or standard GELU via lookup/erf)
// Electra uses GELU. We will implement a fast GELU kernel.
__global__ void gelu_bias_kernel(const float* input, const float* bias, float* output, int size) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < size) {
        float x = input[idx] + bias[idx];
        // GELU(x) = 0.5 * x * (1 + tanh(sqrt(2/pi) * (x + 0.044715 * x^3)))
        float x3 = x * x * x;
        float inner = sqrtf(2.0f / 3.14159265358979323846f) * (x + 0.044715f * x3);
        float tanh_val = tanhf(inner);
        output[idx] = 0.5f * x * (1.0f + tanh_val);
    }
}

// Kernel 3: Final Linear Layer (Matmul) without activation, just bias add if needed
// For the final logits layer, we often just need Matmul + Bias Add
__global__ void linear_logits_kernel(const float* A, const float* B, const float* bias, float* C, int M, int N, int K) {
    // Simplified GEMM for large output dimensions where shared memory might be limited or overhead high
    // Using a simple row-major load and compute pattern optimized for coalesced access
    
    int row = blockIdx.y * blockDim.y + threadIdx.y;
    int col = blockIdx.x * blockDim.x + threadIdx.x;

    if (row < M && col < N) {
        float sum = 0.0f;
        // Unroll or loop over K
        for (int k = 0; k < K; ++k) {
            sum += A[row * K + k] * B[col * K + k]; // Note: B is usually transposed in cuBLAS, here we assume standard layout or handle transpose logic
            // Standard PyTorch linear: out = input @ weight.T + bias
            // If weight is (out_features, in_features), then A is (M, K), B is (N, K) -> need to dot row of A with col of B? 
            // Actually standard matmul C = A * B^T. 
            // Let's assume the caller passes B as transposed or we handle it.
            // To keep it simple and correct for PyTorch Linear:
            // Input: (M, K), Weight: (N, K). Output: (M, N).
            // C[i,j] = sum_k A[i,k] * B[j,k]
        }
        if (bias) {
            sum += bias[col];
        }
        C[row * N + col] = sum;
    }
}

// Wrapper for GEMM
torch::Tensor gemm_cuda(torch::Tensor A, torch::Tensor B) {
    auto M = A.size(0);
    auto K = A.size(1);
    auto N = B.size(0); // Assuming B is (N, K) and we want C = A @ B^T
    
    auto C = torch::zeros({M, N}, A.options());

    const int block_x = 16;
    const int block_y = 16;
    
    dim3 threads(block_x, block_y);
    dim3 blocks((N + block_x - 1) / block_x, (M + block_y - 1) / block_y);

    gemm_kernel<<<blocks, threads>>>(A.data_ptr<float>(), B.data_ptr<float>(), C.data_ptr<float>(), M, N, K);
    
    return C;
}

// Wrapper for GELU + Bias Add
torch::Tensor gelu_bias_cuda(torch::Tensor input, torch::Tensor bias) {
    auto size = input.numel();
    auto output = torch::empty_like(input);
    
    const int block_size = 256;
    const int num_blocks = (size + block_size - 1) / block_size;
    
    gelu_bias_kernel<<<num_blocks, block_size>>>(input.data_ptr<float>(), bias.data_ptr<float>(), output.data_ptr<float>(), size);
    
    return output;
}

// Wrapper for Linear Logits (Matmul + Bias Add)
torch::Tensor linear_logits_cuda(torch::Tensor input, torch::Tensor weight, torch::Tensor bias) {
    auto M = input.size(0);
    auto K = input.size(1);
    auto N = weight.size(0); // Output features
    
    auto output = torch::zeros({M, N}, input.options());

    const int block_x = 32;
    const int block_y = 8;
    
    dim3 threads(block_x, block_y);
    dim3 blocks((N + block_x - 1) / block_x, (M + block_y - 1) / block_y);

    linear_logits_kernel<<<blocks, threads>>>(input.data_ptr<float>(), weight.data_ptr<float>(), bias.data_ptr<float>(), output.data_ptr<float>(), M, N, K);
    
    return output;
}
"""

# Helper class to encapsulate the FFN if we were replacing it, but for this specific output requirement, 
# we focus on the final layer replacement as a demonstration. 
# To make it more robust, let's define the CustomElectraFFN just in case we want to expand, 
# but here we stick to the final layer optimization for clarity and correctness.

class CustomElectraFFN(nn.Module):
    def __init__(self, weight1, bias1, weight2, bias2):
        super().__init__()
        self.weight1 = nn.Parameter(weight1)
        self.bias1 = nn.Parameter(bias1)
        self.weight2 = nn.Parameter(weight2)
        self.bias2 = nn.Parameter(bias2)

    def forward(self, x):
        # Intermediate (GELU)
        # x: (batch, seq_len, hidden_size)
        # weight1: (intermediate_size, hidden_size)
        # bias1: (intermediate_size,)
        
        batch, seq_len, hidden = x.size()
        x_flat = x.view(-1, hidden)
        
        # Custom GELU + Bias Add for intermediate layer
        intermediate_flat = custom_ops.gelu_bias_cuda(custom_ops.gemm_cuda(x_flat, self.weight1.t()), self.bias1)
        
        # Output Linear (no activation)
        output_flat = custom_ops.linear_logits_cuda(intermediate_flat, self.weight2, self.bias2)
        
        return output_flat.view(batch, seq_len, -1)
