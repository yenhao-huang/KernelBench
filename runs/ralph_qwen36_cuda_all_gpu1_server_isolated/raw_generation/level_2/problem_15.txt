    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, bias=True):
        super(ModelNew, self).__init__()
        self.conv_transpose = nn.ConvTranspose3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, bias=bias)
        self.batch_norm = nn.BatchNorm3d(out_channels)
        
        # Initialize weights for the custom operator to match PyTorch's ConvTranspose3d and BatchNorm3d
        # We need to extract the parameters from the initialized layers
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        
    def forward(self, x):
        # Get parameters from PyTorch layers
        weight = self.conv_transpose.weight
        bias = self.conv_transpose.bias
        running_mean = self.batch_norm.running_mean
        running_var = self.batch_norm.running_var
        gamma = self.batch_norm.weight
        beta = self.batch_norm.bias
        
        stride_d, stride_h, stride_w = self.stride if isinstance(self.stride, tuple) else (self.stride, self.stride, self.stride)

            bias,
            running_mean,
            running_var,
            gamma,
            beta,
            stride_d, stride_h, stride_w,

        )
