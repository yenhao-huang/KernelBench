import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Define the custom CUDA kernel for fused matmul, scale, and residual addition
# This kernel performs: out = (x @ W^T) * scaling + x_residual
# Note: The original model does: x = matmul(x), then x = x * scale, then x = x + original_x
# Since original_x is a clone of the pre-scaled matmul output, this is effectively:
# out = (x @ W^T) * scaling + (x @ W^T)
#     = (x @ W^T) * (scaling + 1.0)
# However, to strictly follow the "replace operators" instruction and maintain flexibility,
# we can implement a fused kernel that takes x, weight, bias (if any, here none), scaling factor,
# and performs the matmul, scaling, and addition in one go if possible, or just optimize the matmul.
# Given the simplicity, a fused GEMM + Scale + Add is ideal.

fused_ops_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <stdio.h>

// Kernel for: out[i] = (sum_k x[i,k] * w[j,k]) * scale + sum_k x[i,k] * w[j,k]
// Wait, the original code is:
// 1. x_new = x @ W.T
// 2. original_x = x_new.clone()
// 3. x_new = x_new * scale
// 4. x_new = x_new + original_x
// So: x_final = (x @ W.T) * scale + (x @ W.T) = (x @ W.T) * (scale + 1.0)

// Let's implement a fused GEMM that applies the combined scaling factor directly.
// This avoids storing intermediate tensors and reduces memory bandwidth.

__global__ void fused_matmul_scale_add_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w, // Transposed weight matrix: [out_features, in_features]
    float* __restrict__ out,
    int batch_size,
    int in_features,
    int out_features,
    float combined_scale
) {
    int row = blockIdx.y * blockDim.y + threadIdx.y;
    int col = blockIdx.x * blockDim.x + threadIdx.x;

    if (row < batch_size && col < out_features) {
        float sum = 0.0f;
        // Unroll or loop over in_features
        for (int k = 0; k < in_features; ++k) {
            sum += x[row * in_features + k] * w[col * in_features + k];
        }
        out[row * out_features + col] = sum * combined_scale;
    }
}

torch::Tensor fused_matmul_scale_add_cuda(
    torch::Tensor x,
    torch::Tensor w,
    float scaling_factor
) {
    // x: [batch_size, in_features]
    // w: [out_features, in_features] (already transposed from nn.Linear's weight which is [out, in])
    
    auto batch_size = x.size(0);
    auto in_features = x.size(1);
    auto out_features = w.size(0);

    auto out = torch::empty({batch_size, out_features}, x.options());

    const int block_x = 32;
    const int block_y = 32;
    
    dim3 block(block_x, block_y);
    dim3 grid((out_features + block_x - 1) / block_x, 
              (batch_size + block_y - 1) / block_y);

    float combined_scale = scaling_factor + 1.0f;

    fused_matmul_scale_add_kernel<<<grid, block>>>(
        x.data_ptr<float>(),
        w.data_ptr<float>(),
        out.data_ptr<float>(),
        batch_size,
        in_features,
        out_features,
        combined_scale
    );

    return out;
}
"""

fused_ops_cpp_source = (
    "torch::Tensor fused_matmul_scale_add_cuda(torch::Tensor x, torch::Tensor w, float scaling_factor);"
)

# Compile the inline CUDA code
fused_ops = load_inline(
    name="fused_ops",
    cpp_sources=fused_ops_cpp_source,
    cuda_sources=fused_ops_source,
    functions=["fused_matmul_scale_add_cuda"],
    verbose=False,
    extra_cflags=["-O3"],
    extra_ldflags=[""]
)


class ModelNew(nn.Module):
    """
    Optimized model using a custom fused CUDA operator for matmul, scaling, and residual addition.
    """
    def __init__(self, in_features, out_features, scaling_factor):
        super(ModelNew, self).__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.scaling_factor = scaling_factor
        
        # Initialize weight manually to match nn.Linear behavior
        # nn.Linear uses Kaiming uniform initialization by default for weights
        # We will replicate this or just use a standard init. 
        # For correctness, we need the same weight initialization as the original model if we want identical outputs.
        # However, the prompt asks to optimize the architecture. The weights are parameters.
        # We store the weight tensor here.
        self.weight = nn.Parameter(torch.empty(out_features, in_features))
        nn.init.kaiming_uniform_(self.weight, a=0, mode='fan_in', nonlinearity='linear')

    def forward(self, x):
        """
        Forward pass using the fused CUDA operator.
        
        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, in_features).
            
        Returns:
            torch.Tensor: Output tensor of shape (batch_size, out_features).
        """
        # The original model does:
        # 1. x = self.matmul(x) -> x @ W.T + bias (bias is None here)
        # 2. original_x = x.clone().detach()
        # 3. x = x * self.scaling_factor
        # 4. x = x + original_x
        # Result: x_final = (x @ W.T) * scaling + (x @ W.T) = (x @ W.T) * (scaling + 1.0)
        
        return fused_ops.fused_matmul_scale_add_cuda(x, self.weight, self.scaling_factor)


def get_inputs():

    a = torch.rand(16384, 4096).cuda()
    return [a]

    # randomly generate tensors required for initialization based on the model architecture
    return [4096, 4096, 0.5]
