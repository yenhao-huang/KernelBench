    float sum = 0.0f;
    
    // Determine the range of input coordinates that contribute to this output pixel
    // For transposed convolution, we iterate over the kernel and source input positions

                    sum += val * w_val;
                }
            }
        }
    }

        sum += bias[c_out];
    }

    if (sum < 0.0f) {

// Kernel for GroupNorm (simplified, assuming channels are grouped)
__global__ void group_norm_kernel(
    const float* input,       // [N, C, D, H, W]
    float* output,            // [N, C, D, H, W]
    const float* weight,      // [C] (gamma)
    const float* bias,        // [C] (beta)
    int N, int C, int D, int H, int W,
    int num_groups,
    float eps) 
{
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    long long total_elements = (long long)N * C * D * H * W;

    if (idx >= total_elements) return;

    int w = idx % W;
    int h = (idx / W) % H;
    int d = (idx / (W * H)) % D;
    int c = (idx / (W * H * D)) % C;
    int n = idx / (W * H * D * C);

    // Calculate group index for this channel
    int channels_per_group = C / num_groups;
    int group_idx = c / channels_per_group;
    
    // We need to compute mean and variance over all elements in the same group across N, D, H, W for each spatial position?
    // No, GroupNorm computes mean/variance per sample (N) and per channel group.
    // Specifically: For each n, g, d, h, w, we look at all channels c' in group g.
    // Mean = mean(input[n, c', d, h, w] for c' in group g)
    // Var = var(input[n, c', d, h, w] for c' in group g)
    
    float sum = 0.0f;
    float sum_sq = 0.0f;
    int count = channels_per_group;

    for (int k = 0; k < count; ++k) {
        int c_k = group_idx * channels_per_group + k;
        float val = input[INDEX(n, c_k, d, h, w, C, D, H, W)];
        sum += val;
        sum_sq += val * val;
    }

    float mean = sum / count;
    float var = (sum_sq / count) - (mean * mean);
    // Add epsilon for numerical stability
    float inv_std = 1.0f / sqrtf(var + eps);

    // Normalize current channel
    float val = input[INDEX(n, c, d, h, w, C, D, H, W)];
    float normalized = (val - mean) * inv_std;

    // Apply scale and shift
    if (weight != nullptr && bias != nullptr) {
        output[INDEX(n, c, d, h, w, C, D, H, W)] = normalized * weight[c] + bias[c];
    } else {
        output[INDEX(n, c, d, h, w, C, D, H, W)] = normalized;
    }
}

class ModelNew(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, groups, bias=False) -> None:
        super(ModelNew, self).__init__()

        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        self.groups = groups
        self.bias = bias

        # In a real scenario, these would be loaded from a pretrained model or trained.
        # Here we initialize them randomly for the sake of the example running.
        k = 1.0 / (in_channels * kernel_size[0] * kernel_size[1] * kernel_size[2]) if isinstance(kernel_size, tuple) else 1.0 / (in_channels * kernel_size**3)

        self.weight = nn.Parameter(torch.randn(out_channels, in_channels, kernel_size, kernel_size, kernel_size) * k)
        
        if bias:
            self.bias = nn.Parameter(torch.zeros(out_channels))
        else:
            self.register_buffer('bias', torch.zeros(0))
            
        # GroupNorm parameters (for completeness, though we might not fuse them in this specific kernel implementation above)
        # We will use standard PyTorch GroupNorm for the final step to ensure correctness and simplicity of parameter passing
        self.group_norm = nn.GroupNorm(num_groups=groups, num_channels=out_channels)

    def forward(self, x):
        """
        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, in_channels, D, H, W).

        Returns:
            torch.Tensor: Output tensor of shape (batch_size, out_channels, D, H, W).
        """

        # For this example, we assume standard stride=1 and pad=0.

        if isinstance(k, int):
            kD, kH, kW = k, k, k
        else:
            kD, kH, kW = k
