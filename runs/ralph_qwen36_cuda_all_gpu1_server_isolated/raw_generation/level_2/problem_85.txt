torch::Tensor fused_scale_maxpool_clamp_cuda(
    torch::Tensor input,
    torch::Tensor scale_param,
    int maxpool_kernel_size,
    float clamp_min,
    float clamp_max
) {

    auto channels = input.size(1);
    auto height = input.size(2);
    auto width = input.size(3);

    int out_w = width / maxpool_kernel_size;
    int out_h = height / maxpool_kernel_size;
    
    // Check divisibility
    if (width % maxpool_kernel_size != 0 || height % maxpool_kernel_size != 0) {
        throw std::runtime_error("Input dimensions must be divisible by maxpool_kernel_size for this custom kernel.");
    }

    auto output = torch::empty({batch_size, channels, out_h, out_w}, input.options());

    int total_elements = batch_size * channels * out_h * out_w;
    if (total_elements == 0) return output;

    const int block_size = 256;
    const int num_blocks = (total_elements + block_size - 1) / block_size;

    fused_scale_maxpool_clamp_kernel<<<num_blocks, block_size>>>(
        input.data_ptr<float>(),
        scale_param.data_ptr<float>(),
        output.data_ptr<float>(),
        batch_size,
        channels,
        height,
        width,
        maxpool_kernel_size,
        clamp_min,
        clamp_max
    );

    return output;
}
