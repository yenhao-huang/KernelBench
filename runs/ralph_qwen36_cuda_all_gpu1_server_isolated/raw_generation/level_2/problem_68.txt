import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Define the custom CUDA kernel for fused matmul + min + sub
# This kernel performs: out[i] = min(linear(x)[i], constant) - constant
# Which is equivalent to: out[i] = min(linear(x)[i] - constant, 0.0)
# However, to be safe and explicit with the original logic:
# 1. Compute y = Wx + b (standard matmul)
# 2. z = min(y, c)
# 3. out = z - c

fused_ops_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <stdio.h>

// Kernel for matrix multiplication: C = A * B^T
// A is (M, K), B is (N, K), C is (M, N)
__global__ void matmul_kernel(const float* __restrict__ A, const float* __restrict__ B, float* __restrict__ C, int M, int N, int K) {
    int row = blockIdx.y * blockDim.y + threadIdx.y;
    int col = blockIdx.x * blockDim.x + threadIdx.x;

    if (row < M && col < N) {
        float sum = 0.0f;
        for (int k = 0; k < K; ++k) {
            sum += A[row * K + k] * B[col * K + k]; // B is stored as (N, K), so B[col][k]
        }
        C[row * N + col] = sum;
    }
}

// Kernel for element-wise min and subtract
__global__ void min_sub_kernel(const float* __restrict__ x, const float constant, float* __restrict__ out, int size) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < size) {
        float val = x[idx];
        float min_val = fminf(val, constant);
        out[idx] = min_val - constant;
    }
}

torch::Tensor fused_ops_cuda(torch::Tensor x, torch::Tensor weight, torch::Tensor bias, float constant) {
    // x: (batch_size, in_features)
    // weight: (out_features, in_features)
    // bias: (out_features,)
    
    int batch_size = x.size(0);
    int in_features = x.size(1);
    int out_features = weight.size(0);

    auto options = torch::TensorOptions().dtype(torch::kFloat32).device(x.device());
    torch::Tensor intermediate = torch::empty({batch_size, out_features}, options);

    // Perform Matmul: (B, K) * (N, K)^T -> (B, N)
    // Note: PyTorch Linear does x @ W^T + b. 
    // Here weight is (out, in). So we want x @ weight.T
    // matmul_kernel expects A=(M,K), B=(N,K), C=(M,N) where C = A * B^T
    // So we pass x as A, weight as B.

    const int block_size_x = 32;
    const int block_size_y = 8;
    
    dim3 threads(block_size_x, block_size_y);
    dim3 blocks((out_features + block_size_x - 1) / block_size_x, 
                (batch_size + block_size_y - 1) / block_size_y);

    matmul_kernel<<<blocks, threads>>>(x.data_ptr<float>(), weight.data_ptr<float>(), intermediate.data_ptr<float>(), batch_size, out_features, in_features);

    // Add bias
    torch::Tensor biased = intermediate + bias.unsqueeze(0);

    // Perform min and subtract
    int total_elements = batch_size * out_features;
    auto result = torch::empty_like(biased);

    const int block_size_elem = 256;
    const int num_blocks_elem = (total_elements + block_size_elem - 1) / block_size_elem;

    min_sub_kernel<<<num_blocks_elem, block_size_elem>>>(biased.data_ptr<float>(), constant, result.data_ptr<float>(), total_elements);

    return result;
}
"""

fused_ops_cpp_source = (
    "torch::Tensor fused_ops_cuda(torch::Tensor x, torch::Tensor weight, torch::Tensor bias, float constant);"
)

# Compile the inline CUDA code
fused_ops = load_inline(
    name="fused_ops",
    cpp_sources=fused_ops_cpp_source,
    cuda_sources=fused_ops_source,
    functions=["fused_ops_cuda"],
    verbose=True,
    extra_cflags=["-O3"],
    extra_ldflags=["-lcudart"]
)


class ModelNew(nn.Module):
    """
    Optimized model using custom CUDA operators for Matmul, Min, and Subtraction.
    """
    def __init__(self, in_features, out_features, constant):
        super(ModelNew, self).__init__()
        # Initialize weights and biases manually to match nn.Linear behavior
        # nn.Linear initializes weight with Kaiming uniform and bias with zeros
        stdv = 1. / (in_features ** 0.5)
        self.weight = nn.Parameter(torch.empty(out_features, in_features).uniform_(-stdv, stdv))
        self.bias = nn.Parameter(torch.zeros(out_features))
        self.constant = constant

    def forward(self, x):
        # Use custom fused operator
        return fused_ops.fused_ops_cuda(x, self.weight, self.bias, self.constant)
