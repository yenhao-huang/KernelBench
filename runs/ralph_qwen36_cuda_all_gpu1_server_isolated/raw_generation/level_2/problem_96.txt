// Simplified Kernel: Each thread computes one element of the final output [B, C, 1, 1, 1]
// It iterates over all spatial positions to compute MaxPool then Average.
// This is inefficient for large spatial dims but simple to implement.
// For D=31, H=63, W=63, the number of MaxPool cells is ~14k. 
// Iterating 14k times per thread (2048 threads) is 28M ops. This is acceptable for a single forward pass optimization if it saves memory bandwidth.

__global__ void fused_scale_maxpool_gavgpool_clamp_kernel(
    const float* input, 
    float* output, 
    float scale, 
    float clamp_min, 
    float clamp_max,
    int batch_size,
    int channels,
    int d_in, int h_in, int w_in,
    int d_mp, int h_mp, int w_mp
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total_elements = batch_size * channels;
    
    if (idx >= total_elements) return;
    
    int b = idx / channels;
    int c = idx % channels;
    
    const float* channel_data = input + (b * channels + c) * d_in * h_in * w_in;
    
    float sum_maxes = 0.0f;
    int num_max_cells = d_mp * h_mp * w_mp;
    
    // Iterate over all MaxPool cells
    for (int dz = 0; dz < d_mp; ++dz) {
        int z_start = dz * 2;
        for (int dh = 0; dh < h_mp; ++dh) {
            int y_start = dh * 2;
            for (int dw = 0; dw < w_mp; ++dw) {
                int x_start = dw * 2;
                
                float local_max = -FLT_MAX;
                
                // Unroll the 2x2x2 window manually for performance
                // z loop
                for (int k = 0; k < 2; ++k) {
                    int z = z_start + k;
                    if (z >= d_in) break;
                    
                    // y loop
                    for (int l = 0; l < 2; ++l) {
                        int y = y_start + l;
                        if (y >= h_in) break;
                        
                        // x loop
                        for (int m = 0; m < 2; ++m) {
                            int x = x_start + m;
                            if (x >= w_in) break;
                            
                            float val = channel_data[z * h_in * w_in + y * w_in + x];
                            if (val > local_max) {
                                local_max = val;
                            }
                        }
                    }
                }
                
                sum_maxes += local_max;
            }
        }
    }
    
    // Global Average Pool: Sum of MaxPool outputs / Number of MaxPool cells
    float avg_val = sum_maxes / (float)num_max_cells;
    
    // Scale was applied before MaxPool, so it's already in the input data.
    // Wait, the input to this kernel is the output of ConvTranspose3d * scale?
    // In the original model: x = conv(x); x = x * scale; x = maxpool(x)...
    // So we should pass the scaled input to this kernel.
    
    // Clamp
    if (avg_val < clamp_min) avg_val = clamp_min;
    if (avg_val > clamp_max) avg_val = clamp_max;
    
    output[idx] = avg_val;
}

torch::Tensor fused_post_process_cuda(
    torch::Tensor input, 
    float scale, 
    float clamp_min, 
    float clamp_max
) {
    // Input shape: [B, C, D, H, W]
    auto batch_size = input.size(0);
    auto channels = input.size(1);
    auto d_in = input.size(2);
    auto h_in = input.size(3);
    auto w_in = input.size(4);
    
    // Calculate MaxPool output dimensions
    int d_mp = (d_in - MAX_POOL_KERNEL_SIZE) / 2 + 1;
    int h_mp = (h_in - MAX_POOL_KERNEL_SIZE) / 2 + 1;
    int w_mp = (w_in - MAX_POOL_KERNEL_SIZE) / 2 + 1;
    
    auto output = torch::empty({batch_size, channels}, input.options());
    
    const int block_size = 256;
    const int total_elements = batch_size * channels;
    const int num_blocks = (total_elements + block_size - 1) / block_size;
    
    fused_scale_maxpool_gavgpool_clamp_kernel<<<num_blocks, block_size>>>(
        input.data_ptr<float>(),
        output.data_ptr<float>(),
        scale,

        clamp_max,
        batch_size,
        channels,
        d_in, h_in, w_in,
        d_mp, h_mp, w_mp
    );
    
    return output;
}

    Optimized Model using custom CUDA operators for post-processing.
