__global__ void gemm_kernel(const float* A, const float* B, const float* bias, float* C, 
                            int M, int N, int K) {

    const int BLOCK_SIZE = 32;
    __shared__ float As[BLOCK_SIZE][BLOCK_SIZE];
    __shared__ float Bs[BLOCK_SIZE][BLOCK_SIZE];

    }

// Sigmoid Kernel
__global__ void sigmoid_kernel(const float* input, float* output, int size) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < size) {
        float x = input[idx];
        // Numerically stable sigmoid: 1 / (1 + exp(-x))
        // For large positive x, exp(-x) is small. For large negative x, exp(-x) is huge.
        // Better: if x >= 0, 1/(1+exp(-x)); else exp(x)/(1+exp(x))
        float y;
        if (x >= 0) {
            y = 1.0f / (1.0f + expf(-x));
        } else {
            float ex = expf(x);
            y = ex / (1.0f + ex);
        }
        output[idx] = y;
    }
}

// LogSumExp Kernel
// Computes log(sum(exp(x))) along dim 1 for each row.
// Input: [M, N], Output: [M]
__global__ void logsumexp_kernel(const float* input, float* output, int M, int N) {
    int row = blockIdx.x * blockDim.x + threadIdx.x;
    if (row < M) {
        // Find max for numerical stability
        float max_val = -INFINITY;
        for (int j = 0; j < N; ++j) {
            if (input[row * N + j] > max_val) {
                max_val = input[row * N + j];
            }
        }

        // Sum exp(x - max)
        float sum_exp = 0.0f;
        for (int j = 0; j < N; ++j) {
            sum_exp += expf(input[row * N + j] - max_val);
        }

        output[row] = logf(sum_exp) + max_val;
    }
}

torch::Tensor gemm_sigmoid_cuda(torch::Tensor A, torch::Tensor B, torch::Tensor bias) {

                        // The kernel expects B to be [N, K] where B[j][k] is accessed.
                        // A @ B^T means C[i][j] = sum_k A[i][k] * B[j][k].

    auto out = torch::empty({M, N}, torch::dtype(torch::kFloat32).device(torch::kCUDA));

}

class ModelNew(nn.Module):
    """
    Optimized Model using custom CUDA operators.
    Fuses Linear1 and Sigmoid into a single GEMM+Sigmoid kernel.
    Uses optimized GEMM for Linear2.
    Uses optimized LogSumExp reduction.
    """
    def __init__(self, input_size, hidden_size, output_size):
        super(ModelNew, self).__init__()
        self.input_size = input_size
        self.hidden_size = hidden_size
        self.output_size = output_size
        
        # Initialize weights and biases manually to ensure they are on CUDA
        # We need to register them as parameters so they can be trained
        self.register_parameter('weight1', nn.Parameter(torch.randn(output_size, input_size))) # Wait, Linear1 is input->hidden
        # Correction:
        # linear1: input_size -> hidden_size. Weight shape: [hidden_size, input_size]
        # linear2: hidden_size -> output_size. Weight shape: [output_size, hidden_size]
        
        self.weight1 = nn.Parameter(torch.randn(hidden_size, input_size))
        self.bias1 = nn.Parameter(torch.zeros(hidden_size))
        self.weight2 = nn.Parameter(torch.randn(output_size, hidden_size))
        self.bias2 = nn.Parameter(torch.zeros(output_size))

    def forward(self, x):
        # x: [batch_size, input_size]
        
        # Step 1: Linear1 + Sigmoid fused
        # Custom kernel expects A=[M, K], B=[N, K] for A @ B^T
        # PyTorch Linear: out = x @ W.T + b. W is [out_features, in_features].
        # So W1 is [hidden_size, input_size].
        # x is [batch_size, input_size].
        # We want to compute x @ W1^T.
        # Our kernel gemm_sigmoid_cuda(A, B, bias) computes A @ B^T + bias.
        # So we pass x as A, weight1 as B.

        # Step 2: Linear2
        # h is [batch_size, hidden_size]
        # weight2 is [output_size, hidden_size]
        # We want h @ W2^T + b2
        
        z = custom_ops.gemm_cuda(h, self.weight2, self.bias2)
        
        # Step 3: LogSumExp over dim 1 (features)
        # z is [batch_size, output_size]
        # logsumexp expects input [M, N], returns [M]
        
        out = custom_ops.logsumexp_cuda(z)
        
        return out
