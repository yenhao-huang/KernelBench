import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Define the custom CUDA kernel that fuses:
# 1. Transposed Convolution (using cuDNN via PyTorch's native call for correctness, 
#    but we will wrap the subsequent element-wise ops in a custom kernel for fusion).
# However, since transposed conv is complex to implement from scratch efficiently without libraries,
# and the prompt allows replacing "some" operators, we focus on fusing the element-wise operations:
# Add -> Min(0.0) -> GELU -> Multiply.
# This is a classic candidate for kernel fusion to reduce memory bandwidth pressure.

custom_ops_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cmath>

// Helper for GELU approximation or exact calculation
__device__ inline float gelu(float x) {
    return 0.5f * x * (1.0f + tanhf(0.7978845608028654f * (x + 0.044715f * x * x * x)));
}

__global__ void fused_ops_kernel(const float* input, const float* bias_or_scalar_add, 
                                 float out_add_val, float min_val, float multiply_val, 
                                 float* output, int size) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < size) {
        // 1. Add value
        float val = input[idx] + out_add_val;
        
        // 2. Min with 0.0 (ReLU-like behavior for negative values, but keeping positive as is? 
        //    Wait: torch.min(x, 0.0) means element-wise minimum. 
        //    If x > 0, min is 0. If x < 0, min is x.
        //    This is effectively ReLU's inverse or just clamping at 0 from above?
        //    No: min(x, 0) -> if x=5, min(5,0)=0. If x=-5, min(-5,0)=-5.
        //    So it zeros out positive values and keeps negative values.
        if (val > min_val) {
            val = min_val;
        }
        
        // 3. GELU
        val = gelu(val);
        
        // 4. Multiply
        output[idx] = val * multiply_val;
    }
}

torch::Tensor fused_ops_cuda(torch::Tensor x, float add_value, float multiply_value) {
    auto size = x.numel();
    auto out = torch::empty_like(x);
    
    if (size == 0) return out;

    const int block_size = 256;
    const int num_blocks = (size + block_size - 1) / block_size;
    
    // We pass add_value and multiply_value as constants to the kernel
    fused_ops_kernel<<<num_blocks, block_size>>>(
        x.data_ptr<float>(), 
        nullptr, // Not used directly in this simplified signature, we use scalar args
        add_value, 
        0.0f, 
        multiply_value, 
        out.data_ptr<float>(), 
        size
    );
    
    return out;
}
"""

custom_ops_cpp_source = (
    "torch::Tensor fused_ops_cuda(torch::Tensor x, float add_value, float multiply_value);"
)

# Compile the inline CUDA code
fused_ops_lib = load_inline(
    name="fused_ops",
    cpp_sources=custom_ops_cpp_source,
    cuda_sources=custom_ops_source,
    functions=["fused_ops_cuda"],
    verbose=False,
    extra_cflags=["-O2"],
    extra_ldflags=[""]
)


class ModelNew(nn.Module):
    """
    Optimized Model that performs a transposed convolution, then fuses 
    add, min(0.0), gelu, and multiply into a single custom CUDA kernel.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, add_value, multiply_value):
        super(ModelNew, self).__init__()
        self.conv_transpose = nn.ConvTranspose2d(in_channels, out_channels, kernel_size, stride=stride)
        self.add_value = add_value
        self.multiply_value = multiply_value

    def forward(self, x):
        # 1. Transposed Convolution (Standard PyTorch op, hard to optimize from scratch without cuDNN)
        x = self.conv_transpose(x)
        
        # 2. Fused Element-wise Operations: Add -> Min(0.0) -> GELU -> Multiply
        # Using the custom CUDA kernel for better performance by reducing memory traffic
        x = fused_ops_lib.fused_ops_cuda(x, self.add_value, self.multiply_value)
        
        return x

# Re-defining get_inputs and get_init_inputs to match the required interface if needed, 
# though the prompt asks for the ModelNew code block. 
# We include them here to ensure the file is self-contained as per "Output the new code".

batch_size = 128
in_channels = 64
out_channels = 128
height, width = 64, 64
kernel_size = 4
stride = 2
add_value = 0.5
multiply_value = 2.0

def get_inputs():
    return [torch.rand(batch_size, in_channels, height, width)]

def get_init_inputs():
    return [in_channels, out_channels, kernel_size, stride, add_value, multiply_value]
