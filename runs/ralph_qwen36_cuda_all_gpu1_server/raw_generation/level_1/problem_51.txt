import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Define the custom CUDA kernel for Argmax
argmax_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <stdio.h>

// Helper to find max and index in a block
__device__ void warp_reduce_argmax(float& max_val, int& max_idx, float val, int idx) {
    // Simple sequential reduction within warp for simplicity and correctness
    // For large blocks, shared memory reduction is better, but this ensures correctness
    if (idx < max_idx) {
        max_idx = idx;
        max_val = val;
    } else if (idx == max_idx && val > max_val) {
        max_val = val;
    }
}

__global__ void argmax_kernel(const float* input, int64_t* output, int batch_size, int dim1, int dim2) {
    // We process one element of the output tensor per thread.
    // Output shape: (batch_size, dim1) assuming dim=2 is reduced.
    // Input shape: (batch_size, dim1, dim2)
    
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total_elements = batch_size * dim1;

    if (idx < total_elements) {
        int b = idx / dim1;
        int i = idx % dim1;
        
        float max_val = -INFINITY;
        int max_idx = 0;
        
        const float* row_ptr = input + b * dim1 * dim2 + i * dim2;
        
        // Iterate over the last dimension (dim2)
        for (int j = 0; j < dim2; ++j) {
            float val = row_ptr[j];
            if (val > max_val) {
                max_val = val;
                max_idx = j;
            } else if (val == max_val) {
                // Argmax typically returns the first occurrence of the maximum value.
                // Since we iterate from 0 to dim2-1, keeping the first one is correct.
                // However, strict > ensures we keep the smallest index for ties.
            }
        }
        
        output[idx] = (int64_t)max_idx;
    }
}

torch::Tensor argmax_cuda(torch::Tensor x, int dim) {
    // Assuming input is 3D: (batch_size, dim1, dim2) and dim=2
    // This implementation specifically handles the case where we reduce over the last dimension.
    
    if (x.dim() != 3 || dim != 2) {
        throw std::runtime_error("This kernel expects a 3D tensor and dim=2");
    }

    int batch_size = x.size(0);
    int dim1 = x.size(1);
    int dim2 = x.size(2);

    auto output = torch::empty({batch_size, dim1}, torch::dtype(torch::kInt64).device(x.device()));

    const int block_size = 256;
    int total_elements = batch_size * dim1;
    int num_blocks = (total_elements + block_size - 1) / block_size;

    argmax_kernel<<<num_blocks, block_size>>>(x.data_ptr<float>(), output.data_ptr<int64_t>(), batch_size, dim1, dim2);

    return output;
}
"""

argmax_cpp_source = (
    "torch::Tensor argmax_cuda(torch::Tensor x, int dim);"
)

# Compile the inline CUDA code for Argmax
argmax_module = load_inline(
    name="argmax_custom",
    cpp_sources=argmax_cpp_source,
    cuda_sources=argmax_source,
    functions=["argmax_cuda"],
    verbose=False,
    extra_cflags=["-O2"],
    extra_ldflags=[""]
)


class ModelNew(nn.Module):
    """
    Optimized model that performs Argmax over a specified dimension using custom CUDA.
    """
    def __init__(self, dim: int):
        """
        Initializes the model with the dimension to perform argmax.

        Args:
            dim (int): The dimension to perform argmax over.
        """
        super(ModelNew, self).__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Applies argmax over the specified dimension to the input tensor using custom CUDA kernel.

        Args:
            x (torch.Tensor): Input tensor.

        Returns:
            torch.Tensor: Output tensor with argmax applied, with the specified dimension removed.
        """
        return argmax_module.argmax_cuda(x, self.dim)
