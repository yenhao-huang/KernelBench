import torch
import torch.nn as nn
import torch.nn.functional as F
from torch.utils.cpp_extension import load_inline

# The optimization strategy is to fuse the GroupNorm and Hardtanh operations.
# GEMM (Linear) is best left to cuBLAS/cuDNN for maximum performance.
# However, GroupNorm involves calculating mean and variance per group, 
# followed by normalization, scaling (gamma), shifting (beta), and finally Hardtanh.
# We can fuse GroupNorm + Hardtanh into a single kernel to reduce memory bandwidth.

fused_ops_cuda_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math.h>

__global__ void groupnorm_hardtanh_kernel(
    const float* __restrict__ input,
    const float* __restrict__ gamma,
    const float* __restrict__ beta,
    float* __restrict__ output,
    int batch_size,
    int num_groups,
    int channels_per_group,
    float eps,
    float h_min,
    float h_max) 
{
    // Each block handles one (batch_idx, group_idx)
    int b = blockIdx.y;
    int g = blockIdx.x;
    
    if (b >= batch_size || g >= num_groups) return;

    int group_offset = g * channels_per_group;
    int total_elements_in_group = channels_per_group;
    
    // We use a simple reduction for mean and variance within the block
    // For high performance, one would use shared memory and tree reduction,
    // but for simplicity and correctness in this inline example, we use a loop.
    // Note: In a production environment, we'd use a more optimized reduction.
    
    extern __shared__ float shared_mem[];
    float* s_data = shared_mem; // This is not quite right for a simple loop, 
                                // let's use a more robust approach.
}

// Since writing a fully optimized shared-memory reduction kernel in an inline block 
// is error-prone, we will implement a kernel that performs the element-wise 
// transformation after the mean/var are computed, or use a two-pass approach.
// To ensure stability and correctness, we'll implement the fused kernel 
// assuming mean and var are provided, or compute them efficiently.

__global__ void fused_gn_ht_kernel(
    const float* __restrict__ x,
    const float* __restrict__ gamma,
    const float* __restrict__ beta,
    float* __restrict__ out,
    int batch_size,
    int num_groups,
    int channels_per_group,
    int num_channels,
    float eps,
    float h_min,
    float h_max) 
{
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total_elements = batch_size * num_channels;
    if (idx >= total_elements) return;

    int b = idx / num_channels;
    int c = idx % num_channels;
    
    int g = c / channels_per_group;
    int group_start_c = g * channels_per_group;
    
    // To avoid race conditions and heavy synchronization, we compute mean/var 
    // per group. For a single kernel, we'll use a simple approach:
    // This kernel is designed to be called after mean/var are computed, 
    // OR we can compute mean/var in a separate kernel.
    // Let's do: 1. Compute Mean/Var (Standard) 2. Fused GN + Hardtanh.
}
"""

# To provide a robust and working solution, we will fuse GroupNorm's 
# element-wise part (scaling, shifting, and Hardtanh) with the 
# output of the GEMM.

fused_gn_ht_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void fused_gn_ht_kernel(
    const float* __restrict__ x,
    const float* __restrict__ mean,
    const float* __restrict__ var,
    const float* __restrict__ gamma,
    const float* __restrict__ beta,
    float* __restrict__ out,
    int batch_size,
    int num_groups,
    int channels_per_group,
    int num_channels,
    float eps,
    float h_min,
    float h_max) 
{
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total_elements = batch_size * num_channels;
    if (idx >= total_elements) return;

    int b = idx / num_channels;
    int c = idx % num_channels;
    int g = c / channels_per_group;

    int mean_idx = b * num_groups + g;
    float m = mean[mean_idx];
    float v = var[mean_idx];
    
    float val = (x[idx] - m) / sqrtf(v + eps);
    val = val * gamma[c] + beta[c];
    
    // Hardtanh
    if (val < h_min) val = h_min;
    else if (val > h_max) val = h_max;
    
    out[idx] = val;
}

torch::Tensor fused_gn_ht_cuda(
    torch::Tensor x,
    torch::Tensor mean,
    torch::Tensor var,
    torch::Tensor gamma,
    torch::Tensor beta,
    float eps,
    float h_min,
    float h_max) 
{
    int batch_size = x.size(0);
    int num_channels = x.size(1);
    int num_groups = mean.size(1);
    int channels_per_group = num_channels / num_groups;
    
    auto out = torch::empty_like(x);
    int total_elements = x.numel();
    int block_size = 256;
    int num_blocks = (total_elements + block_size - 1) / block_size;

    fused_gn_ht_kernel<<<num_blocks, block_size>>>(
        x.data_ptr<float>(),
        mean.data_ptr<float>(),
        var.data_ptr<float>(),
        gamma.data_ptr<float>(),
        beta.data_ptr<float>(),
        out.data_ptr<float>(),
        batch_size,
        num_groups,
        channels_per_group,
        num_channels,
        eps,
        h_min,
        h_max
    );

    return out;
}
"""

fused_gn_ht_cpp_source = (
    "torch::Tensor fused_gn_ht_cuda(torch::Tensor x, torch::Tensor mean, torch::Tensor var, "
    "torch::Tensor gamma, torch::Tensor beta, float eps, float h_min, float h_max);"
)

fused_ops = load_inline(
    name="fused_gn_ht",
    cpp_sources=fused_gn_ht_cpp_source,
    cuda_sources=fused_gn_ht_source,
    functions=["fused_gn_ht_cuda"],
    verbose=False,
)

class ModelNew(nn.Module):
    def __init__(self, in_features, out_features, num_groups, hardtanh_min, hardtanh_max):
        super(ModelNew, self).__init__()
        self.gemm = nn.Linear(in_features, out_features)
        self.num_groups = num_groups
        self.eps = 1e-5 # Default GroupNorm eps
        self.hardtanh_min = hardtanh_min
        self.hardtanh_max = hardtanh_max
        
        # We keep the parameters for GroupNorm
        self.gamma = nn.Parameter(torch.ones(out_features))
        self.beta = nn.Parameter(torch.zeros(out_features))
        self.fused_ops = fused_ops

    def forward(self, x):
        # 1. GEMM
        x = self.gemm(x)
        
        # 2. GroupNorm Mean and Var calculation
        # We use PyTorch's optimized implementation for mean/var to ensure correctness
        # and then fuse the element-wise part.
        batch_size, num_channels = x.shape
        x_reshaped = x.view(batch_size, self.num_groups, -1)
        
        mean = x_reshaped.mean(dim=2)
        var = x_reshaped.var(dim=2, unbiased=False)
        
        # 3. Fused GN (scaling/shifting) + Hardtanh
        # We pass the pre-computed mean and var to our custom kernel
        return self.fused_ops.fused_gn_ht_cuda(
            x, 
            mean, 
            var, 
            self.gamma, 
            self.beta, 
            self.eps, 
            self.hardtanh_min, 
            self.hardtanh_max
        )
