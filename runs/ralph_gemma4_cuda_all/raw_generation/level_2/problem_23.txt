# Custom CUDA kernel to fuse GroupNorm and the global mean reduction.
# GroupNorm: (x - mean_group) / sqrt(var_group + eps)
# Then we take the mean of the resulting tensor across all dimensions except batch.
# Since the mean is a linear operator, we can optimize the reduction.
# However, for simplicity and robustness, we implement a fused kernel that 
# performs GroupNorm and then uses a reduction kernel or a single pass.
# To maximize speedup, we fuse the GroupNorm calculation and the final mean reduction.

fused_ops_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math.h>

__global__ void fused_gn_mean_kernel(
    const float* __restrict__ input,
    const float* __restrict__ gamma,
    const float* __restrict__ beta,
    float* __restrict__ output,
    int batch_size,
    int num_groups,
    int channels_per_group,
    int spatial_elements, // D * H * W
    float eps) 
{
    // Each block handles one batch element
    int b = blockIdx.x;
    if (b >= batch_size) return;

    // We need to compute mean and variance for each group within the batch
    // GroupNorm parameters: 
    // input shape: [B, C, D, H, W]
    // group size: C / num_groups
    // elements per group: (C/num_groups) * D * H * W
    
    int total_channels = (channels_per_group * num_groups);
    int elements_per_group = channels_per_group * spatial_elements;
    
    // We use shared memory to accumulate sums for the mean reduction
    // But since we need to output one value per batch, we can do it in two passes 
    // or one pass with a lot of registers. 
    // Let's compute GN first, then accumulate the sum for the final mean.

    // To avoid complex synchronization, we'll compute GN and the sum for the mean in one kernel.
    // However, GN requires the mean/var of the group first.
    // So we do: 1. Compute group mean/var. 2. Compute GN and add to global sum.
    
    // For simplicity in this fused kernel, we'll use a two-step approach within the block
    // or just compute the GN and then use a separate reduction.
    // Actually, the most efficient way for a single kernel is to use shared memory.
}

// Since writing a fully optimized fused GN + Mean kernel in a single raw CUDA block 
// is highly complex due to the two-pass nature of GN, we will provide a 
// highly optimized version of the reduction part and the GN part.

// Let's implement a kernel that performs:
// 1. GroupNorm (element-wise after computing stats)
// 2. Summation for the final mean.

__global__ void gn_and_sum_kernel(
    const float* __restrict__ input,
    const float* __restrict__ gamma,
    const float* __restrict__ beta,
    float* __restrict__ output,
    int batch_size,
    int num_groups,
    int channels_per_group,
    int spatial_elements,
    float eps) 
{
    int b = blockIdx.x;
    int tid = threadIdx.x;
    int total_threads = blockDim.x;

    int total_channels = channels_per_group * num_groups;
    int elements_per_group = channels_per_group * spatial_elements;
    int total_elements_per_batch = total_channels * spatial_elements;

    // Step 1: Compute Mean and Var for each group
    // We'll use a simple approach: each block handles one batch.
    // Within the block, we use shared memory to reduce.
    
    // For the sake of this implementation, we will perform the GN 
    // and then a reduction. To keep the code clean and functional:
    // We'll use a kernel that computes the GN and stores it in a temporary buffer,
    // then we'll use a reduction. But to be "custom", we'll fuse the mean.
    
    // Actually, let's just implement the GN part efficiently and the mean part.
}
"""

# Given the complexity of writing a perfect fused kernel in a single block, 
# we will implement a kernel that performs the GroupNorm and then 
# immediately accumulates the sum for the mean.

fused_ops_cuda_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void fused_gn_mean_kernel(
    const float* __restrict__ input,
    const float* __restrict__ gamma,
    const float* __restrict__ beta,
    float* __restrict__ output,
    int batch_size,
    int num_groups,
    int channels_per_group,
    int spatial_elements,
    float eps) 
{
    int b = blockIdx.x;
    int group_idx = blockIdx.y;
    int tid = threadIdx.x;

    int total_channels = num_groups * channels_per_group;
    int elements_per_group = channels_per_group * spatial_elements;
    
    // This kernel is designed to be called per group per batch to compute stats
    // But that's inefficient. Let's use a simpler approach:
    // One block per batch.
}

// Let's use a more practical approach for the custom operator:
// A kernel that computes the GroupNorm and then the mean.
// To do this in one pass, we need to compute mean and var first.

__global__ void gn_stats_kernel(
    const float* __restrict__ input,
    float* __restrict__ group_means,
    float* __restrict__ group_vars,
    int batch_size,
    int num_groups,
    int elements_per_group,
    int total_channels,
    int spatial_elements) 
{
    int b = blockIdx.x;
    int g = blockIdx.y;
    int tid = threadIdx.x;
    int total_threads = blockDim.x;

    int offset = b * total_channels * spatial_elements + g * elements_per_group;
    
    float sum = 0;
    float sq_sum = 0;

    for (int i = tid; i < elements_per_group; i += total_threads) {
        float val = input[offset + i];
        sum += val;
        sq_sum += val * val;
    }

    // Reduction in shared memory
    extern __shared__ float sdata[];
    sdata[tid] = sum;
    sdata[tid + total_threads] = sq_sum;

    __syncthreads();

    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (tid < s) {
            sdata[tid] += sdata[tid + s];
            sdata[tid + total_threads] += sdata[tid + s + total_threads];
        }
        __syncthreads();
    }

    if (tid == 0) {
        group_means[b * num_groups + g] = sdata[0] / elements_per_group;
        float m = sdata[0] / elements_per_group;
        float var = (sdata[total_threads] / elements_per_group) - (m * m);
        group_vars[b * num_groups + g] = var;
    }
}

__global__ void gn_apply_and_mean_kernel(
    const float* __restrict__ input,
    const float* __restrict__ gamma,
    const float* __restrict__ beta,
    const float* __restrict__ group_means,
    const float* __restrict__ group_vars,
    float* __restrict__ output,
    int batch_size,
    int num_groups,
    int channels_per_group,
    int spatial_elements,
    float eps) 
{
    int b = blockIdx.x;
    int tid = threadIdx.x;
    int total_threads = blockDim.x;

    int total_channels = num_groups * channels_per_group;
    int elements_per_group = channels_per_group * spatial_elements;
    int total_elements_per_batch = total_channels * spatial_elements;

    // We'll use a reduction to compute the mean of the GN'd values
    // To keep it simple and avoid massive shared memory, we'll use a single thread 
    // to accumulate the sum for the mean after computing GN.
    
    // However, to be fast, we'll compute GN and then use a reduction.
    // For this specific task, we'll compute the GN and the sum in one kernel.
    
    // We'll use a simple loop for each thread to process its assigned elements
    // and then use atomicAdd for the final mean.
    
    float batch_sum = 0;
    
    // Each thread processes multiple elements to cover the whole batch
    for (int i = tid; i < total_elements_per_batch; i += total_threads) {
        int c = i / spatial_elements;
        int g = c / channels_per_group;
        int spatial_idx = i % spatial_elements;
        
        int input_idx = b * total_channels * spatial_elements + i;
        float m = group_means[b * num_groups + g];
        float v = group_vars[b * num_groups + g];
        
        float val = (input[input_idx] - m) / sqrtf(v + eps);
        val = val * gamma[c] + beta[c];
        
        // We don't actually need to write back the whole tensor if we only want the mean
        // But the original model returns the mean.
        // To match the original model's behavior (which might be used for gradients),
        // we'd need to write the tensor. But the return is just the mean.
        // The problem says: "Returns: torch.Tensor: Output tensor of shape (batch_size, 1)."
        // So we only need the mean.
        
        batch_sum += val;
    }

    // Use atomicAdd to accumulate the sum for the batch
    // We need to divide by total_elements_per_batch at the end.
    // To avoid precision issues with many atomicAdds, we'll use a more robust way.
    // But for a custom operator, atomicAdd is a valid starting point.
    
    // We'll use a temporary array for sums to avoid contention
    // Or just one atomicAdd per thread.
    atomicAdd(&output[b], batch_sum);
}

class ModelNew(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, num_groups):
        super(ModelNew, self).__init__()
        self.conv = nn.Conv3d(in_channels, out_channels, kernel_size)
        self.num_groups = num_groups
        self.out_channels = out_channels
        self.eps = 1e-5
        # GroupNorm uses gamma and beta. We'll extract them from a GroupNorm layer.
        self.gn = nn.GroupNorm(num_groups, out_channels)
        self.fused_ops = fused_ops

    def forward(self, x):
        x = self.conv(x)
        # We use the weights from the initialized GroupNorm layer
        gamma = self.gn.weight
        beta = self.gn.bias
        # The custom kernel handles GroupNorm + Mean reduction
        return self.fused_ops.fused_gn_mean_cuda(x, gamma, beta, self.num_groups, self.eps)
