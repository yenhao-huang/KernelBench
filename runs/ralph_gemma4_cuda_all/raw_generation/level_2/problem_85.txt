import torch
import torch.nn as nn
import torch.nn.functional as F
from torch.utils.cpp_extension import load_inline

# Custom CUDA kernel to fuse: Scale (element-wise), MaxPool2d, and Clamp
# This reduces memory bandwidth by performing multiple operations in a single pass over the data.
fused_ops_cuda_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <algorithm>

__global__ void scale_maxpool_clamp_kernel(
    const float* __restrict__ input,
    const float* __restrict__ scale,
    float* __restrict__ output,
    int batch_size,
    int channels,
    int in_h,
    int in_w,
    int out_h,
    int out_w,
    int pool_size,
    int pool_stride,
    int pool_padding,
    float clamp_min,
    float clamp_max) 
{
    // Each thread handles one output element (batch_idx, channel_idx, out_y, out_x)
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total_elements = batch_size * channels * out_h * out_w;

    if (idx < total_elements) {
        int w_out = idx % out_w;
        int h_out = (idx / out_w) % out_h;
        int c_idx = (idx / (out_w * out_h)) % channels;
        int b_idx = idx / (out_w * out_h * channels);

        float max_val = -1e38f; // Approximate negative infinity for FP32

        // Calculate the window boundaries in the input tensor
        int start_y = h_out * pool_stride - pool_padding;
        int start_x = w_out * pool_stride - pool_padding;
        int end_y = start_y + pool_size;
        int end_x = start_x + pool_size;

        for (int y = start_y; y < end_y; ++y) {
            for (int x = start_x; x < end_x; ++x) {
                if (y >= 0 && y < in_h && x >= 0 && x < in_w) {
                    // Input index: [b, c, y, x]
                    int in_idx = ((b_idx * channels + c_idx) * in_h + y) * in_w + x;
                    float val = input[in_idx];
                    if (val > max_val) {
                        max_val = val;
                    }
                }
            }
        }

        // If no valid pixels were found in the window (e.g. all padding), max_val remains -inf.
        // However, for standard MaxPool, we usually treat padding as -inf.
        
        // Apply Scale: scale is (channels, 1, 1)
        float scaled_val = max_val * scale[c_idx];

        // Apply Clamp
        float clamped_val = std::max(clamp_min, std::min(clamp_max, scaled_val));

        output[idx] = clamped_val;
    }
}

torch::Tensor scale_maxpool_clamp_cuda(
    torch::Tensor input,
    torch::Tensor scale,
    int pool_size,
    int pool_stride,
    int pool_padding,
    float clamp_min,
    float clamp_max) 
{
    const int batch_size = input.size(0);
    const int channels = input.size(1);
    const int in_h = input.size(2);
    const int in_w = input.size(3);

    // Calculate output dimensions manually to match MaxPool2d behavior
    int out_h = (in_h + 2 * pool_padding - pool_size) / pool_stride + 1;
    int out_w = (in_w + 2 * pool_padding - pool_size) / pool_stride + 1;

    auto output = torch::empty({batch_size, channels, out_h, out_w}, input.options());

    const int total_elements = batch_size * channels * out_h * out_w;
    const int block_size = 256;
    const int num_blocks = (total_elements + block_size - 1) / block_size;

    scale_maxpool_clamp_kernel<<<num_blocks, block_size>>>(
        input.data_ptr<float>(),
        scale.data_ptr<float>(),
        output.data_ptr<float>(),
        batch_size,
        channels,
        in_h,
        in_w,
        out_h,
        out_w,
        pool_size,
        pool_stride,
        pool_padding,
        clamp_min,
        clamp_max
    );

    return output;
}
"""

fused_ops_cpp_source = """
torch::Tensor scale_maxpool_clamp_cuda(
    torch::Tensor input,
    torch::Tensor scale,
    int pool_size,
    int pool_stride,
    int pool_padding,
    float clamp_min,
    float clamp_max);
"""

# Compile the inline CUDA code
fused_ops = load_inline(
    name="fused_ops",
    cpp_sources=fused_ops_cpp_source,
    cuda_sources=fused_ops_cuda_source,
    functions=["scale_maxpool_clamp_cuda"],
    verbose=False,
)

class ModelNew(nn.Module):
    """
    Optimized Model using a fused CUDA kernel for Scale, MaxPool, and Clamp.
    """
    def __init__(self, in_channels, out_channels, kernel_size, num_groups, scale_shape, maxpool_kernel_size, clamp_min, clamp_max):
        super(ModelNew, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.group_norm = nn.GroupNorm(num_groups, out_channels)
        self.scale = nn.Parameter(torch.ones(scale_shape))
        
        # Parameters for the fused kernel
        self.maxpool_kernel_size = maxpool_kernel_size
        self.clamp_min = clamp_min
        self.clamp_max = clamp_max
        self.fused_ops = fused_ops

    def forward(self, x):
        # Conv and GroupNorm are kept as standard PyTorch ops as they are highly optimized 
        # via cuDNN and are the primary computational bottlenecks.
        x = self.conv(x)
        x = self.group_norm(x)
        
        # Fuse: x * scale -> MaxPool -> Clamp
        # We pass the scale parameter and pooling parameters to the custom kernel.
        # Note: scale is expected to be (C, 1, 1) or (C)
        x = self.fused_ops.scale_maxpool_clamp_cuda(
            x, 
            self.scale.view(-1), # Flatten scale to 1D for easier indexing in kernel
            self.maxpool_kernel_size,
            self.maxpool_kernel_size, # stride = kernel_size for standard MaxPool2d behavior in this context
            0,                        # padding
            self.clamp_min,
            self.clamp_max
        )
        return x
