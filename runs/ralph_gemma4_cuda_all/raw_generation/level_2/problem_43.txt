import torch.nn.functional as F
from torch.utils.cpp_extension import load_inline

# Custom CUDA kernel to fuse LogSumExp and ReLU.
# Since LogSumExp is performed over the channel dimension (dim=1), 
# and the output shape is (B, 1, D, H, W), we can process each spatial location independently.
# We use the online LogSumExp algorithm (similar to online softmax) to maintain numerical stability.
#

logsumexp_relu_cuda_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cmath>
#include <algorithm>

__global__ void logsumexp_relu_kernel(
    const float* __restrict__ input,
    float* __restrict__ output,
    int batch_size,
    int channels,
    int depth,
    int height,
    int width,
    int spatial_size) 
{
    // Each thread handles one spatial location (batch_idx, depth_idx, height_idx, width_idx)
    // across all channels.
    int spatial_idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (spatial_idx >= batch_size * depth * height * width) {
        return;
    }

    // Calculate the base index for the spatial location
    // spatial_idx = b * (D*H*W) + d * (H*W) + h * (W) + w
    // We need to find the indices for the specific (b, d, h, w)
    int b_idx = spatial_idx / (depth * height * width);
    int rem = spatial_idx % (depth * height * width);
    int d_idx = rem / (height * width);
    int rem2 = rem % (height * width);
    int h_idx = rem2 / width;
    int w_idx = rem2 % width;

    // The channel dimension is the one we reduce over.
    // The target output is (B, 1, D, H, W).
    
    // We use a simple loop over channels for each thread.
    // For a larger number of channels, one might use shared memory or reduction, 
    // but for a single thread per spatial location is robust.
    
    float max_val = -INFINITY;
    for (int c = 0; c < channels; ++c) {
        float val = input[b_idx * (channels * depth * height * width) + c * (depth * height * width) + rem];
        if (val > max_val) {
            max_val = val;
        }
    }

    float sum_exp = 0.0f;
    for (int int_c = 0; int_c < channels; ++int_c) {
        float val = input[b_idx * (channels * depth * height * width) + int_c * (depth * height * width) + rem];
        float current_val = input[b_idx * (channels * depth * height * width) + int_c * (depth * height * width) + rem];
        sum_exp += expf(current_val - max_val);
    }
    
    float lse = max_val + logf(sum_exp);
    
    // Apply ReLU: max(0, x)
    float res = lse > 0.0f ? lse : 0.0f;
    
    output[spatial_idx] = res;
}

    }

    // Calculate the offset for the spatial location in the input tensor
    // We need to find the indices for the (b, d, h, w)
    // We can calculate the offset for the spatial location in a way that 
    // minimizes the number of channel-wise jumps.

    //    Input index = b * (C*D*H*W) + c * (D*H*W) + d * (H*W) + h * (W) + w
    //    Input index = (b * D * H * W + d * H * W + h * W + w) * C + c? No.
    //    Input index = (b * D * H * W + d * H * W + h * W + w) * C + c? No.
    //    Standard PyTorch layout: (B, C, D, H, W)
    //    Input index = b * (C*D*H*W) + c * (D*H*W) + d * (H*W) + h * (W) + w
    
    // Let's use the spatial index as the primary iterator.
    // Let's define spatial_idx = b * (D*H*W) + d * (H*W) + h * (W) + w.
    //    Then Input index = b * (C*D*H*W) + c * (D*H*W) + spatial_idx_within_batch
