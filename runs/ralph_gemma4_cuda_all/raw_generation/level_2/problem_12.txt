import torch.nn as nn
import torch.nn.functional as F
from torch.utils.cpp_extension import load_inline

# Define the custom CUDA kernel for fused Multiplier and LeakyReLU
# This kernel reduces memory bandwidth by performing both operations in a single pass.
fused_elementwise_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void fused_elementwise_kernel(const float* __restrict__ input, 
                                         float* __restrict__ output, 
                                         float multiplier, 
                                         float negative_slope, 
                                         int size) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < size) {
        float val = input[idx] * multiplier;
        if (val < 0.0f) {
            val = val * negative_slope;
        }
        output[idx] = val;
    }
}

torch::Tensor fused_elementwise_cuda(torch::Tensor input, float multiplier, float negative_slope) {
    auto size = input.numel();
    auto output = torch::empty_like(input);

    const int block_size = 256;
    const int num_blocks = (size + block_size - 1) / block_size;

    fused_elementwise_kernel<<<num_blocks, block_size>>>(
        input.data_ptr<float>(), 
        output.data_ptr<float>(), 
        multiplier, 
        negative_slope, 
        size
    );

    return output;
}
"""

fused_elementwise_cpp_source = """
torch::Tensor fused_elementwise_cuda(torch::Tensor input, float multiplier, float negative_slope);
"""

# Compile the inline CUDA code
fused_ops = load_inline(
    name="fused_ops",
    cpp_sources=fused_elementwise_cpp_source,
    cuda_sources=fused_elementwise_source,
    functions=["fused_elementwise_cuda"],
    verbose=False,
)

class ModelNew(nn.Module):
    """
    Optimized model that uses a custom CUDA kernel to fuse the multiplier 
    and LeakyReLU operations, reducing memory access overhead.
    """
    def __init__(self, in_features, out_features, multiplier, negative_slope):
        super(ModelNew, self).__init__()
        self.gemm = nn.Linear(in_features, out_features)
        self.multiplier = multiplier
        self.negative_slope = negative_slope
        self.fused_ops = fused_ops

    def forward(self, x):
        # Perform GEMM using highly optimized cuBLAS (via PyTorch)
        x = self.gemm(x)
        
        # Fuse element-wise multiplication and LeakyReLU into a single CUDA kernel
        # to minimize global memory reads/writes.
        x = self.fused_ops.fused_elementwise_cuda(x, self.multiplier, self.negative_slope)
        
        return x
