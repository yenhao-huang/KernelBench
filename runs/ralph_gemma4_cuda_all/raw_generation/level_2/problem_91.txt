import torch
import torch.nn as nn
import torch.nn.functional as F
from torch.utils.cpp_extension import load_inline

# The optimization strategy:
# 1. Keep ConvTranspose2d as is (it's highly optimized in cuDNN).
# 2. Fuse the subsequent element-wise operations: Softmax (dim=1) + Bias + Scale + Sigmoid.
# 3. Softmax is performed over the channel dimension (dim=1). 
#    To do this efficiently in a single kernel, we process each (batch, height, width) 
#    pixel as a vector of length 'out_channels'.
# 4. We use the "Online Softmax" approach or a two-pass approach within the kernel 
#    to ensure numerical stability and high performance.

fused_ops_cuda_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math.h>

__global__ void fused_post_process_kernel(
    const float* __restrict__ input,
    const float* __restrict__ bias,
    float* __restrict__ output,
    int batch_size,
    int channels,
    int height,
    int width,
    float scaling_factor) 
{
    // Each thread handles one (batch, height, width) location
    int hw_idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total_hw = height * width;

    if (hw_idx < total_hw) {
        int b = blockIdx.y;
        int h_w_offset = hw_idx;
        
        // We need to perform softmax over the 'channels' dimension for this specific pixel.
        // To avoid excessive global memory reads, we'll use shared memory or registers.
        // Since 'channels' can be large, we'll use a two-pass approach per thread.
        
        // 1. Find Max for numerical stability
        float max_val = -1e38f;
        for (int c = 0; c < channels; ++c) {
            float val = input[(b * channels * total_hw) + (c * total_hw) + h_w_offset];
            if (val > max_val) max_val = val;
        }

        // 2. Compute Sum of Exponentials
        float sum_exp = 0.0f;
        for (int c = 0; c < channels; ++c) {
            float val = input[(b * channels * total_hw) + (c * total_hw) + h_w_offset];
            sum_exp += expf(val - max_val);
        }

        // 3. Compute Softmax, add Bias, Scale, and Sigmoid
        for (int c = 0; c < channels; ++c) {
            float softmax_val = expf(input[(b * channels * total_hw) + (c * total_hw) + h_w_offset] - max_val) / sum_exp;
            
            // bias is shape (out_channels, 1, 1)
            float b_val = bias[c];
            
            float res = (softmax_val + b_val) * scaling_factor;
            
            // Sigmoid: 1 / (1 + exp(-x))
            output[(b * channels * total_hw) + (c * total_hw) + h_w_offset] = 1.0f / (1.0f + expf(-res));
        }
    }
}

torch::Tensor fused_post_process_cuda(
    torch::Tensor input, 
    torch::Tensor bias, 
    float scaling_factor) 
{
    auto batch_size = input.size(0);
    auto channels = input.size(1);
    auto height = input.size(2);
    auto width = input.size(3);
    
    auto output = torch::empty_like(input);

    int total_hw = height * width;
    dim3 threads(256);
    dim3 blocks((total_hw + threads.x - 1) / threads.x, batch_size);

    fused_post_process_kernel<<<blocks, threads>>>(
        input.data_ptr<float>(),
        bias.data_ptr<float>(),
        output.data_ptr<float>(),
        batch_size,
        channels,
        height,
        width,
        scaling_factor
    );

    return output;
}
"""

fused_ops_cpp_source = "torch::Tensor fused_post_process_cuda(torch::Tensor input, torch::Tensor bias, float scaling_factor);"

# Compile the inline CUDA code
fused_ops = load_inline(
    name="fused_ops",
    cpp_sources=fused_ops_cpp_source,
    cuda_sources=fused_ops_cuda_source,
    functions=["fused_post_process_cuda"],
    verbose=False,
)

class ModelNew(nn.Module):
    """
    Optimized Model using fused CUDA kernel for post-processing operations.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, bias_shape, scaling_factor):
        super(ModelNew, self).__init__()
        self.conv_transpose = nn.ConvTranspose2d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, output_padding=output_padding)
        self.bias = nn.Parameter(torch.randn(bias_shape)) 
        self.scaling_factor = scaling_factor
        self.fused_ops = fused_ops

    def forward(self, x):
        # Step 1: Standard ConvTranspose2d (highly optimized in cuDNN)
        x = self.conv_transpose(x)
        
        # Step 2: Fused Softmax + Bias + Scale + Sigmoid
        # Note: The kernel expects bias to be (C,) or compatible with indexing.
        # Our bias is (C, 1, 1), so we flatten it for the kernel.
        flat_bias = self.bias.view(-1)
        
        x = self.fused_ops.fused_post_process_cuda(x, flat_bias, self.scaling_factor)
        return x
