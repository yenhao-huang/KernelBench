import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# The optimization strategy is to fuse the Linear layer (Matmul + Bias), 
# the Sigmoid activation, and the Row-wise Sum reduction into a single CUDA kernel.
# This avoids multiple global memory round-trips for the intermediate (batch, hidden) tensor.

fused_cuda_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math.h>

// Kernel to compute: out[i] = sum_{j=0}^{hidden_size-1} sigmoid(x[i, k] * w[j, k] + b[j])
// Note: nn.Linear stores weight as (out_features, in_features)
// x is (batch, in_features), w is (out_features, in_features), b is (out_features)
// Result is (batch, 1)

__global__ void fused_linear_sigmoid_sum_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ b,
    float* __restrict__ out,
    int batch_size,
    int input_size,
    int hidden_size) 
{
    // Each block handles one row of the batch
    int batch_idx = blockIdx.x;
    if (batch_idx >= batch_size) return;

    // We use shared memory to accumulate the sum for the row
    // To handle large hidden_size, we use a reduction pattern within the block
    extern __shared__ float shared_mem[];
    
    // Each thread calculates a partial sum of the hidden dimension
    // We'll use a simple tiling/reduction approach.
    // For simplicity and robustness across different hidden_sizes, 
    // we use a grid-stride loop for the hidden dimension.

    float row_sum = 0.0f;

    // Each thread processes a subset of the hidden_size dimension
    for (int h = threadIdx.x; h < hidden_size; h += blockDim.x) {
        // Compute dot product for a single element of the hidden dimension
        // element_j = sum_{k=0}^{input_size-1} (x[batch_idx, k] * w[h, k]) + b[h]
        float dot = b[h];
        for (int k = 0; k < input_size; ++k) {
            dot += x[batch_idx * input_size + k] * w[h * input_size + k];
        }
        // Apply sigmoid
        float sig = 1.0f / (1.0f + expf(-dot));
        row_sum += sig;
    }

    // Block reduction
    shared_mem[threadIdx.x] = row_sum;
    __syncthreads();

    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (threadIdx.x < s) {
            shared_mem[threadIdx.x] += shared_mem[threadIdx.x + s];
        }
        __syncthreads();
    }

    if (threadIdx.x == 0) {
        out[batch_idx] = shared_mem[0];
    }
}

torch::Tensor fused_linear_sigmoid_sum_cuda(
    torch::Tensor x, 
    torch::Tensor w, 
    torch::Tensor b) 
{
    const int batch_size = x.size(0);
    const int input_size = x.size(1);
    const int hidden_size = w.size(0);

    auto out = torch::zeros({batch_size, 1}, x.options());

    // We launch one block per batch element. 
    // Block size is 256 threads.
    const int threads = 256;
    const int blocks = batch_size;
    
    // Shared memory size: threads * sizeof(float)
    size_t shared_mem_size = threads * sizeof(float);

    fused_linear_sigmoid_sum_kernel<<<blocks, threads, shared_mem_size>>>(
        x.data_ptr<float>(),
        w.data_ptr<float>(),
        b.data_ptr<float>(),
        out.data_ptr<float>(),
        batch_size,
        input_size,
        hidden_size
    );

    return out;
}
"""

fused_cpp_source = "torch::Tensor fused_linear_sigmoid_sum_cuda(torch::Tensor x, torch::Tensor w, torch::Tensor b);"

# Compile the inline CUDA code
fused_ops = load_inline(
    name="fused_ops",
    cpp_sources=fused_cpp_source,
    cuda_sources=fused_cuda_source,
    functions=["fused_linear_sigmoid_sum_cuda"],
    verbose=False,
)

class ModelNew(nn.Module):
    """
    Optimized model using a fused CUDA kernel for Linear + Sigmoid + Sum.
    """
    def __init__(self, input_size, hidden_size):
        super(ModelNew, self).__init__()
        self.input_size = input_size
        self.hidden_size = hidden_size
        self.linear = nn.Linear(input_size, hidden_size)
        self.fused_ops = fused_ops

    def forward(self, x):
        """
        Args:
            x: Input tensor of shape (batch_size, input_size).

        Returns:
            Output tensor of shape (batch_size, 1).
        """
        # Ensure inputs are contiguous for the CUDA kernel
        x = x.contiguous()
        weight = self.linear.weight.contiguous()
        bias = self.linear.bias.contiguous()
        
        # Call the fused kernel
        return self.fused_ops.fused_linear_sigmoid_sum_cuda(x, weight, bias)
