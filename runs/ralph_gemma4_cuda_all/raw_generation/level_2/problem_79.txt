import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# The optimization strategy:
# The original sequence is:
# 1. x = conv3d(x)
# 2. x = x * multiplier
# 3. x = instance_norm(x)
# 4. x = clamp(x, min, max)
# 5. x = x * multiplier
# 6. x = max(x, dim=1)
#
# We can fuse steps 2, 3, 4, 5, and 6 into a single kernel.
# Step 3 (InstanceNorm3d) requires calculating mean and variance per channel per sample.
# We will compute mean and variance in a separate pass (or fused with a reduction) 
# and then apply the fused kernel.
#
# Fused Kernel Logic:
# For each (batch, channel, d, h, w):
#   val = ( (x * mult) - mean ) / (std + eps)
#   val = clamp(val, min, max)
#   val = val * mult
# Then reduce over the channel dimension (dim=1) using max.

fused_ops_cuda_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <vector>

__global__ void fused_post_conv_kernel(
    const float* __restrict__ x,
    const float* __restrict__ multiplier,
    const float* __restrict__ mean,
    const float* __restrict__ inv_std,
    float* __restrict__ out_max,
    int batch_size,
    int channels,
    int spatial_size,
    float clamp_min,
    float clamp_max) 
{
    // We want to compute max over dim=1 (channels).
    // The output shape is (batch, spatial_size).
    // Each thread handles one (batch, spatial_idx) to find the max across channels.
    
    int b = blockIdx.y;
    int s = blockIdx.x * blockDim.x + threadIdx.x;

    if (b < batch_size && s < spatial_size) {
        float max_val = -1e38f; // Approximate -infinity
        
        for (int c = 0; c < channels; ++c) {
            // Indexing for x: [b, c, s_d, s_h, s_w] -> flattened spatial
            // Since we are iterating over spatial_size, we assume spatial_size = D*H*W
            int x_idx = (b * channels * spatial_size) + (c * spatial_size) + s;
            int m_idx = c; // multiplier is (C, 1, 1, 1)
            int mean_idx = (b * channels) + c;
            int std_idx = (b * channels) + c;

            float val = x[x_idx];
            float mult = multiplier[m_idx];
            
            // Step 2: x * multiplier
            val = val * mult;
            
            // Step 3: Instance Norm (simplified: (val - mean) * inv_std)
            // Note: InstanceNorm3d usually has learnable gamma/beta, 
            // but the original code uses default (gamma=1, beta=0).
            val = (val - mean[mean_idx]) * inv_std[std_idx];
            
            // Step 4: Clamp
            if (val < clamp_min) val = clamp_min;
            if (val > clamp_max) val = clamp_max;
            
            // Step 5: x * multiplier
            val = val * mult;
            
            if (val > max_val) max_val = val;
        }
        out_max[b * spatial_size + s] = max_val;
    }
}

torch::Tensor fused_post_conv_cuda(
    torch::Tensor x, 
    torch::Tensor multiplier, 
    torch::Tensor mean, 
    torch::Tensor inv_std, 
    float clamp_min, 
    float clamp_max) 
{
    int batch_size = x.size(0);
    int channels = x.size(1);
    int spatial_size = x.size(2) * x.size(3) * x.size(4);
    
    auto options = torch::TensorOptions().dtype(x.dtype()).device(x.device());
    auto out_max = torch::empty({batch_size, spatial_size}, options);

    dim3 block(256);
    dim3 grid((spatial_size + block.x - 1) / block.x, batch_size);

    fused_post_conv_kernel<<<grid, block>>>(
        x.data_ptr<float>(),
        multiplier.data_ptr<float>(),
        mean.data_ptr<float>(),
        inv_std.data_ptr<float>(),
        out_max.data_ptr<float>(),
        batch_size,
        channels,
        spatial_size,
        clamp_min,
        clamp_max
    );

    return out_max;
}
"""

fused_ops_cpp_source = "torch::Tensor fused_post_conv_cuda(torch::Tensor x, torch::Tensor multiplier, torch::Tensor mean, torch::Tensor inv_std, float clamp_min, float clamp_max);"

fused_ops = load_inline(
    name="fused_ops",
    cpp_sources=fused_ops_cpp_source,
    cuda_sources=fused_ops_cuda_source,
    functions=["fused_post_conv_cuda"],
    verbose=False,
)

class ModelNew(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, multiplier_shape, clamp_min, clamp_max):
        super(ModelNew, self).__init__()
        self.conv = nn.Conv3d(in_channels, out_channels, kernel_size)
        self.multiplier = nn.Parameter(torch.randn(multiplier_shape))
        self.instance_norm = nn.InstanceNorm3d(out_channels)
        self.clamp_min = clamp_min
        self.clamp_max = clamp_max
        self.fused_ops = fused_ops

    def forward(self, x):
        # 1. Conv
        x = self.conv(x)
        
        # To fuse the rest, we need the mean and inv_std of the (x * multiplier) stage
        # because InstanceNorm is applied to (x * multiplier).
        # Let's compute the intermediate x_mult for the purpose of calculating stats.
        x_mult = x * self.multiplier
        
        # We use PyTorch's efficient built-in reduction for mean/var to get stats
        # for the InstanceNorm step.
        # InstanceNorm3d calculates mean/var over (D, H, W) for each (N, C).
        # x_mult shape: (N, C, D, H, W)
        # We need mean and std over dims (2, 3, 4)
        dims = (2, 3, 4)
        mean = x_mult.mean(dim=dims, keepdim=True)
        # var = E[x^2] - (E[x])^2
        var = x_mult.var(dim=dims, keepdim=True, unbiased=False)
        inv_std = 1.0 / torch.sqrt(var + 1e-5) # 1e-5 is default eps for InstanceNorm3d
        
        # Flatten mean and inv_std to (N, C) for the kernel
        mean_flat = mean.view(x.size(0), x.size(1))
        inv_std_flat = inv_std.view(x.size(0), x.size(1))
        
        # 2-6. Fused Kernel
        # The kernel expects x to be the original conv output.
        # It performs: x_mult = x * mult; val = (x_mult - mean)/std; clamp; val * mult; max
        # Note: The original code does: x = x * mult; x = inst_norm(x); x = clamp(x); x = x * mult;
        # Our kernel follows this logic exactly.
        
        # Reshape x to (N, C, D*H*W) for easier indexing in kernel
        x_reshaped = x.view(x.size(0), x.size(1), -1)
        
        # The kernel returns (N, D*H*W)
        out = self.fused_ops.fused_post_conv_cuda(
            x_reshaped, 
            self.multiplier.view(-1), 
            mean_flat, 
            inv_std_flat, 
            self.clamp_min, 
            self.clamp_max
        )
        
        # The kernel output is (N, spatial_size). 
        # The original code: x = torch.max(x, dim=1)[0]
        # If x was (N, C, D, H, W), max(dim=1) results in (N, D, H, W).
        # Our kernel already performs the max over C, so it returns (N, D*H*W).
        # We just need to reshape it back to (N, D, H, W).
        
        # To get the correct D, H, W, we need the original dimensions.
        # Since we don't store them in the kernel, we use the input x's spatial dims.
        # However, the kernel output is already the result of the max operation.
        # The original output shape is (batch, D, H, W).
        
        # We need to know the spatial dimensions to reshape.
        # We can infer them from the input x.
        d, h, w = x.shape[2], x.shape[3], x.shape[4]
        return out.view(x.size(0), d, h, w)
