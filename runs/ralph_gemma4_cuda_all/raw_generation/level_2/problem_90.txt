import torch
import torch.nn as nn
import torch.nn.functional as F
from torch.utils.cpp_extension import load_inline

# Define the custom CUDA kernel for fused-elementwise operations: 
# LeakyReLU, Addition (with broadcasting), Clamp, and GELU.
fused_elementwise_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cmath>

__global__ void fused_elementwise_kernel(
    const float* __restrict__ x, 
    const float* __restrict__ sum_tensor, 
    float* __restrict__ out, 
    int total_size, 
    int channels, 
    int spatial_size, 
    float negative_slope, 
    float clamp_min, 
    float clamp_max
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < total_size) {
        // x shape: (N, C, D, H, W)
        // sum_tensor shape: (C, 1, 1, 1)
        
        // To handle broadcasting sum_tensor (C, 1, 1, 1) to x (N, C, D, H, W):
        // The channel index 'c' for a given global index 'idx' is:
        // idx = n * (C * spatial_size) + c * spatial_size + spatial_idx
        // Therefore, c = (idx / spatial_size) % channels
        
        int c = (idx / spatial_size) % channels;
        float val = x[idx];
        
        // 1. LeakyReLU
        if (val < 0.0f) {
            val = val * negative_slope;
        }
        
        // 2. Addition (Broadcasting sum_tensor)
        val = val + sum_tensor[c];
        
        // 3. Clamp
        if (val < clamp_min) val = clamp_min;
        else if (val > clamp_max) val = clamp_max;
        
        // 4. GELU (using erf implementation)
        // GELU(x) = 0.5 * x * (1 + erf(x / sqrt(2)))
        val = 0.5f * val * (1.0f + erf(val * 0.70710678118f));
        
        out[idx] = val;
    }
}

torch::Tensor fused_elementwise_cuda(
    torch::Tensor x, 
    torch::Tensor sum_tensor, 
    float negative_slope, 
    float clamp_min, 
    float clamp_max
) {
    auto out = torch::empty_like(x);
    int total_size = x.numel();
    int channels = x.size(1);
    // spatial_size is D * H * W
    int spatial_size = x.size(2) * x.size(3) * x.size(4);

    const int block_size = 256;
    const int num_blocks = (total_size + block_size - 1) / block_size;

    fused_elementwise_kernel<<<num_blocks, block_size>>>(
        x.data_ptr<float>(), 
        sum_tensor.data_ptr<float>(), 
        out.data_ptr<float>(), 
        total_size, 
        channels, 
        spatial_size, 
        negative_slope, 
        clamp_min, 
        clamp_max
    );

    return out;
}
"""

    """
    Optimized Model using a fused CUDA kernel for LeakyReLU, Addition, Clamp, and GELU.
    """

        super(ModelNew, self).__init__()
        self.conv = nn.Conv3d(in_channels, out_channels, kernel_size)
        self.sum_tensor = nn.Parameter(torch.randn(sum_tensor_shape))
        self.fused_ops = fused_ops
