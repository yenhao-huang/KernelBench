import torch.nn as nn

#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math.h>

__device__ __forceinline__ float gelu(float x) {
    return 0.5f * x * (1.0f + tanhf(0.79788456f * (x + 0.044715f * x * x * x)));
}

__global__ void fused_gelu_softmax_kernel(const float* __restrict__ input, float* __restrict__ output, int batch_size, int out_features) {
    int row = blockIdx.x;
    if (row >= batch_size) return;

    const float* row_input = input + row * out_features;
    float* row_output = output + row * out_features;

    // Shared memory for reduction
    // We'll use a simple approach: 
    // Each thread computes its local max and local sum, then we reduce.
    // To avoid complex reduction, we'll use a shared memory array to store 
    // the local max and sum of each thread.
    // But we'll use a fixed size for simplicity.
    extern __shared__ float s_mem[]; 
    // s_mem[0...blockDim.x-1] = local_max
    // s_mem[blockDim.x...2*blockDim.x-1] = local_sum

    int tid = threadIdx.x;
    float local_max = -1e38f;
    float local_sum = 0.0f;

    // Pass 1: Find local max and local sum
    for (int i = tid; i < out_features; i += blockDim.x) {
        float val = gelu(row_input[i]);
        if (val > local_max) local_max = val;
    }
    
    // We need the max to compute the sum. 
    // But we can't compute the sum without the max.
    // So we must do it in two passes.
    // Pass 1: Find max
    // Pass 2: Find sum
    // Pass 3: Write output
    
    // Let's store the local_max in shared memory.
    s_mem[tid] = local_max;
    __syncthreads();

    // Block reduction for max
    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (tid < s) {
            if (s_mem[tid] > s_mem[tid + s]) s_mem[tid] = s_mem[tid + s];
        }
        __syncthreads();
    }
    float max_val = s_mem[0];

    // Pass 2: Find local sum
    for (int i = tid; i < out_features; i += blockDim.x) {
        float val = gelu(row_input[i]);
        local_sum += expf(val - max_val);
    }
    s_mem[tid] = local_sum;
    __syncthreads();

    // Block reduction for sum
    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (tid < s) {
            s_mem[tid] += s_mem[tid + s];
        }
        __syncthreads();
    }
    float sum_val = s_mem[0];

    // Pass 3: Write output
    for (int i = tid; i < out_features; i += blockDim.x) {
        float val = gelu(row_input[i]);
        row_output[i] = expf(val - max_val) / sum_val;

    for (int i = tid; i < out_features; i += blockDim.x) {
        float val = gelu(row_input[i]);
        if (val > local_max) local_max = val;
    }
    
    s_mem[tid] = local_max;
    __syncthreads();

    // Block reduction for max
    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (tid < s) {
            if (s_mem[tid] > s_mem[tid + s]) s_mem[tid] = s_mem[tid + s];
        }
        __syncthreads();
    }
    float max_val = s_mem[0];

    // Pass 2: Find local sum
    for (int i = tid; i < out_features; i += blockDim.x) {
        float val = gelu(row_input[i]);
        local_sum += expf(val - max_val);
    }
    s_mem[tid] = local_sum;
    __syncthreads();

    // Block reduction for sum
    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (tid < s) {
            s_mem[tid] += s_mem[tid + s];
        }
        __syncthreads();
    }
    float sum_val = s_mem[0];

    // Pass 3: Write output
    for (int i = tid; i < out_features; i += blockDim.x) {
        float val = gelu(row_input[i]);
        row_output[i] = expf(val - max_val) / sum_val;
    }
}
