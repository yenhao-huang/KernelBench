# Smooth L1 Loss formula:
# loss = 0.5 * (x - y)^2 / beta, if |x - y| < beta
# loss = |x - y| - 0.5 * beta, otherwise
# In PyTorch's default smooth_l1_loss, beta is 1.0 unless specified.
smooth_l1_loss_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cmath>

__global__ void smooth_l1_loss_kernel(const float* __restrict__ predictions, 
                                     const float* __restrict__ targets, 
                                     float* __restrict__ out, 
                                     int size, 
                                     float beta) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < size) {
        float diff = predictions[idx] - targets[idx];
        float abs_diff = fabsf(diff);
        float loss;
        if (abs_diff < beta) {
            loss = 0.5f * diff * diff / beta;
        } else {
            loss = abs_diff - 0.5f * beta;
        }
        out[idx] = loss;
    }
}

__global__ void reduce_sum_kernel(const float* __restrict__ loss, 
                                  float* __restrict__ out, 
                                  int size) {

    // one would use a more sophisticated tree reduction or cub/thrust.
    // However, for a single scalar output, we can use atomicAdd.
    extern __shared__ float sdata[];
    int tid = threadIdx.x;
    int i = blockIdx.x * blockDim.x + threadIdx.x;

    float val = (i < size) ? loss[i] : 0.0f;
    sdata[tid] = val;
    __syncthreads();

        if (tid < s) {
            sdata[tid] += sdata[tid + s];
        }
        __syncthreads();

    if (tid == 0) {
        atomicAdd(out, sdata[0]);

    An optimized model that computes Smooth L1 (Huber) Loss using custom CUDA kernels.
    """

        super(ModelNew, self).__init__()
        self.beta = beta
        self.cuda_op = smooth_l1_lib

    def forward(self, predictions, targets):
        # Ensure inputs are float32 and on CUDA

            predictions = predictions.cuda()
        if not targets.is_cuda:
            targets = targets.cuda()
            
        return self.cuda_op.smooth_l1_loss_fused_cuda(predictions, targets, self.beta)
