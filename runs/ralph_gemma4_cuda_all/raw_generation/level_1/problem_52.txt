import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Custom CUDA kernel for argmin along dimension 1
# Input shape: (Batch, Dim1, Dim2)
# Output shape: (Batch, Dim2)
# We reduce along Dim1 (the middle dimension)
argmin_cuda_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <float.h>

__global__ void argmin_dim1_kernel(const float* __restrict__ input, int64_t* __restrict__ output, 
                                  int batch_size, int dim1, int dim2) {
    // Each block handles one (batch_idx, dim2_idx) pair
    // Grid size: (batch_size * dim2)
    // Block size: dim1 (or a subset if dim1 is large)
    
    int idx = blockIdx.x; // This represents the unique (batch, dim2) combination
    if (idx >= batch_size * dim2) return;

    int b = idx / dim2;
    int d2 = idx % dim2;

    float min_val = FLT_MAX;
    int64_t min_idx = 0;

    // We use a simple loop here. For very large dim1, one could use shared memory reduction,
    // but for dim1=4096, a single thread per (b, d2) pair is efficient if we launch 
    // enough blocks, or we can use a warp-level reduction.
    // Given the constraints and typical GPU occupancy, we'll use a loop per thread.
    // To optimize, we'll launch one thread per (b, d2) and let it loop through dim1.
    
    for (int i = 0; i < dim1; ++i) {
        float val = input[(b * dim1 * dim2) + (i * dim2) + d2];
        if (val < min_val) {
            min_val = val;
            min_idx = i;
        }
    }

    output[idx] = min_idx;
}

torch::Tensor argmin_dim1_cuda(torch::Tensor input) {
    auto batch_size = input.size(0);
    auto dim1 = input.size(1);
    auto dim2 = input.size(2);

    auto output = torch::empty({batch_size, dim2}, torch::dtype(torch::kInt64).device(input.device()));

    int total_elements = batch_size * dim2;
    
    // Launching one thread per (batch, dim2) element.
    // Since dim1 is 4096, this is a heavy thread, but it avoids complex synchronization.
    // For better performance on massive dim1, we'd use a block-reduction approach.
    // However, for dim1=4096, this is highly parallel across the batch and dim2.
    argmin_dim1_kernel<<<total_elements, 1>>>(
        input.data_ptr<float>(), 
        output.data_ptr<int64_t>(), 
        batch_size, dim1, dim2
    );

    return output;
}
"""

argmin_cpp_source = """
torch::Tensor argmin_dim1_cuda(torch::Tensor input);
"""

# Compile the inline CUDA code
argmin_extension = load_inline(
    name="argmin_extension",
    cpp_sources=argmin_cpp_source,
    cuda_sources=argmin_cuda_source,
    functions=["argmin_dim1_cuda"],
    verbose=False,
)

class ModelNew(nn.Module):
    """
    Optimized model that finds the index of the minimum value along dimension 1.
    """
    def __init__(self, dim: int):
        super(ModelNew, self).__init__()
        self.dim = dim
        self.argmin_cuda = argmin_extension

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # The original model performs argmin on self.dim.
        # Based on the provided architecture, dim=1.
        # Our custom kernel is specifically optimized for dim=1 on a 3D tensor.
        if self.dim == 1 and x.dim() == 3:
            return self.argmin_cuda.argmin_dim1_cuda(x)
        else:
            # Fallback for other dimensions or shapes
            return torch.argmin(x, dim=self.dim)
