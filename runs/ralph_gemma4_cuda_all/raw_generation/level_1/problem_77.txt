import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# The optimization strategy here is to implement a fused kernel for the 3D Transposed Convolution.
# Standard ConvTranspose3d in PyTorch is highly optimized but can be memory intensive due to 
# intermediate gradient/buffer allocations. For a custom implementation, we use a 
# "scatter-add" style approach where each input voxel contributes to a local window in the output.
# This is particularly efficient for sparse or large-stride scenarios.

cuda_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void conv_transpose3d_kernel(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    float* __restrict__ output,
    int batch_size, int in_channels, int in_d, int in_h, int in_w,
    int out_channels, int out_d, int out_h, int out_w,
    int kernel_size, int stride, int padding, int dilation) 
{
    // Calculate global index for the input voxel
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total_input_elements = batch_size * in_channels * in_d * in_h * in_w;

    if (idx < total_input_elements) {
        // Decompose idx into coordinates
        int w_in = idx % in_w;
        int h_in = (idx / in_w) % in_h;
        int d_in = (idx / (in_w * in_h)) % in_d;
        int c_in = (idx / (in_w * in_h * in_d)) % in_channels;
        int b = idx / (in_w * in_h * in_d * in_channels);

        float val = input[idx];

        // The output position for the top-left-front corner of the kernel application
        // In ConvTranspose3d, the output coordinate is roughly (in_coord - padding) * stride
        int out_d_start = d_in * stride - padding;
        int out_h_start = h_in * stride - padding;
        int out_w_start = w_in * stride - padding;

        for (int oc = 0; oc < out_channels; ++oc) {
            for (int kd = 0; kd < kernel_size; ++kd) {
                int od = out_d_start + kd * dilation;
                if (od < 0 || od >= out_d) continue;

                for (int kh = 0; kh < kernel_size; ++kh) {
                    int oh = out_h_start + kh * dilation;
                    if (oh < 0 || oh >= out_h) continue;

                    for (int kw = 0; kw < kernel_size; ++kw) {
                        int ow = out_w_start + kw * dilation;
                        if (ow < 0 || ow >= out_w) continue;

                        // weight shape: [in_channels, out_channels, kD, kH, kW]
                        // Note: PyTorch weight for ConvTranspose3d is [in_channels, out_channels, kD, kH, kW]
                        int weight_idx = ((c_in * out_channels + oc) * kernel_size + kd) * (kernel_size * kernel_size) + (kh * kernel_size + kw);
                        int out_idx = (((b * out_channels + oc) * out_d + od) * out_h + oh) * out_w + ow;

                        // Use atomicAdd to handle overlapping kernel contributions
                        atomicAdd(&output[out_idx], val * weight[weight_idx]);
                    }
                }
            }
        }
    }
}

torch::Tensor conv_transpose3d_cuda_forward(
    torch::Tensor input,
    torch::Tensor weight,
    at::optional<torch::Tensor> bias,
    int stride, int padding, int dilation,
    int out_d, int out_h, int out_w) 
{
    const int batch_size = input.size(0);
    const int in_channels = input.size(1);
    const int in_d = input.size(2);
    const int in_h = input.size(3);
    const int in_w = input.size(4);

    const int out_channels = weight.size(1);
    const int kernel_size = weight.size(2);

    auto output = torch::zeros({batch_size, out_channels, out_d, out_h, out_w}, input.options());

    int total_input_elements = input.numel();
    const int block_size = 256;
    const int num_blocks = (total_input_elements + block_size - 1) / block_size;

    conv_transpose3d_kernel<<<num_blocks, block_size>>>(
        input.data_ptr<float>(),
        weight.data_ptr<float>(),
        bias.has_value() ? bias.value().data_ptr<float>() : nullptr,
        output.data_ptr<float>(),
        batch_size, in_channels, in_d, in_h, in_w,
        out_channels, out_d, out_h, out_w,
        kernel_size, stride, padding, dilation
    );

    if (bias.has_value()) {
        // Add bias: output += bias.view(1, C, 1, 1, 1)
        output. += bias.value().view({1, out_channels, 1, 1, 1});
    }

    return output;
}
"""

cpp_source = """
torch::Tensor conv_transpose3d_cuda_forward(
    torch::Tensor input,
    torch::Tensor weight,
    at::optional<torch::Tensor> bias,
    int stride, int padding, int dilation,
    int out_d, int out_h, int out_w);
"""

conv_transpose_lib = load_inline(
    name="conv_transpose_lib",
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=["conv_transpose3d_cuda_forward"],
    verbose=False,
)

class ModelNew(nn.Module):
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, dilation: int = 1, bias: bool = False):
        super(ModelNew, self).__init__()
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.dilation = dilation
        
        # We still use the weight and bias from a standard layer for parameter management
        self.conv_transpose3d = nn.ConvTranspose3d(
            in_channels, out_channels, 
            kernel_size=(kernel_size, kernel_size, kernel_size), 
            stride=stride, padding=padding, dilation=dilation, bias=bias
        )
        self.custom_op = conv_transpose_lib

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Calculate output dimensions manually to match PyTorch's ConvTranspose3d logic
        # Formula: H_out = (H_in - 1) * stride - 2 * padding + dilation * (kernel_size - 1) + 1
        def get_out_dim(in_dim):
            return (in_dim - 1) * self.stride - 2 * self.padding + self.dilation * (self.kernel_size - 1) + 1

        out_d = get_out_dim(x.size(2))
        out_h = get_out_dim(x.size(3))
        out_w = get_out_dim(x.size(4))

        # Ensure input is contiguous for the CUDA kernel
        x = x.contiguous()
        weight = self.conv_transpose3d.weight.contiguous()
        bias = self.conv_transpose3d.bias
        if bias is not None:
            bias = bias.contiguous()

        return self.custom_op.conv_transpose3d_cuda_forward(
            x, weight, bias, 
            self.stride, self.padding, self.dilation,
            out_d, out_h, out_w
        )
