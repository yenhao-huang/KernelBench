import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Define the custom CUDA kernel for 3D Average Pooling
avg_pool3d_cuda_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void avg_pool3d_kernel(
    const float* __restrict__ input,
    float* __restrict__ output,
    int batch_size, int channels, int in_d, int in_h, int in_w,
    int out_d, int out_h, int out_w,
    int k_d, int k_h, int k_w,
    int stride_d, int stride_h, int stride_w,
    int pad_d, int pad_h, int pad_w
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total_elements = batch_size * channels * out_d * out_h * out_w;

    if (idx < total_elements) {
        // Decompose idx into output coordinates
        int w_out = idx % out_w;
        int rem = idx / out_w;
        int h_out = rem % out_h;
        rem /= out_h;
        int d_out = rem % out_d;
        rem /= out_d;
        int c = rem % channels;
        int n = rem / channels;

        // Calculate start positions in input
        int d_start = d_out * stride_d - pad_d;
        int h_start = h_out * stride_h - pad_h;
        int w_start = w_out * stride_w - pad_w;

        float sum = 0.0f;

        for (int kd = 0; kd < k_d; ++kd) {
            int d_in = d_start + kd;
            if (d_in >= 0 && d_in < in_d) {
                for (int kh = 0; kh < k_h; ++kh) {
                    int h_in = h_start + kh;
                    if (h_in >= 0 && h_in < in_h) {
                        for (int kw = 0; kw < k_w; ++kw) {
                            int w_in = w_start + kw;
                            if (w_in >= 0 && w_in < in_w) {
                                int input_idx = n * (channels * in_d * in_h * in_w) +
                                                c * (in_d * in_h * in_w) +
                                                d_in * (in_h * in_w) +
                                                h_in * (in_w) +
                                                w_in;
                                sum += input[input_idx];
                            }
                        }
                    }
                }
            }
        }

        // PyTorch AvgPool3d divides by the product of kernel dimensions
        output[idx] = sum / (float)(k_d * k_h * k_w);
    }
}

torch::Tensor avg_pool3d_cuda_forward(
    torch::Tensor input,
    int k_d, int k_h, int k_w,
    int stride_d, int stride_h, int stride_w,
    int pad_d, int pad_h, int pad_w
) {
    const int batch_size = input.size(0);
    const int channels = input.size(1);
    const int in_d = input.size(2);
    const int in_h = input.size(3);
    const int in_w = input.size(4);

    // Calculate output dimensions
    int out_d = (in_d + 2 * pad_d - k_d) / stride_d + 1;
    int out_h = (in_h + 2 * pad_h - k_h) / stride_h + 1;
    int out_w = (in_w + 2 * pad_w - k_w) / stride_w + 1;

    auto output = torch::empty({batch_size, channels, out_d, out_h, out_w}, input.options());

    int total_output_elements = output.numel();
    const int block_size = 256;
    const int num_blocks = (total_output_elements + block_size - 1) / block_size;

    avg_pool3d_kernel<<<num_blocks, block_size>>>(
        input.data_ptr<float>(),
        output.data_ptr<float>(),
        batch_size, channels, in_d, in_h, in_w,
        out_d, out_h, out_w,
        k_d, k_h, k_w,
        stride_d, stride_h, stride_w,
        pad_d, pad_h, pad_w
    );

    return output;
}
"""

    Optimized model that performs 3D Average Pooling using a custom CUDA kernel.
    """

        super(ModelNew, self).__init__()
