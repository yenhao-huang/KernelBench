import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# CUDA kernel for 1D transposed convolution
# We implement a kernel where each thread calculates one output element.
# This is efficient for large output lengths.
# The kernel handles stride, dilation, padding, and bias.

conv_transpose_1d_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void conv_transpose_1d_kernel(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    float* __restrict__ output,
    int batch_size,
    int in_channels,
    int out_channels,
    int input_length,
    int output_length,
    int kernel_size,
    int stride,
    int padding,
    int dilation) 
{
    // Each thread handles one element of the output tensor: (b, oc, l)
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total_elements = batch_size * out_channels * output_length;

    if (idx < total_elements) {
        int l = idx % output_length;
        int oc = (idx / output_length) % out_channels;
        int b = idx / (output_length * out_channels);

        float val = 0.0f;

        // In a transposed convolution, the output element at (b, oc, l) 
        // is a sum of contributions from input elements (b, ic, i_l)
        // where l = i_l * stride + k * dilation - padding
        // Rearranging for i_l and k:
        // i_l * stride + k * dilation = l + padding
        
        // We iterate over the kernel size
        for (int k = 0; k < kernel_size; ++k) {
            int target_pos = l + padding - k * dilation;
            
            if (target_pos % stride == 0) {
                int i_l = target_pos / stride;
                if (i_l >= 0 && i_l < input_length) {
                    for (int ic = 0; ic < in_channels; ++ic) {
                        // weight shape: (in_channels, out_channels, kernel_size)
                        // input shape: (batch_size, in_channels, input_length)
                        float w = weight[ic * out_channels * kernel_size + oc * kernel_size + k];
                        float in_val = input[b * in_channels * input_length + ic * input_length + i_l];
                        val += in_val * w;
                    }
                }
            }
        }

        if (bias != nullptr) {
            val += bias[oc];
        }
        output[idx] = val;
    }
}

torch::Tensor conv_transpose_1d_cuda(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    int stride,
    int padding,
    int dilation,
    int output_length) 
{
    const int batch_size = input.size(0);
    const int in_channels = input.size(1);
    const int input_length = input.size(2);
    const int out_channels = weight.size(1);
    const int kernel_size = weight.size(2);

    auto output = torch::zeros({batch_size, out_channels, output_length}, input.options());

    const int total_elements = batch_size * out_channels * output_length;
    const int block_size = 256;
    const int num_blocks = (total_elements + block_size - 1) / block_size;

    conv_transpose_1d_kernel<<<num_blocks, block_size>>>(
        input.data_ptr<float>(),
        weight.data_ptr<float>(),
        bias.defined() ? bias.data_ptr<float>() : nullptr,
        output.data_ptr<float>(),
        batch_size,
        in_channels,
        out_channels,
        input_length,
        output_length,
        kernel_size,
        stride,
        padding,
        dilation
    );

    return output;
}
"""

conv_transpose_1d_cpp_source = """
torch::Tensor conv_transpose_1d_cuda(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    int stride,
    int padding,
    int dilation,
    int output_length);
"""

conv_transpose_1d_lib = load_inline(
    name="conv_transpose_1d_lib",
    cpp_sources=conv_transpose_1d_cpp_source,
    cuda_sources=conv_transpose_1d_source,
    functions=["conv_transpose_1d_cuda"],
    verbose=False,
)

class ModelNew(nn.Module):
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, dilation: int = 1, bias: bool = False):
        super(ModelNew, self).__init__()
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.dilation = dilation
        
        # We still use the standard layer to manage parameters
        self.conv1d_transpose = nn.ConvTranspose1d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, dilation=dilation, bias=bias)
        self.has_bias = bias

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Calculate output length manually to match PyTorch's ConvTranspose1d behavior
        # PyTorch formula for ConvTranspose1d output length:
        # L_out = (L_in - 1) * stride - 2 * padding + dilation * (kernel_size - 1) + output_padding + 1
        # Since output_padding is not provided in the original Model, we assume it's 0.
        
        L_in = x.size(2)
        L_out = (L_in - 1) * self.stride - 2 * self.padding + self.dilation * (self.kernel_size - 1) + 1
        
        bias = self.conv1d_transpose.bias if self.has_bias else torch.Tensor()

        return conv_transpose_1d_lib.conv_transpose_1d_cuda(
            x, 
            self.conv1d_transpose.weight, 
            bias, 
            self.stride, 
            self.padding, 
            self.dilation, 
            L_out
        )
