import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Custom CUDA kernel for Softmax using an online algorithm (Welford-like)
# to handle very large dimensions efficiently in a single pass per row.
# Given the massive dimension (393216), we use a grid-stride loop approach
# where each block processes one row. To handle the large dimension, 
# we use a multi-pass approach within the kernel or a reduction-based approach.
# For simplicity and robustness in a single kernel, we implement a 
# two-pass approach: 1. Find Max, 2. Compute Sum of Exps and Output.

softmax_cuda_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math.h>

__global__ void softmax_kernel(const float* __restrict__ input, float* __restrict__ output, int rows, int cols) {
    int row = blockIdx.x;
    if (row >= rows) return;

    const float* row_input = input + row * cols;
    float* row_output = output + row * cols;

    // Pass 1: Find Max
    float max_val = -INFINITY;
    for (int i = threadIdx.x; i < cols; i += blockDim.x) {
        float val = row_input[i];
        if (val > max_val) {
            max_val = val;
        }
    }
    
    // Block-level reduction for max
    extern __shared__ float shared_mem[];
    float* s_max = shared_mem;
    s_max[threadIdx.x] = max_val;
    __syncthreads();

    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (threadIdx.x < s) {
            if (s_max[threadIdx.x] > s_max[threadIdx.x + s]) {
                s_max[threadIdx.x] = s_max[threadIdx.x]; // This is a placeholder logic
            }
        }
        // Correct reduction logic
        if (threadIdx.x < s) {
            if (s_max[threadIdx.x + s] > s_max[threadIdx.x]) {
                s_max[threadIdx.x] = s_max[threadIdx.x + s];
            }
        }
        // Note: Standard reduction is complex in inline. 
        // Let's use a simpler approach: atomicMax is not for float.
        // We'll use a simple loop-based reduction for the max value 
        // since we are in a single block per row.
    }
    
    // Re-implementing a robust reduction for the kernel
    // To ensure it works for any block size, we use a simple loop for max
    // because the dimension is huge and we want to avoid complex syncs.
    // However, for performance, we'll use a more standard approach.
}
"""

# Since the dimension is extremely large (393216), a single block per row 
# with shared memory reduction is the most efficient way to avoid global memory 
# roundtrips. We'll use a kernel that performs:
# 1. Max reduction
# 2. Sum of Exps reduction
# 3. Final division

softmax_cuda_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math.h>

__global__ void softmax_kernel(const float* __restrict__ input, float* __restrict__ output, int rows, int cols) {
    int row = blockIdx.x;
    if (row >= rows) return;

    const float* r_in = input + row * cols;
    float* r_out = output + row * cols;

    // 1. Find Max
    float local_max = -INFINITY;
    for (int i = threadIdx.x; i < cols; i += blockDim.x) {
        local_max = fmaxf(local_max, r_in[i]);
    }

    // Block reduction for max
    extern __shared__ float s_mem[];
    s_mem[threadIdx.x] = local_max;
    __syncthreads();

    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (threadIdx.x < s) {
            s_mem[threadIdx.x] = fmaxf(s_mem[threadIdx.x], s_mem[threadIdx.x + s]);
        }
        __syncthreads();
    }
    float max_val = s_mem[0];

    // 2. Find Sum of Exps
    float local_sum = 0.0f;
    for (int i = threadIdx.x; i < cols; i += blockDim.x) {
        local_sum += expf(r_in[i] - max_val);
    }

    // Block reduction for sum
    s_mem[threadIdx.x] = local_sum;
    __syncthreads();

    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (threadIdx.x < s) {
            s_mem[threadIdx.x] = s_mem[threadIdx.x] + s_mem[threadIdx.x + s];
        }
        __syncthreads();
    }
    float sum_exp = s_mem[0];

    // 3. Compute Output
    for (int i = threadIdx.x; i < cols; i += blockDim.x) {
        r_out[i] = expf(r_in[i] - max_val) / sum_exp;
    }
}

torch::Tensor softmax_cuda(torch::Tensor input) {
    auto rows = input.size(0);
    auto cols = input.size(1);
    auto output = torch::empty_like(input);

    // We use 1 block per row. 
    // To handle large cols, we use a large number of threads.
    // However, blockDim.x is limited to 1024.
    // The grid-stride loop handles the rest.
    int threads = 512; 
    dim3 blocks(rows);
    dim3 grid(rows);
    dim3 threads_per_block(threads);
    size_t shared_mem_size = threads * sizeof(float);

    softmax_kernel<<<grid, threads_per_block, shared_mem_size>>>(
        input.data_ptr<float>(), 
        output.data_ptr<float>(), 
        rows, 
        cols
    );

    return output;
}
"""

softmax_cpp_source = "torch::Tensor softmax_cuda(torch::Tensor input);"

softmax_lib = load_inline(
    name="softmax_lib",
    cpp_sources=softmax_cpp_source,
    cuda_sources=softmax_cuda_source,
    functions=["softmax_cuda"],
    verbose=False,
)

class ModelNew(nn.Module):
    def __init__(self):
        super(ModelNew, self).__init__()
        self.softmax_cuda = softmax_lib.softmax_cuda

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Ensure input is contiguous and on CUDA for the custom kernel
        if not x.is_cuda:
            return torch.softmax(x, dim=1)
        x = x.contiguous()
        return self.softmax_cuda(x)
