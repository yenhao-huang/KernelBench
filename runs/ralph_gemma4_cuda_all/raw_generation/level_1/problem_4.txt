import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Custom CUDA kernel for Matrix-Vector Multiplication (C = A * B)
# Since K is extremely large (10^6) and M is relatively small (2048),
# we use a grid where each block handles one row of A.
# Within each block, we use a reduction pattern to compute the dot product.
# To handle the massive K, we use multiple threads per block to process segments of the row.

matvec_cuda_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void matvec_kernel(const float* __restrict__ A, const float* __restrict__ B, float* __restrict__ out, int M, int K) {
    // Each block handles one row of the matrix A
    int row = blockIdx.x;
    if (row >= M) return;

    // Shared memory for partial sums within the block
    extern __shared__ float sdata[];

    int tid = threadIdx.x;
    float sum = 0.0f;

    // Each thread processes multiple elements in the row to handle large K
    // We use a loop to iterate through the K elements in chunks
    for (int i = tid; i < K; i += blockDim.x) {
        sum += A[row * K + i] * B[i];
    }

    // Store partial sum in shared memory
    sdata[tid] = sum;
    __syncthreads();

    // Perform reduction in shared memory
    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (tid < s) {
            sdata[tid] += sdata[tid + s];
        }
        __syncthreads();
    }

    // The first thread writes the final result for this row
    if (tid == 0) {
        out[row] = sdata[0];
    }
}

    auto M = A.size(0);
    auto K = A.size(1);

__global__ void matvec_kernel(const float* __restrict__ A, const float* __restrict__ B, float* __restrict__ out, int M, int K) {
    int row = blockIdx.x;
    if (row >= M) return;

    // Shared memory for reduction
    // We use a fixed block size of 256 for the reduction logic
    extern __shared__ float sdata[];

    int tid = threadIdx.x;
    float sum = 0.0f;

    // Each thread processes a subset of the K elements in the row
    for (int i = tid; i < K; i += blockDim.x) {
        sum += A[row * K + i] * B[i];
    }

    sdata[tid] = sum;
    __syncthreads();

    // Reduction in shared memory
    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (tid < s) {
            sdata[tid] += sdata[tid + s];
        }
        __syncthreads();
    }

    // Write the result to the output vector
    if (tid == 0) {
        out[row] = sdata[0];
    }
}

torch::Tensor matvec_cuda(torch::Tensor A, torch::Tensor B) {
    // A: (M, K), B: (K, 1)
    int M = A.size(0);
    int K = A.size(1);

    // Using 256 threads per block for efficient reduction
    int threads = 256;
    dim3 blocks(M);
    dim3 threads_dim(threads);
    size_t shared_mem_size = threads * sizeof(float);

    matvec_kernel<<<blocks, threads_dim, shared_mem_size>>>(

        B.data_ptr<float>(), 
        out.data_ptr<float>(), 
        M, 
        K
    );
