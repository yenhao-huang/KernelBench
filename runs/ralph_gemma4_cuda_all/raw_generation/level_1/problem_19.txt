import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Define the custom CUDA kernel for ReLU activation
# This kernel is optimized for FP32 and uses a simple grid-stride loop 
# to handle any tensor size efficiently.
relu_cuda_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void relu_kernel(const float* __restrict__ input, float* __restrict__ output, int size) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int stride = blockDim.x * gridDim.x;
    
    for (int i = idx; i < size; i += stride) {
        float val = input[i];
        output[i] = (val > 0.0f) ? val : 0.0f;
    }
}

torch::Tensor relu_cuda(torch::Tensor input) {
    auto output = torch::empty_like(input);
    int size = input.numel();

    const int block_size = 256;
    // Calculate number of blocks needed, but cap it to avoid excessive overhead
    // or use grid-stride loop to handle large tensors.
    int num_blocks = (size + block_size - 1) / block_size;
    if (num_blocks > 65535) num_blocks = 65535; 

    relu_kernel<<<num_blocks, block_size>>>(
        input.data_ptr<float>(), 
        output.data_ptr<float>(), 
        size
    );

    return output;
}
"""

relu_cpp_source = """
torch::Tensor relu_cuda(torch::Tensor input);
"""

# Compile the inline CUDA code
relu_extension = load_inline(
    name="relu_extension",
    cpp_sources=relu_cpp_source,
    cuda_sources=relu_cuda_source,
    functions=["relu_cuda"],
    verbose=False,
)

class ModelNew(nn.Module):
    """
    Optimized model that performs a ReLU activation using a custom CUDA kernel.
    """
    def __init__(self):
        super(ModelNew, self).__init__()
        self.relu_cuda = relu_extension.relu_cuda
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Applies ReLU activation to the input tensor using custom CUDA kernel.

        Args:
            x (torch.Tensor): Input tensor of any shape (must be on CUDA and FP32).

        Returns:
            torch.Tensor: Output tensor with ReLU applied, same shape as input.
        """
        if not x.is_cuda:
            return torch.relu(x)
        if x.dtype != torch.float32:
            return torch.relu(x)
            
        return self.relu_cuda(x)
