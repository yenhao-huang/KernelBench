import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Define the custom CUDA kernel for Softsign activation: x / (1 + abs(x))
softsign_cuda_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void softsign_kernel(const float* __restrict__ x, float* __restrict__ out, int size) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < size) {
        float val = x[idx];
        out[idx] = val / (1.0f + fabsf(val));
    }
}

torch::Tensor softsign_cuda(torch::Tensor x) {
    auto size = x.numel();
    auto out = torch::empty_like(x);

    const int block_size = 256;
    const int num_blocks = (size + block_size - 1) / block_size;

    softsign_kernel<<<num_blocks, block_size>>>(
        x.data_ptr<float>(), 
        out.data_ptr<float>(), 
        size
    );

    return out;
}
"""

softsign_cpp_source = (
    "torch::Tensor softsign_cuda(torch::Tensor x);"
)

# Compile the inline CUDA code
softsign_lib = load_inline(
    name="softsign_lib",
    cpp_sources=softsign_cpp_source,
    cuda_sources=softsign_cuda_source,
    functions=["softsign_cuda"],
    verbose=False,
)

class ModelNew(nn.Module):
    """
    Optimized model that performs a Softsign activation using a custom CUDA kernel.
    """
    def __init__(self):
        super(ModelNew, self).__init__()
        self.softsign_cuda = softsign_lib.softsign_cuda
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Applies Softsign activation to the input tensor using custom CUDA kernel.

        Args:
            x (torch.Tensor): Input tensor of any shape (must be on CUDA and FP32).

        Returns:
            torch.Tensor: Output tensor with Softsign applied, same shape as input.
        """
        if not x.is_cuda:
            return x / (1 + torch.abs(x))
        return self.softsign_cuda(x)
