import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Custom CUDA kernel for MaxPool1d
# This implementation uses a simple grid-stride loop approach for each output element.
# To handle the large sequence length (65536), we parallelize over (Batch * Features * Output_Length).
# Each thread calculates one output value.
maxpool1d_cuda_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <float.h>

__global__ void maxpool1d_kernel(
    const float* __restrict__ input,
    float* __restrict__ output,
    int batch_size,
    int num_features,
    int input_len,
    int output_len,
    int kernel_size,
    int stride,
    int padding,
    int dilation,
    long long* __restrict__ indices,
    bool return_indices) 
{
    // Calculate global thread index
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total_elements = batch_size * num_features * output_len;

    if (idx < total_elements) {
        // Decompose index
        int out_seq_idx = idx % output_len;
        int remaining = idx / output_len;
        int feat_idx = remaining % num_features;
        int batch_idx = remaining / num_features;

        float max_val = -FLT_MAX;
        int max_idx = -1;

        // Calculate the start position in the input sequence for this output element
        // The output index 'i' corresponds to input index: i * stride - padding
        int start_in_input = out_seq_idx * stride - padding;

        for (int k = 0; k < kernel_size; ++k) {
            int current_input_idx = start_in_input + k * dilation;
            
            if (current_input_idx >= 0 && current_input_idx < input_len) {
                float val = input[((batch_idx * num_features + feat_idx) * input_len) + current_input_idx];
                if (val > max_val) {
                    max_val = val;
                    max_idx = ((batch_idx * num_features + feat_idx) * input_len) + current_input_idx;
                }
            }
        }

        // If the kernel window is entirely in the padding zone, max_val remains -FLT_MAX
        // However, standard PyTorch MaxPool1d behavior for padding is to treat it as -inf.
        // If the window is entirely padding, we return -inf.
        // Note: PyTorch uses -inf for padding.
        if (max_idx == -1) {
            max_val = -INFINITY;
        }

        output[idx] = max_val;

        if (return_indices && indices != nullptr) {
            indices[idx] = (long long)max_idx;
        }
    }
}

torch::Tensor maxpool1d_cuda_forward(
    torch::Tensor input,
    int kernel_size,
    int stride,
    int padding,
    int dilation,
    bool return_indices) 
{
    auto batch_size = input.size(0);
    auto num_features = input.size(1);
    auto input_len = input.size(2);

    // Calculate output length
    // L_out = floor((L_in + 2*padding - dilation*(kernel_size-1) - 1) / stride + 1)
    int output_len = (input_len + 2 * padding - dilation * (kernel_size - 1) - 1) / stride + 1;

    auto output_shape = torch::Tensor({batch_size, num_features, output_len});
    auto output = torch::empty(output_shape, input.options());

    if (return_indices) {
        auto indices = torch::empty(output_shape, torch::Tensor({torch::kInt64}).options());
        
        const int block_size = 256;
        const int num_blocks = (batch_size * num_features * output_len + block_size - 1) / block_size;

        maxpool1d_kernel<<<num_blocks, block_size>>>(
            input.data_ptr<float>(),
            output.data_ptr<float>(),
            batch_size,
            num_features,
            input_len,
            output_len,
            kernel_size,
            stride,
            padding,
            dilation,
            indices.data_ptr<long long>(),
            true
        );
        return torch::cat({output, indices}, 1); // This is not quite right for PyTorch return_indices, 
        // PyTorch returns a tuple (output, indices). We'll handle that in the wrapper.
    } else {
        const int block_size = 256;
        const int num_blocks = (batch_size * num_features * output_len + block_size - 1) / block_size;

        maxpool1d_kernel<<<num_blocks, block_size>>>(
            input.data_ptr<float>(),
            output.data_ptr<float>(),
            batch_size,
            num_features,
            input_len,
            output_len,
            kernel_size,
            stride,
            padding,
            dilation,
            nullptr,
            false
        );
        return output;
    }
}

// To handle the tuple return correctly, we need a specific function for indices
torch::Tensor maxpool1d_cuda_indices_only(
    torch::Tensor input,
    int kernel_size,
    int stride,
    int padding,
    int dilation) 
{
    auto batch_size = input.size(0);
    int num_features = input.size(1);
    int input_len = input.size(2);
    int output_len = (input_len + 2 * padding - dilation * (kernel_size - 1) - 1) / stride + 1;
    auto output_shape = torch::Tensor({batch_size, num_features, output_len});
    auto indices = torch::empty(output_shape, torch::kInt64).to(input.device());

    const int block_size = 256;
    const int num_blocks = (batch_size * num_features * output_len + block_size - 1) / block_size;

    maxpool1d_kernel<<<num_blocks, block_size>>>(
        input.data_ptr<float>(),
        nullptr, // output is not needed if we only want indices, but kernel writes to output
        nullptr, // we'll pass a dummy or just use the kernel to write to indices
        batch_size,
        num_features,
        input_len,
        output_len,
        kernel_size,
        stride,
        padding,
        dilation,
        indices.data_ptr<long long>(),
        true
    );
    // Note: The kernel above is written to write to both. For simplicity in this implementation, 
    // we'll just call the main kernel and return both.
    return indices;
}
