import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# The optimization strategy here is to use a highly optimized Batched Matrix Multiplication.
# While writing a raw CUDA kernel for GEMM from scratch is complex, we can leverage 
# the highly optimized cuBLAS library via the C++ extension to ensure maximum performance 
# for the specific dimensions provided, which is significantly faster than a naive kernel.

cublas_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cublas_v2.h>

torch::Tensor batched_matmul_cuda(torch::Tensor A, torch::Tensor B) {
    // A: (batch, m, k), B: (batch, k, n)
    // Output C: (batch, m, n)
    
    const int batch_size = A.size(0);
    const int m = A.size(1);
    const int k = A.size(2);
    const int n = B.size(2);

    auto C = torch::empty({batch_size, m, n}, A.options());

    cublasHandle_t handle;
    cublasCreate(&handle);

    const float alpha = 1.0f;
    const float beta = 0.0f;

    // cuBLAS uses column-major order. 
    // To perform C = A * B in row-major:
    // We treat A as (k x m) and B as (n x k) in column-major, 
    // which is equivalent to (m x k) and (k x n) in row-major.
    // Or more simply: C^T = B^T * A^T
    // In row-major, A*B is equivalent to cublasSgemm with:
    // transA = CUBLAS_OP_N, transB = CUBLAS_OP_N on the transposed inputs.
    // However, the standard way to do row-major GEMM in cuBLAS is:
    // C = A * B  =>  C^T = B^T * A^T
    // Since we pass pointers to row-major data, we can interpret them as column-major.
    // A (row-major m,k) is A_col (k,m). B (row-major k,n) is B_col (n,k).
    // We want C (row-major m,n) which is C_col (n,m).
    // C_col = B_col * A_col  => (n,m) = (n,k) * (k,m).
    
    for (int b = 0; b < batch_size; ++b) {
        float* ptr_A = A[b].data_ptr<float>();
        float* ptr_B = B[b].data_ptr<float>();
        float* ptr_C = C[b].data_ptr<float>();

        cublasSgemm(handle, 
                    CUBLAS_OP_N, CUBLAS_OP_N, 
                    n, m, k, 
                    &alpha, 
                    ptr_B, n, 
                    ptr_A, k, 
                    &beta, 
                    ptr_C, n);
    }

    cublasDestroy(handle);
    return C;
}
"""

cublas_cpp_source = """
torch::Tensor batched_matmul_cuda(torch::Tensor A, torch::Tensor B);
"""

# Compile the inline CUDA code
batched_matmul_lib = load_inline(
    name="batched_matmul_lib",
    cpp_sources=cublas_cpp_source,
    cuda_sources=cublas_source,
    functions=["batched_matmul_cuda"],
    verbose=False,
)

class ModelNew(nn.Module):
    """
    Optimized version of Model using custom CUDA operator leveraging cuBLAS.
    """
    def __init__(self):
        super(ModelNew, self).__init__()
        self.batched_matmul = batched_matmul_lib.batched_matmul_cuda

    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        """
        Performs batched matrix multiplication using optimized cuBLAS kernel.

        Args:
            A: Input tensor of shape (batch_size, m, k).
            B: Input tensor of shape (batch_size, k, n).

        Returns:
            C: Output tensor of shape (batch_size, m, n).
        """
        # Ensure inputs are contiguous for the CUDA kernel
        if not A.is_contiguous():
            A = A.contiguous()
        if not B.is_contiguous():
            B = B.contiguous()
            
        return self.batched_matmul(A, B)
