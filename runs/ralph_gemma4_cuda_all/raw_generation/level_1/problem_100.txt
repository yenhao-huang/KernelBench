import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Define the custom CUDA kernel for Hinge Loss
# The kernel uses a block-level reduction followed by an atomicAdd to a global scalar.
hinge_loss_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void hinge_loss_kernel(const float* predictions, const float* targets, float* out, int size) {
    extern __shared__ float sdata[];
    __shared__ float val;
    __shared__ float partial_sum;

    int tid = threadIdx.x;
    int idx = blockIdx.x * blockDim.x + threadIdx.x;

    float local_loss = 0.0f;
    if (idx < size) {
        float p = predictions[idx];
        float t = targets[idx];
        float loss_val = 1.0f - p * t;
        local_loss = loss_val > 0.0f ? loss_val : 0.0f;
    }

    // Block-level reduction
    sdata[tid] = local_loss;
    __syncthreads();

    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (tid < s) {
            sdata[tid] += sdata[tid + s];
        }
        __syncthreads();
    }

    if (tid == 0) {
        // Use atomicAdd to add the block's sum to the global output
        atomicAdd(out, sdata[0]);
        // Note: In a real-world scenario, 
        // we would use a more robust reduction 
        // or a more sophisticated grid-level reduction.
        // atomicAdd is sufficient for the 
        // single scalar output for the batch size provided.
    }
}

torch::Tensor hinge_loss_cuda(torch::Tensor predictions, torch::Tensor targets) {
    int size = predictions.numel();
    auto out = torch::zeros({1}, predictions.options());
    
    const int block_size = 256;
    const int num_blocks = (size + block_size - 1) / block_size;

    // Shared memory size: block_size * sizeof(float)
    hinge_loss_kernel<<<num_blocks, block_size, block_size * sizeof(float)>>>(
        predictions.data_ptr<float>(), 
        targets.data_ptr<float>(), 
        out.data_ptr<float>(), 
        size
    );

    return out / (float)size;
}
"""

hinge_loss_cpp_source = "torch::Tensor hinge_loss_cuda(torch::Tensor predictions, torch::Tensor targets);"

# Compile the inline CUDA code
hinge_loss_ext = load_inline(
    name="hinge_loss_ext",
    cpp_sources=hinge_loss_cpp_source,
    cuda_sources=hinge_loss_source,
    functions=["hinge_loss_cuda"],
    verbose=False
)

class ModelNew(nn.Module):
    """
    An optimized model that computes Hinge Loss for binary classification tasks.
    """
    def __init__(self):
        super(ModelNew, self).__init__()
        self.hinge_loss_ext = hinge_loss_ext

    def forward(self, predictions, targets):
        # Ensure inputs are float32 and on CUDA
        return self.hinge_loss_ext.hinge_loss_cuda(predictions.float(), targets.float())
