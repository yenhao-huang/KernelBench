# This kernel performs: 
# 1. Scaled Dot Product (q @ k.T)
# 2. Causal Masking (setting upper triangle to -inf)
# 3. Softmax
# 4. Weighted Sum (att @ v)

# we implement a fused kernel that handles the attention mechanism.
# For high performance, we use a simplified version of the FlashAttention logic:
# We compute the attention scores row by row to avoid storing the large (B, nh, T, T) matrix.

#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math.h>

__global__ void fused_attention_kernel(
    const float* __restrict__ q, // (B, nh, T, hs)
    const float* __restrict__ k, // (B, nh, T, hs)
    const float* __restrict__ v, // (B, nh, T, hs)
    float* __restrict__ out,     // (B, nh, T, hs)
    int B, int nh, int T, int hs, float scale) 
{
    // Each block handles one row (one query vector) of the attention matrix
    // Block index: (batch_idx, head_idx, query_idx)
    int b = blockIdx.z;
    int nh_idx = blockIdx.y;
    int t_q = blockIdx.x;

    if (b >= B || nh_idx >= nh || t_q >= T) return;

    extern __shared__ float shared_mem[];
    // shared_mem will be used for online softmax: [max_val, sum_exp]
    // But for simplicity in this inline version, we'll use registers for the row.
    
    float max_val = -1e20f;
    
    // 1. Compute dot products and find max for numerical stability
    // We need to iterate through all keys (T)
    // Since we can't easily store all scores in shared memory if T is large,
    // we do two passes or use a local buffer.
    
    // For a single thread per row is too slow. We use one warp per row or 
    // one thread per hs. Let's use one thread per hs to compute the dot product.
    
    // Actually, a simpler approach for this specific task:
    // Each block handles one (b, nh, t_q). 
    // We use one thread to compute the entire row's dot products to find max and sum.
    // This is efficient if T is not extremely large.
    
    int tid = threadIdx.x;
    
    // We'll use a single thread per block to compute the row to avoid complex synchronization
    // and ensure correctness for the softmax.
    if (tid == 0) {
        float row_max = -1e20f;
        
        // First pass: find max
        for (int t_k = 0; t_k <= t_q; ++t_k) { // Causal mask: t_k <= t_q
            float score = 0.0f;
            for (int i = 0; i < hs; ++i) {
                score += q[((b * nh + nh_idx) * T + t_q) * hs + i] * 
                         k[((b * nh + nh_idx) * T + t_k) * hs + i];
            }
            score *= scale;
            if (score > row_max) row_max = score;
        }
        
        float sum_exp = 0.0f;
        // Second pass: compute sum of exps
        // We need to store scores to avoid recomputing, but T can be 1024.
        // Let's use a small local array if T is small, or just recompute.
        // Recomputing is safer for memory.
        for (int t_k = 0; t_k <= t_q; ++t_k) {
            float score = 0.0f;
            for (int i = 0; i < hs; ++i) {
                score += q[((b * nh + nh_idx) * T + t_q) * hs + i] * 
                         k[((b * nh + nh_idx) * T + t_k) * hs + i];
            }
            score = (score * scale) - row_max;
            sum_exp += expf(score);
        }

        // Third pass: compute weighted sum of V
        for (int t_k = 0; t_k <= t_q; ++t_k) {
            float score = 0.0f;
            for (int i = 0; i < hs; ++i) {
                score += q[((b * nh + nh_idx) * T + t_q) * hs + i] * 
                         k[((b * nh + nh_idx) * T + t_k) * hs + i];
            }
            float weight = expf((score * scale) - row_max) / sum_exp;
            
            for (int i = 0; i < hs; ++i) {
                out[((b * nh + nh_idx) * T + t_q) * hs + i] += weight * v[((b * nh + nh_idx) * T + t_k) * hs + i];
            }
        }
    }
}

# Final attempt at a clean, working, and efficient kernel:
# Each block handles one row (b, nh, t_q).
# Threads in block = hs.

fused_attention_source_v3 = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math.h>

__global__ void attention_kernel_v3(
    const float* __restrict__ q,
    const float* __restrict__ k,
    const float* __restrict__ v,
    float* __restrict__ out,
    int B, int nh, int T, int hs, float scale) 
{
    int t_q = blockIdx.x;
    int nh_idx = blockIdx.y;
    int b = blockIdx.z;
    int tid = threadIdx.x; // 0 to hs-1

    if (t_q >= T) return;

    int q_offset = ((b * nh + nh_idx) * T + t_q) * hs;
    
    // We need to find max and sum_exp for the row.
    // Since we can't easily do a reduction across t_k with threads being hs,
    // we'll use a single thread to find max and sum_exp, or use a shared memory buffer.
    // Let's use a shared memory buffer for max and sum_exp.
    
    __shared__ float s_max;
    __shared__ float s_sum_exp;

    // To find max and sum_exp, we'll use a single thread (tid=0) to loop over t_k.
    // This is efficient because T is not huge.
    if (tid == 0) {
        float row_max = -1e20f;
        for (int tk = 0; tk <= t_q; ++tk) {
            float score = 0.0f;
            int k_offset = ((b * nh + nh_idx) * T + tk) * hs;
            for (int i = 0; i < hs; ++i) {
                score += q[q_offset + i] * k[k_offset + i];
            }
            score *= scale;
            if (score > row_max) row_max = score;
        }
        s_max = row_max;

        float sum_e = 0.0f;
        for (int tk = 0; tk <= t_q; ++tk) {
            float score = 0.0f;
            int k_offset = ((b * nh + nh_idx) * T + tk) * hs;
            for (int i = 0; i < hs; ++i) {
                score += q[q_offset + i] * k[k_offset + i];
            }
            score *= scale;
            sum_e += expf(score - row_max);
        }
        s_sum_exp = sum_e;
    }
    __syncthreads();

    float row_max = s_max;
    float row_sum_exp = s_sum_exp;

    // Now each thread computes its part of the weighted sum for the hs dimension.
    // We need to loop over t_k again.
    float acc = 0.0f;
    for (int tk = 0; tk <= t_q; ++tk) {
        float score = 0.0f;
        int k_offset = ((b * nh + nh_idx) * T + tk) * hs;
        for (int i = 0; i < hs; ++i) {
            score += q[q_offset + i] * k[k_offset + i];
        }
        float weight = expf(score * scale - row_max) / row_sum_exp;
        acc += weight * v[k_offset + tid];
    }

    out[q_offset + tid] = acc;
}

torch::Tensor attention_cuda(torch::Tensor q, torch::Tensor k, torch::Tensor v, float scale) {
    int B = q.size(0);
    int nh = q.size(1);
    int T = q.size(2);
    int hs = q.size(3);

    auto out = torch::zeros_like(q);

    dim3 block(hs);
    dim3 grid(T, nh, B);

    attention_kernel_v3<<<grid, block>>>(
        q.data_ptr<float>(),
        k.data_ptr<float>(),
        v.data_ptr<float>(),
        out.data_ptr<float>(),
        B, nh, T, hs, scale
    );

    return out;
}
"""

attention_cpp_source = "torch::Tensor attention_cuda(torch::Tensor q, torch::Tensor k, torch::Tensor v, float scale);"

attention_lib = load_inline(
    name="attention_lib",
    cpp_sources=attention_cpp_source,
    cuda_sources=fused_attention_source_v3,
    functions=["attention_cuda"],
    verbose=False,
)

class ModelNew(nn.Module):
    def __init__(self, n_embd, n_head, attn_pdrop, resid_pdrop, max_seqlen):
        super().__init__()
        assert n_embd % n_head == 0
        self.c_attn = nn.Linear(n_embd, 3 * n_embd)
        self.c_proj = nn.Linear(n_embd, n_embd)
        self.attn_dropout = nn.Dropout(attn_pdrop)
        self.resid_dropout = nn.Dropout(resid_pdrop)
        self.register_buffer("bias", torch.tril(torch.ones(max_seqlen, max_seqlen))
                                     .view(1, 1, max_seqlen, max_seqlen))
        self.n_head = n_head
        self.n_embd = n_embd
        self.attention_lib = attention_lib

    def forward(self, x):
        B, T, C = x.size()

        q, k, v = self.c_attn(x).split(self.n_embd, dim=2)
        
        # Reshape for attention: (B, nh, T, hs)
        k = k.view(B, T, self.n_head, C // self.n_head).transpose(1, 2)
        q = q.view(B, T, self.n_head, C // self.n_head).transpose(1, 2)
        v = v.view(B, T, self.n_head, C // self.n_head).transpose(1, 2)

        # Use custom fused kernel
        scale = 1.0 / math.sqrt(k.size(-1))
        y = self.attention_lib.attention_cuda(q, k, v, scale)

        # Re-assemble all head outputs side by side
        y = y.transpose(1, 2).contiguous().view(B, T, C)

        # output projection
        y = self.resid_dropout(self.c_proj(y))
        return y
