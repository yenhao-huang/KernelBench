__global__ void intra_chunk_kernel(
    const float* __restrict__ B,       // [B, C, L, H, N]
    const float* __restrict__ A_cumsum, // [B, H, C, L]
    const float* __restrict__ X,       // [B, C, L, H, P]
    float* __restrict__ states,        // [B, H, C, P, N]
    int B_size, int H, int C, int L, int P, int N) {

    int n = blockIdx.x * blockDim.x + threadIdx.x;
    int p = blockIdx.y * blockDim.y + threadIdx.y;
    int ch = blockIdx.z; // index for (b * H * C)

    if (n >= N || p >= P || ch >= B_size * H * C) return;

    int b = ch / (H * C);
    int h = (ch / C - b * H);
    int c = ch % C;

    float sum = 0.0f;
    float a_last = A_cumsum[((b * H + h) * C + c) * L + (L - 1)];

    for (int l = 0; l < L; ++l) {
        float a_l = A_cumsum[((b * H + h) * C + c) * L + l];
        float decay = expf(a_last - a_l);
        
        // B[b, c, l, h, n]
        float b_val = B[(((b * C + c) * L + l) * H + h) * N + n];
        // X[b, c, l, h, p]
        float x_val = X[(((b * C + c) * L + l) * H + h) * P + p];
        
        sum += b_val * decay * x_val;
    }

    states[((b * H + h) * C + c) * (P * N) + p * N + n] = sum;
}

    int H = B.size(3);
    int N = B.size(4);
    int P = X.size(4);

    auto states = torch::zeros({B_size, H, C, P, N}, B.options());

    dim3 block(16, 16);
    dim3 grid((
