        assignment = F.softmax(assignment, dim=1)

        # assignment: BN x K
        assignment = assignment[:, :self.cluster_size]
        # assignment: B x N x K
        assignment_bnk = assignment.view(batch_size, max_sample, self.cluster_size)
        
        # 3. Compute a_sum for the subtraction part
        # a_sum: B x 1 x K
        a_sum = torch.sum(assignment_bnk, dim=1, keepdim=True)
        
        # Prepare inputs for the custom fused kernel
        # a_sum_flat: B x K
        a_sum_flat = a_sum.view(batch_size, self.cluster_size)
        # clusters2_flat: D x K
        clusters2_flat = self.clusters2.view(self.feature_size, self.cluster_size)
        
        # 4. Fused Aggregation: vlad = (assignment^T @ x) - (a_sum * clusters2)
        # This replaces: transpose, matmul, transpose, and subtraction.
        # Output vlad: B x D x K
        vlad = self.vlad_ops.vlad_aggregation_cuda(
            x, 
            assignment_bnk, 
            clusters2_flat, 
            a_sum_flat
        )
        
        # 5. L2 intra norm (normalize over D dimension)
        vlad = F.normalize(vlad, p=2, dim=1)

        # 6. Flattening + L2 norm (normalize over DK dimension)
        vlad = vlad.reshape(batch_size, -1)
        vlad = F.normalize(vlad, p=2, dim=1)
        
        return vlad
```
