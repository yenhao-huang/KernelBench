import torch
import torch.nn as nn
import torch.nn.functional as F
from torch.utils.cpp_extension import load_inline

# Custom CUDA kernel for Channel Shuffle
# This kernel performs the reshape, transpose, and flatten in a single pass
# to avoid multiple memory copies and kernel launches.
channel_shuffle_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void channel_shuffle_kernel(
    const float* __restrict__ input,
    float* __restrict__ output,
    int batch_size,
    int channels,
    int groups,
    int channels_per_group,
    int height,
    int width) 
{
    // Total elements in the output tensor
    int total_elements = batch_size * channels * height * width;
    int idx = blockIdx.x * blockDim.x + threadIdx.x;

    if (idx < total_elements) {
        // Calculate output coordinates
        int w = idx % width;
        int h = (idx / width) % height;
        int c_out = (idx / (width * height)) % channels;
        int b = idx / (channels * width * height);

        // The shuffle logic:
        // Original view: (B, G, C_per_G, H, W)
        // Transposed view: (B, C_per_G, G, H, W)
        // Flattened view: (B, C_per_G * G, H, W)
        
        // In the output, c_out is the index in the flattened dimension.
        // c_out = g * channels_per_group + c_pg
        int g = c_out / channels_per_group;
        int c_pg = c_out % channels_per_group;

        // Map back to the input index:
        // Input index = b * (channels * H * W) + (g * channels_per_group + c_pg) * H * W + h * W + w
        // Wait, the logic above is for the input. Let's re-derive:
        // The input is (B, G, C_per_G, H, W).
        // The output is (B, C_per_G, G, H, W) flattened.
        // So, output[b][c_pg][g][h][w] comes from input[b][g][c_pg][h][w]
        
        // Let's find the index in the output tensor for (b, c_pg, g, h, w)
        // This is actually what we are calculating.
        // We need to find which input element maps to the current output index 'idx'.
        
        // Let's use the inverse mapping:
        // Output index 'idx' corresponds to:
        // b = idx / (channels * height * width)
        // c_out = (idx / (height * width)) % channels
        // g = c_out / channels_per_group
        // c_pg = c_out % channels_per_group
        // h = (idx / width) % height
        // w = idx % width
        
        // The input index is:
        // input_idx = b * (channels * height * width) + (g * channels_per_group + c_pg) * height * width + h * width + w
        // Wait, the input is (B, G, C_per_G, H, W). 
        // The input index is: b * (G * C_per_G * H * W) + g * (C_per_G * H * W) + c_pg * (H * W) + h * W + w
        // Since G * C_per_G = channels:
        // input_idx = b * (channels * H * W) + g * (channels_per_group * H * W) + c_pg * (H * W) + h * W + w
        
        // Actually, the standard ShuffleNet shuffle is:
        // x = x.view(B, G, C_per_G, H, W).transpose(1, 2).reshape(B, C, H, W)
        // This means output[b][c_out][h][w] = input[b][g][c_pg][h][w]
        // where c_out = c_pg * G + g  <-- NO, that's not right.
        // Let's look at the original code:
        // x = x.view(B, G, C_per_G, H, W)
        // x = x.transpose(1, 2) -> shape (B, C_per_G, G, H, W)
        // x = x.view(B, C, H, W)
        // So, output[b][c_out][h][w] where c_out = c_pg * G + g is WRONG.
        // The transpose(1, 2) swaps G and C_per_G.
        // The new index for c_out is: c_out = c_pg * G + g is also not quite it.
        // Let's trace:
        // Input index: idx_in = b*(G*C_pg*H*W) + g*(C_pg*H*W) + c_pg*(H*W) + h*W + w
        // After transpose(1, 2), the dimension at index 1 is C_pg and index 2 is G.
        // The new index for c_out is: c_out = c_pg * G + g.
        // Wait, if we want to find the input element for a given output index:
        // c_out = c_pg * G + g is not correct for the reshape.
        // The reshape(B, C, H, W) takes the elements in order.
        // The elements in the transposed tensor are ordered by: b, c_pg, g, h, w.
        // So, c_out = c_pg * G + g.
        // Therefore:
        // c_pg = c_out / G
        // g = c_out % G
        
        // Let's re-verify:
        // If G=2, C_pg=2. 
        // Input: [ (g0, c0), (g0, c1), (g1, c0), (g1, c1) ]
        // Transpose(1,2): [ (c0, g0), (c0, g1), (c1, g0), (c1, g1) ]
        // Flatten: [ (c0, g0), (c0, g1), (c1, g0), (c1, g1) ]
        // c_out 0: c_pg=0, g=0
        // c_out 1: c_pg=0, g=1
        // c_out 2: c_pg=1, g=0
        // c_out 3: c_pg=1, g=1
        // This matches c_pg = c_out / G and g = c_out % G.
        
        // Wait, the original code uses:
        // x = x.view(batch_size, self.groups, channels_per_group, height, width)
        // x = x.transpose(1, 2).contiguous()
        // x = x.view(batch_size, -1, height, width)
        // Here, the dimension 1 is 'groups' and dimension 2 is 'channels_per_group'.
        // So transpose(1, 2) makes dimension 1 'channels_per_group' and dimension 2 'groups'.
        // Thus, c_out = c_pg * G + g is WRONG.
        // The correct mapping is: c_out = c_pg * G + g? No.
        // Let's re-read: x.view(B, G, C_pg, H, W).transpose(1, 2) -> (B, C_pg, G, H, W).
        // The flattened index c_out = c_pg * G + g.
        // So:
        // c_pg = c_out / G
        // g = c_out % G
        // Input index: b * (G * C_pg * H * W) + g * (C_pg * H * W) + c_pg * (H * W) + h * W + w
        
        // Let's re-calculate based on the code:
        // x.view(B, G, C_pg, H, W)
        // x.transpose(1, 2) -> (B, C_pg, G, H, W)
        // x.view(B, C_pg * G, H, W)
        // So c_out = c_pg * G + g.
        // This means:
        // c_pg = c_out / G
        // g = c_out % G
        // Input index = b * (channels * H * W) + g * (channels_per_group * H * W) + c_pg * (H * W) + h * W + w
        
        // Wait, the original code:
        // x = x.view(batch_size, self.groups, channels_per_group, height, width)
        // x = x.transpose(1, 2).contiguous()
        // x = x.view(batch_size, -1, height, width)
        // This means the dimension 1 is G, dimension 2 is C_pg.
        // After transpose, dimension 1 is C_pg, dimension 2 is G.
        // The flattened index c_out = c_pg * G + g.
        // So c_pg = c_out / G and g = c_out % G.
        // Let's use this.
        
        // Wait, the input tensor 'input' is (B, C, H, W).
        // The view(B, G, C_pg, H, W) is just a view.
        // The input index is: b * (C * H * W) + (g * C_pg + c_pg) * H * W + h * W + w
        // No, the view is (B, G, C_pg, H, W).
        // So the index in the original (B, C, H, W) tensor is:
        // idx_in = b * (C * H * W) + (g * C_pg + c_pg) * H * W + h * W + w
        // Wait, if the view is (B, G, C_pg, H, W), then the index is:
        // idx_in = b * (G * C_pg * H * W) + g * (C_pg * H * W) + c_pg * (H * W) + h * W + w
        // Since G * C_pg = C, this is the same.
        
        // Let's re-verify the logic one more time.
        // Input: [g0c0, g0c1, g1c0, g1c1] (for B=1, H=1, W=1, G=2, C_pg=2)
        // View (1, 2, 2, 1, 1): [[g0c0, g0c1], [g1c0, g1c1]]
        // Transpose (1, 2): [[g0c0, g1c0], [g0c1, g1c1]]
        // Flatten: [g0c0, g1c0, g0c1, g1c1]
        // c_out 0: g=0, c_pg=0. Input: g0c0. Correct.
        // c_out 1: g=1, c_pg=0. Input: g1c0. Correct.
        // c_out 2: g=0, c_pg=1. Input: g0c1. Correct.
        // c_out 3: g=1, c_pg=1. Input: g1c1. Correct.
        
        // So:
        // c_pg = c_out / G
        // g = c_out % G
        // input_idx = b * (channels * height * width) + (g * channels_per_group + c_pg) * (height * width) + h * width + w
        
        // Wait, in the transpose(1, 2) case:
        // The new dimension 1 is C_pg, dimension 2 is G.
        // The flattened index c_out = c_pg * G + g.
        // This is what I wrote.
        
        // Let's implement.
        
        int G = groups;
        int C_pg = channels_per_group;
        
        int c_pg = c_out / G;
        int g = c_out % G;
        
        long long input_idx = (long long)b * channels * height * width + 
                              (long long)(g * C_pg + c_pg) * height * width + 
                              (long long)h * width + w;
        
        output[idx] = input[input_idx];
    }
}

__global__ void channel_shuffle_kernel_v2(
    const float* __restrict__ input,
    float* __restrict__ output,
    int batch_size,
    int channels,
    int groups,
    int channels_per_group,
    int height,
    int width) 
{
    int total_elements = batch_size * channels * height * width;
    int idx = blockIdx.x * blockDim.x + threadIdx.x;

    if (idx < total_elements) {
        int w = idx % width;
        int h = (idx / width) % height;
        int c_out = (idx / (width * height)) % channels;
        int b = idx / (channels * width * height);

        int c_pg = c_out / groups;
        int g = c_out % groups;

        long long input_idx = (long long)b * channels * height * width + 
                              (long long)(g * channels_per_group + c_pg) * height * width + 
                              (long long)h * width + w;
        
        output[idx] = input[input_idx];
    }
}

torch::Tensor channel_shuffle_cuda(torch::Tensor x, int groups) {
    auto batch_size = x.size(0);
    auto channels = x.size(1);
    auto height = x.size(2);
    auto width = x.size(3);
    auto channels_per_group = channels / groups;
    
    auto out = torch::empty_like(x);

    const int block_size = 256;
    int total_elements = batch_size * channels * height * width;
    int num_blocks = (total_elements + block_size - 1) / block_size;

    channel_shuffle_kernel_v2<<<num_blocks, block_size>>>(
        x.data_ptr<float>(), 
        out.data_ptr<float>(), 
        batch_size, 
        channels, 
        groups, 
        channels_per_group, 
        height, 
        width
    );

    return out;
}
"""

channel_shuffle_cpp_source = (
    "torch::Tensor channel_shuffle_cuda(torch::Tensor x, int groups);"
)

channel_shuffle_lib = load_inline(
    name="channel_shuffle_lib",
    cpp_sources=channel_shuffle_cpp_source,
    cuda_sources=channel_shuffle_source,
    functions=["channel_shuffle_cuda"],
    verbose=False,
)

class ChannelShuffleNew(nn.Module):
    def __init__(self, groups):
        super(ChannelShuffleNew, self).__init__()
        self.groups = groups
        self.lib = channel_shuffle_lib
    
    def forward(self, x):
        return self.lib.channel_shuffle_cuda(x, self.groups)

class ModelNew(nn.Module):
    def __init__(self, in_channels, out_channels, groups=3):
        super(ModelNew, self).__init__()
        
        assert out_channels % 4 == 0
        mid_channels = out_channels // 4
        
        self.conv1 = nn.Conv2d(in_channels, mid_channels, kernel_size=1, stride=1, padding=0, groups=groups, bias=False)
        self.bn1 = nn.BatchNorm2d(mid_channels)
        
        self.conv2 = nn.Conv2d(mid_channels, mid_channels, kernel_size=3, stride=1, padding=1, groups=mid_channels, bias=False)
        self.bn2 = nn.BatchNorm2d(mid_channels)
        
        self.conv3 = nn.Conv2d(mid_channels, out_channels, kernel_size=1, stride=1, padding=0, groups=groups, bias=False)
        self.bn3 = nn.BatchNorm2d(out_channels)
        
        self.shuffle = ChannelShuffleNew(groups)
        
        if in_channels == out_channels:
            self.shortcut = nn.Sequential()
        else:
            self.shortcut = nn.Sequential(
                nn.Conv2d(in_channels, out_channels, kernel_size=1, stride=1, padding=0, bias=False),
                nn.BatchNorm2d(out_channels)
            )
    
    def forward(self, x):
        # We keep the standard convolutions and BN as they are highly optimized in cuDNN
        # The main custom optimization is the ChannelShuffle which involves multiple 
        # reshape/transpose/contiguous operations that trigger multiple kernel launches 
        # and memory copies.
        out = F.relu(self.bn1(self.conv1(x)))
        out = self.bn2(self.conv2(out))
        out = self.shuffle(out)
        out = F.relu(self.bn3(self.conv3(out)))
        
        out += self.shortcut(x)
        return out
