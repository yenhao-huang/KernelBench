import torch

cuda_sources = r"""
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math.h>

__device__ __forceinline__ float sigmoidf_fast(float x) {
    return 1.0f / (1.0f + expf(-x));
}

__global__ void lstm_layer_kernel(
    const float* __restrict__ inp,
    const float* __restrict__ h0,
    const float* __restrict__ c0,
    const float* __restrict__ w_ih,
    const float* __restrict__ w_hh,
    const float* __restrict__ b_ih,
    const float* __restrict__ b_hh,
    float* __restrict__ out,
    float* __restrict__ c_final,
    int B, int T, int I, int H, int layer
) {
    int b = blockIdx.x;
    int hid = threadIdx.x;

    extern __shared__ float shared[];
    float* h = shared;
    float* c = shared + H;

    if (hid < H) {
        h[hid] = h0[(layer * B + b) * H + hid];
        c[hid] = c0[(layer * B + b) * H + hid];
    }
    __syncthreads();

    for (int t = 0; t < T; ++t) {
        float ni = 0.0f;
        float nf = 0.0f;
        float ng = 0.0f;
        float no = 0.0f;

        if (hid < H) {
            const float* x_t = inp + (b * T + t) * I;

            ni = b_ih[hid] + b_hh[hid];
            nf = b_ih[H + hid] + b_hh[H + hid];
            ng = b_ih[2 * H + hid] + b_hh[2 * H + hid];
            no = b_ih[3 * H + hid] + b_hh[3 * H + hid];

            const float* wi = w_ih + hid * I;
            const float* wf = w_ih + (H + hid) * I;
            const float* wg = w_ih + (2 * H + hid) * I;
            const float* wo = w_ih + (3 * H + hid) * I;

            #pragma unroll 4
            for (int j = 0; j < I; ++j) {
                float xv = x_t[j];
                ni += wi[j] * xv;
                nf += wf[j] * xv;
                ng += wg[j] * xv;
                no += wo[j] * xv;
            }

            const float* whi = w_hh + hid * H;
            const float* whf = w_hh + (H + hid) * H;
            const float* whg = w_hh + (2 * H + hid) * H;
            const float* who = w_hh + (3 * H + hid) * H;

            #pragma unroll 4
            for (int j = 0; j < H; ++j) {
                float hv = h[j];
                ni += whi[j] * hv;
                nf += whf[j] * hv;
                ng += whg[j] * hv;
                no += who[j] * hv;
            }

            float i_gate = sigmoidf_fast(ni);
            float f_gate = sigmoidf_fast(nf);
            float g_gate = tanhf(ng);
            float o_gate = sigmoidf_fast(no);

            float new_c = f_gate * c[hid] + i_gate * g_gate;
            float new_h = o_gate * tanhf(new_c);

            __syncthreads();
            c[hid] = new_c;
            h[hid] = new_h;
            out[(b * T + t) * H + hid] = new_h;
        }
        __syncthreads();
    }

    if (hid < H) {
        c_final[(layer * B + b) * H + hid] = c[hid];
    }
}

static void launch_layer(
    torch::Tensor inp,
    torch::Tensor h0,
    torch::Tensor c0,
    torch::Tensor w_ih,
    torch::Tensor w_hh,
    torch::Tensor b_ih,
    torch::Tensor b_hh,
    torch::Tensor out,
    torch::Tensor c_final,
    int layer,
    int input_dim
) {
    int B = inp.size(0);
    int T = inp.size(1);
    int H = h0.size(2);
    int threads = H;
    int shared_bytes = 2 * H * sizeof(float);

    lstm_layer_kernel<<<B, threads, shared_bytes>>>(
        inp.data_ptr<float>(),
        h0.data_ptr<float>(),
        c0.data_ptr<float>(),
        w_ih.data_ptr<float>(),
        w_hh.data_ptr<float>(),
        b_ih.data_ptr<float>(),
        b_hh.data_ptr<float>(),
        out.data_ptr<float>(),
        c_final.data_ptr<float>(),
        B, T, input_dim, H, layer
    );
}

torch::Tensor lstm6_forward_cuda(
    torch::Tensor x,
    torch::Tensor h0,
    torch::Tensor c0,
    torch::Tensor w_ih0, torch::Tensor w_hh0, torch::Tensor b_ih0, torch::Tensor b_hh0,
    torch::Tensor w_ih1, torch::Tensor w_hh1, torch::Tensor b_ih1, torch::Tensor b_hh1,
    torch::Tensor w_ih2, torch::Tensor w_hh2, torch::Tensor b_ih2, torch::Tensor b_hh2,
    torch::Tensor w_ih3, torch::Tensor w_hh3, torch::Tensor b_ih3, torch::Tensor b_hh3,
    torch::Tensor w_ih4, torch::Tensor w_hh4, torch::Tensor b_ih4, torch::Tensor b_hh4,
    torch::Tensor w_ih5, torch::Tensor w_hh5, torch::Tensor b_ih5, torch::Tensor b_hh5
) {
    int B = x.size(0);
    int T = x.size(1);
    int H = h0.size(2);

    auto opts = x.options();
    auto buf_a = torch::empty({B, T, H}, opts);
    auto buf_b = torch::empty({B, T, H}, opts);
    auto c_final = torch::empty_like(c0);

    launch_layer(x,     h0, c0, w_ih0, w_hh0, b_ih0, b_hh0, buf_a, c_final, 0, x.size(2));
    launch_layer(buf_a, h0, c0, w_ih1, w_hh1, b_ih1, b_hh1, buf_b, c_final, 1, H);
    launch_layer(buf_b, h0, c0, w_ih2, w_hh2, b_ih2, b_hh2, buf_a, c_final, 2, H);
    launch_layer(buf_a, h0, c0, w_ih3, w_hh3, b_ih3, b_hh3, buf_b, c_final, 3, H);
    launch_layer(buf_b, h0, c0, w_ih4, w_hh4, b_ih4, b_hh4, buf_a, c_final, 4, H);
    launch_layer(buf_a, h0, c0, w_ih5, w_hh5, b_ih5, b_hh5, buf_b, c_final, 5, H);

    return c_final;
}
"""

    def __init__(self, input_size, hidden_size, num_layers, output_size, dropout=0.0):
        super().__init__()
        self.lstm = nn.LSTM(
            input_size,
            hidden_size,
            num_layers,
            batch_first=True,
            dropout=dropout,
            bidirectional=False,
