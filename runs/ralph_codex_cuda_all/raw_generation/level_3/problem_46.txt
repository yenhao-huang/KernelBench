import math
import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_sources = r"""
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cmath>

#define CHECK_CUDA(x) TORCH_CHECK(x.is_cuda(), #x " must be CUDA")
#define CHECK_CONTIGUOUS(x) TORCH_CHECK(x.is_contiguous(), #x " must be contiguous")

__global__ void matmul_kernel(
    const float* __restrict__ x,
    const float* __restrict__ clusters,
    float* __restrict__ logits,
    int M, int D, int C
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = M * C;
    if (idx >= total) return;

    int c = idx % C;
    int m = idx / C;

    float acc = 0.0f;
    const float* xp = x + m * D;
    const float* cp = clusters + c;
    #pragma unroll 4
    for (int d = 0; d < D; ++d) {
        acc += xp[d] * cp[d * C];
    }
    logits[idx] = acc;
}

__global__ void mean_var_kernel(
    const float* __restrict__ logits,
    float* __restrict__ mean,
    float* __restrict__ invstd,
    int M, int C
) {
    int c = blockIdx.x;
    float sum = 0.0f;
    float sumsq = 0.0f;

    for (int m = threadIdx.x; m < M; m += blockDim.x) {
        float v = logits[m * C + c];
        sum += v;
        sumsq += v * v;
    }

    __shared__ float ssum[256];
    __shared__ float ssq[256];
    ssum[threadIdx.x] = sum;
    ssq[threadIdx.x] = sumsq;
    __syncthreads();

    for (int stride = blockDim.x >> 1; stride > 0; stride >>= 1) {
        if (threadIdx.x < stride) {
            ssum[threadIdx.x] += ssum[threadIdx.x + stride];
            ssq[threadIdx.x] += ssq[threadIdx.x + stride];
        }
        __syncthreads();
    }

    if (threadIdx.x == 0) {
        float mu = ssum[0] / (float)M;
        float ex2 = ssq[0] / (float)M;
        float var = fmaxf(ex2 - mu * mu, 0.0f);
        mean[c] = mu;
        invstd[c] = rsqrtf(var + 1.0e-5f);
    }
}

__global__ void softmax_kernel(
    const float* __restrict__ logits,
    const float* __restrict__ mean,
    const float* __restrict__ invstd,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    float* __restrict__ assignment,
    int M, int C, int K
) {
    int m = blockIdx.x * blockDim.x + threadIdx.x;
    if (m >= M) return;

    float vals[64];
    float mx = -3.402823466e38f;

    for (int c = 0; c < C; ++c) {
        float v = (logits[m * C + c] - mean[c]) * invstd[c] * weight[c] + bias[c];
        vals[c] = v;
        mx = fmaxf(mx, v);
    }

    float denom = 0.0f;
    for (int c = 0; c < C; ++c) {
        float e = expf(vals[c] - mx);
        vals[c] = e;
        denom += e;
    }

    float invden = 1.0f / denom;
    for (int k = 0; k < K; ++k) {
        assignment[m * K + k] = vals[k] * invden;
    }
}

__global__ void vlad_kernel(
    const float* __restrict__ x,
    const float* __restrict__ assignment,
    const float* __restrict__ clusters2,
    float* __restrict__ out,
    int B, int N, int D, int K
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = B * D * K;
    if (idx >= total) return;

    int k = idx % K;
    int d = (idx / K) % D;
    int b = idx / (D * K);

    float sum_ax = 0.0f;
    float sum_a = 0.0f;

    int base_m = b * N;
    for (int n = 0; n < N; ++n) {
        float a = assignment[(base_m + n) * K + k];
        float xv = x[(base_m + n) * D + d];
        sum_ax += a * xv;
        sum_a += a;
    }

    out[idx] = sum_ax - sum_a * clusters2[d * K + k];
}

__global__ void intra_norm_kernel(
    float* __restrict__ vlad,
    int B, int D, int K
) {
    int idx = blockIdx.x;
    int b = idx / K;
    int k = idx % K;

    float sum = 0.0f;
    for (int d = threadIdx.x; d < D; d += blockDim.x) {
        float v = vlad[(b * D + d) * K + k];
        sum += v * v;
    }

    __shared__ float ss[256];
    ss[threadIdx.x] = sum;
    __syncthreads();

    for (int stride = blockDim.x >> 1; stride > 0; stride >>= 1) {
        if (threadIdx.x < stride) ss[threadIdx.x] += ss[threadIdx.x + stride];
        __syncthreads();
    }

    float inv = rsqrtf(ss[0] + 1.0e-12f);
    for (int d = threadIdx.x; d < D; d += blockDim.x) {
        vlad[(b * D + d) * K + k] *= inv;
    }
}

__global__ void final_norm_kernel(
    const float* __restrict__ vlad,
    float* __restrict__ out,
    int B, int DK
) {
    int b = blockIdx.x;
    float sum = 0.0f;

    for (int i = threadIdx.x; i < DK; i += blockDim.x) {
        float v = vlad[b * DK + i];
        sum += v * v;
    }

    __shared__ float ss[256];
    ss[threadIdx.x] = sum;
    __syncthreads();

    for (int stride = blockDim.x >> 1; stride > 0; stride >>= 1) {
        if (threadIdx.x < stride) ss[threadIdx.x] += ss[threadIdx.x + stride];
        __syncthreads();
    }

    float inv = rsqrtf(ss[0] + 1.0e-12f);
    for (int i = threadIdx.x; i < DK; i += blockDim.x) {
        out[b * DK + i] = vlad[b * DK + i] * inv;
    }
}

torch::Tensor netvlad_cuda(
    torch::Tensor x,
    torch::Tensor clusters,
    torch::Tensor bn_weight,
    torch::Tensor bn_bias,
    torch::Tensor clusters2
) {
    CHECK_CUDA(x);
    CHECK_CUDA(clusters);
    CHECK_CUDA(bn_weight);
    CHECK_CUDA(bn_bias);
    CHECK_CUDA(clusters2);

    x = x.contiguous();
    clusters = clusters.contiguous();
    clusters2 = clusters2.contiguous();

    int B = (int)x.size(0);
    int N = (int)x.size(1);
    int D = (int)x.size(2);
    int C = (int)clusters.size(1);
    int K = (int)clusters2.size(2);
    int M = B * N;
    int DK = D * K;

    auto opts = x.options();
    auto logits = torch::empty({M, C}, opts);
    auto mean = torch::empty({C}, opts);
    auto invstd = torch::empty({C}, opts);
    auto assignment = torch::empty({M, K}, opts);
    auto vlad = torch::empty({B, D, K}, opts);
    auto out = torch::empty({B, DK}, opts);

    int threads = 256;
    matmul_kernel<<<(M * C + threads - 1) / threads, threads>>>(
        x.data_ptr<float>(), clusters.data_ptr<float>(), logits.data_ptr<float>(), M, D, C
    );

    mean_var_kernel<<<C, 256>>>(
        logits.data_ptr<float>(), mean.data_ptr<float>(), invstd.data_ptr<float>(), M, C
    );

    softmax_kernel<<<(M + threads - 1) / threads, threads>>>(
        logits.data_ptr<float>(), mean.data_ptr<float>(), invstd.data_ptr<float>(),
        bn_weight.data_ptr<float>(), bn_bias.data_ptr<float>(),
        assignment.data_ptr<float>(), M, C, K
    );

    vlad_kernel<<<(B * D * K + threads - 1) / threads, threads>>>(
        x.data_ptr<float>(), assignment.data_ptr<float>(), clusters2.data_ptr<float>(),
        vlad.data_ptr<float>(), B, N, D, K
    );

    intra_norm_kernel<<<B * K, 256>>>(vlad.data_ptr<float>(), B, D, K);
    final_norm_kernel<<<B, 256>>>(vlad.data_ptr<float>(), out.data_ptr<float>(), B, DK);

    return out;
}
"""

cpp_sources = r"""
torch::Tensor netvlad_cuda(
    torch::Tensor x,
    torch::Tensor clusters,
    torch::Tensor bn_weight,
    torch::Tensor bn_bias,
    torch::Tensor clusters2
);
"""

netvlad_ext = load_inline(
    name="netvlad_custom_cuda_ext",
    cpp_sources=cpp_sources,
    cuda_sources=cuda_sources,
    functions=["netvlad_cuda"],
    verbose=False,
    extra_cflags=["-O3"],
    extra_cuda_cflags=["-O3", "--use_fast_math"],
)


class ModelNew(nn.Module):
    def __init__(self, cluster_size, feature_size, ghost_clusters):
        super(ModelNew, self).__init__()
        self.feature_size = feature_size
        self.cluster_size = cluster_size
        self.ghost_clusters = ghost_clusters

        init_sc = 1 / math.sqrt(feature_size)
        clusters = cluster_size + ghost_clusters

        self.clusters = nn.Parameter(init_sc * torch.randn(feature_size, clusters))
        self.batch_norm = nn.BatchNorm1d(clusters)
        self.clusters2 = nn.Parameter(init_sc * torch.randn(1, feature_size, cluster_size))
        self.out_dim = self.cluster_size * feature_size
        self.netvlad_ext = netvlad_ext

    def forward(self, x, mask=None):
        return self.netvlad_ext.netvlad_cuda(
            x,
            self.clusters,
            self.batch_norm.weight,
            self.batch_norm.bias,
            self.clusters2,
        )
