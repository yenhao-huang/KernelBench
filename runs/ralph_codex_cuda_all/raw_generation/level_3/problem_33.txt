import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_sources = r"""
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math.h>

#define TILE_M 16
#define TILE_N 16
#define TILE_K 16

__global__ void rnn_i2h_tanh_kernel(
    const float* __restrict__ x,
    const float* __restrict__ hprev,
    const float* __restrict__ w,
    const float* __restrict__ bias,
    float* __restrict__ hnew,
    int B, int I, int H
) {
    __shared__ float As[TILE_M][TILE_K];
    __shared__ float Bs[TILE_K][TILE_N];

    int tx = threadIdx.x;
    int ty = threadIdx.y;
    int row = blockIdx.y * TILE_M + ty;
    int col = blockIdx.x * TILE_N + tx;

    float acc = 0.0f;
    int K = I + H;

    for (int k0 = 0; k0 < K; k0 += TILE_K) {
        int k_a = k0 + tx;
        int k_b = k0 + ty;

        if (row < B && k_a < K) {
            As[ty][tx] = (k_a < I) ? x[row * I + k_a] : hprev[row * H + (k_a - I)];
        } else {
            As[ty][tx] = 0.0f;
        }

        if (k_b < K && col < H) {
            Bs[ty][tx] = w[col * K + k_b];
        } else {
            Bs[ty][tx] = 0.0f;
        }

        __syncthreads();

        #pragma unroll
        for (int kk = 0; kk < TILE_K; ++kk) {
            acc += As[ty][kk] * Bs[kk][tx];
        }

        __syncthreads();
    }

    if (row < B && col < H) {
        acc += bias[col];
        hnew[row * H + col] = tanhf(acc);
    }
}

__global__ void linear_kernel(
    const float* __restrict__ a,
    const float* __restrict__ w,
    const float* __restrict__ bias,
    float* __restrict__ out,
    int B, int K, int O
) {
    __shared__ float As[TILE_M][TILE_K];
    __shared__ float Bs[TILE_K][TILE_N];

    int tx = threadIdx.x;
    int ty = threadIdx.y;
    int row = blockIdx.y * TILE_M + ty;
    int col = blockIdx.x * TILE_N + tx;

    float acc = 0.0f;

    for (int k0 = 0; k0 < K; k0 += TILE_K) {
        int ka = k0 + tx;
        int kb = k0 + ty;

        As[ty][tx] = (row < B && ka < K) ? a[row * K + ka] : 0.0f;
        Bs[ty][tx] = (kb < K && col < O) ? w[col * K + kb] : 0.0f;

        __syncthreads();

        #pragma unroll
        for (int kk = 0; kk < TILE_K; ++kk) {
            acc += As[ty][kk] * Bs[kk][tx];
        }

        __syncthreads();
    }

    if (row < B && col < O) {
        out[row * O + col] = acc + bias[col];
    }
}

torch::Tensor rnn_i2h_tanh_cuda(torch::Tensor x, torch::Tensor hprev, torch::Tensor w, torch::Tensor bias) {
    int B = x.size(0);
    int I = x.size(1);
    int H = hprev.size(1);

    auto hnew = torch::empty({B, H}, x.options());

    dim3 block(TILE_N, TILE_M);
    dim3 grid((H + TILE_N - 1) / TILE_N, (B + TILE_M - 1) / TILE_M);

    rnn_i2h_tanh_kernel<<<grid, block>>>(
        x.data_ptr<float>(),
        hprev.data_ptr<float>(),
        w.data_ptr<float>(),
        bias.data_ptr<float>(),
        hnew.data_ptr<float>(),
        B, I, H
    );

    return hnew;
}

torch::Tensor linear_cuda(torch::Tensor a, torch::Tensor w, torch::Tensor bias) {
    int B = a.size(0);
    int K = a.size(1);
    int O = w.size(0);

    auto out = torch::empty({B, O}, a.options());

    dim3 block(TILE_N, TILE_M);
    dim3 grid((O + TILE_N - 1) / TILE_N, (B + TILE_M - 1) / TILE_M);

    linear_kernel<<<grid, block>>>(
        a.data_ptr<float>(),
        w.data_ptr<float>(),
        bias.data_ptr<float>(),
        out.data_ptr<float>(),
        B, K, O
    );

    return out;
}
"""

cpp_sources = r"""
torch::Tensor rnn_i2h_tanh_cuda(torch::Tensor x, torch::Tensor hprev, torch::Tensor w, torch::Tensor bias);
torch::Tensor linear_cuda(torch::Tensor a, torch::Tensor w, torch::Tensor bias);
"""

rnn_cuda = load_inline(
    name="vanilla_rnn_custom_cuda",
    cpp_sources=cpp_sources,
    cuda_sources=cuda_sources,
    functions=["rnn_i2h_tanh_cuda", "linear_cuda"],
    verbose=False,
    extra_cflags=["-O3"],
    extra_cuda_cflags=["-O3", "--use_fast_math"],
)


class ModelNew(nn.Module):
    def __init__(self, input_size: int, hidden_size: int, output_size: int):
        super(ModelNew, self).__init__()
        self.input_size = input_size
        self.hidden_size = hidden_size
        self.output_size = output_size
        self.hidden = torch.randn((256, hidden_size))
        self.i2h = nn.Linear(input_size + hidden_size, hidden_size)
        self.h2o = nn.Linear(hidden_size, output_size)

    def forward(self, x: torch.Tensor, initial_hidden=None) -> torch.Tensor:
        if initial_hidden is not None:
            hidden = initial_hidden.contiguous()
        else:
            hidden = self.hidden.to(device=x.device, dtype=x.dtype).contiguous()

        x = x.contiguous()
        self.hidden = rnn_cuda.rnn_i2h_tanh_cuda(
            x,
            hidden,
            self.i2h.weight.contiguous(),
            self.i2h.bias.contiguous(),
        )
        output = rnn_cuda.linear_cuda(
            self.hidden,
            self.h2o.weight.contiguous(),
            self.h2o.bias.contiguous(),
        )
        return output
