import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cpp_sources = """
torch::Tensor linear_bias_cuda(torch::Tensor x, torch::Tensor w, torch::Tensor b, bool relu);
"""

cuda_sources = """
#include <torch/extension.h>
#include <cuda_runtime.h>

#define BM 16
#define BN 16
#define BK 32

__global__ void linear_bias_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ b,
    float* __restrict__ y,
    int M, int K, int N,
    bool relu
) {
    __shared__ float xs[BM][BK];
    __shared__ float ws[BN][BK];

    int row = blockIdx.y * BM + threadIdx.y;
    int col = blockIdx.x * BN + threadIdx.x;

    float acc = 0.0f;

    for (int t = 0; t < K; t += BK) {
        int xk = t + threadIdx.x;
        int wk = t + threadIdx.y;

        xs[threadIdx.y][threadIdx.x] = (row < M && xk < K) ? x[row * K + xk] : 0.0f;
        ws[threadIdx.x][threadIdx.y] = (col < N && wk < K) ? w[col * K + wk] : 0.0f;

        __syncthreads();

        #pragma unroll
        for (int k = 0; k < BK; ++k) {
            acc += xs[threadIdx.y][k] * ws[threadIdx.x][k];
        }

        __syncthreads();
    }

    if (row < M && col < N) {
        acc += b[col];
        if (relu && acc < 0.0f) acc = 0.0f;
        y[row * N + col] = acc;
    }
}

torch::Tensor linear_bias_cuda(torch::Tensor x, torch::Tensor w, torch::Tensor b, bool relu) {
    int M = (int)x.size(0);
    int K = (int)x.size(1);
    int N = (int)w.size(0);

    auto y = torch::empty({M, N}, x.options());

    dim3 block(BN, BM);
    dim3 grid((N + BN - 1) / BN, (M + BM - 1) / BM);

    linear_bias_kernel<<<grid, block>>>(
        x.data_ptr<float>(),
        w.data_ptr<float>(),
        b.data_ptr<float>(),
        y.data_ptr<float>(),
        M, K, N, relu
    );

    return y;
}
"""

mlp_cuda = load_inline(
    name="kernelbench_mlp_linear_bias_relu_v1",
    cpp_sources=cpp_sources,
    cuda_sources=cuda_sources,
    functions=["linear_bias_cuda"],
    verbose=False,
    extra_cuda_cflags=["-O3", "--use_fast_math"],
)


class ModelNew(nn.Module):
    def __init__(self, input_size, hidden_layer_sizes, output_size):
        super(ModelNew, self).__init__()

        sizes = [input_size] + list(hidden_layer_sizes) + [output_size]
        self.layers = nn.ModuleList(
            [nn.Linear(sizes[i], sizes[i + 1]) for i in range(len(sizes) - 1)]
        )
        self.op = mlp_cuda

    def forward(self, x):
        for i, layer in enumerate(self.layers):
            x = self.op.linear_bias_cuda(
                x,
                layer.weight,
                layer.bias,
                i != len(self.layers) - 1,
            )
        return x
