import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_sources = r"""
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <vector>

__global__ void dense_layer_eval_kernel(
    const float* __restrict__ x,
    const float* __restrict__ bn_w,
    const float* __restrict__ bn_b,
    const float* __restrict__ mean,
    const float* __restrict__ var,
    const float* __restrict__ conv_w,
    float* __restrict__ out,
    int N, int C, int H, int W, int G, float eps
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * G * H * W;
    if (idx >= total) return;

    int w = idx % W;
    int h = (idx / W) % H;
    int oc = (idx / (W * H)) % G;
    int n = idx / (G * H * W);

    float acc = 0.0f;
    for (int ic = 0; ic < C; ++ic) {
        float scale = bn_w[ic] * rsqrtf(var[ic] + eps);
        float bias = bn_b[ic] - mean[ic] * scale;

        #pragma unroll
        for (int kh = 0; kh < 3; ++kh) {
            int ih = h + kh - 1;
            if (ih < 0 || ih >= H) continue;
            #pragma unroll
            for (int kw = 0; kw < 3; ++kw) {
                int iw = w + kw - 1;
                if (iw < 0 || iw >= W) continue;

                float v = x[((n * C + ic) * H + ih) * W + iw];
                v = v * scale + bias;
                v = v > 0.0f ? v : 0.0f;
                float wt = conv_w[((oc * C + ic) * 3 + kh) * 3 + kw];
                acc += v * wt;
            }
        }
    }
    out[idx] = acc;
}

__global__ void append_channels_kernel(
    const float* __restrict__ x,
    const float* __restrict__ y,
    float* __restrict__ out,
    int N, int C, int G, int H, int W
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * (C + G) * H * W;
    if (idx >= total) return;

    int spatial = H * W;
    int s = idx % spatial;
    int c = (idx / spatial) % (C + G);
    int n = idx / ((C + G) * spatial);

    if (c < C) {
        out[idx] = x[(n * C + c) * spatial + s];
    } else {
        out[idx] = y[(n * G + (c - C)) * spatial + s];
    }
}

torch::Tensor dense_layer_eval_cuda(
    torch::Tensor x,
    torch::Tensor bn_w,
    torch::Tensor bn_b,
    torch::Tensor mean,
    torch::Tensor var,
    torch::Tensor conv_w,
    double eps
) {
    int N = x.size(0);
    int C = x.size(1);
    int H = x.size(2);
    int W = x.size(3);
    int G = conv_w.size(0);

    auto out = torch::empty({N, G, H, W}, x.options());
    int total = N * G * H * W;
    int block = 256;
    int grid = (total + block - 1) / block;

    dense_layer_eval_kernel<<<grid, block>>>(
        x.data_ptr<float>(),
        bn_w.data_ptr<float>(),
        bn_b.data_ptr<float>(),
        mean.data_ptr<float>(),
        var.data_ptr<float>(),
        conv_w.data_ptr<float>(),
        out.data_ptr<float>(),
        N, C, H, W, G, (float)eps
    );

    return out;
}

torch::Tensor append_channels_cuda(torch::Tensor x, torch::Tensor y) {
    int N = x.size(0);
    int C = x.size(1);
    int H = x.size(2);
    int W = x.size(3);
    int G = y.size(1);

    auto out = torch::empty({N, C + G, H, W}, x.options());
    int total = N * (C + G) * H * W;
    int block = 256;
    int grid = (total + block - 1) / block;

    append_channels_kernel<<<grid, block>>>(
        x.data_ptr<float>(),
        y.data_ptr<float>(),
        out.data_ptr<float>(),
        N, C, G, H, W
    );

    return out;
}
"""

cpp_sources = r"""
torch::Tensor dense_layer_eval_cuda(
    torch::Tensor x,
    torch::Tensor bn_w,
    torch::Tensor bn_b,
    torch::Tensor mean,
    torch::Tensor var,
    torch::Tensor conv_w,
    double eps
);
torch::Tensor append_channels_cuda(torch::Tensor x, torch::Tensor y);
"""

dense_cuda = load_inline(
    name="dense_block_custom_cuda",
    cpp_sources=cpp_sources,
    cuda_sources=cuda_sources,
    functions=["dense_layer_eval_cuda", "append_channels_cuda"],
    verbose=False,
    extra_cflags=["-O3"],
    extra_cuda_cflags=["-O3", "--use_fast_math"],
)


class ModelNew(nn.Module):
    def __init__(self, num_layers: int, num_input_features: int, growth_rate: int):
        super(ModelNew, self).__init__()
        layers = []
        for i in range(num_layers):
            in_features = num_input_features + i * growth_rate
            layers.append(nn.Sequential(
                nn.BatchNorm2d(in_features),
                nn.ReLU(inplace=True),
                nn.Conv2d(in_features, growth_rate, kernel_size=3, padding=1, bias=False),
                nn.Dropout(0.0),
            ))
        self.layers = nn.ModuleList(layers)
        self.op = dense_cuda

    def forward(self, x):
        for layer in self.layers:
            bn = layer[0]
            conv = layer[2]
            y = self.op.dense_layer_eval_cuda(
                x.contiguous(),
                bn.weight,
                bn.bias,
                bn.running_mean,
                bn.running_var,
                conv.weight,
                bn.eps,
            )
            x = self.op.append_channels_cuda(x, y)
        return x
