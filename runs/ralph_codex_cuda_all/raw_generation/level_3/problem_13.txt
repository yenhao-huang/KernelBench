import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_sources = r"""
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math.h>

__global__ void fused_transition_eval_kernel(
    const float* __restrict__ x,
    const float* __restrict__ bn_w,
    const float* __restrict__ bn_b,
    const float* __restrict__ running_mean,
    const float* __restrict__ running_var,
    const float* __restrict__ conv_w,
    float* __restrict__ out,
    int N, int Cin, int H, int W, int Cout,
    float eps
) {
    int OW = W >> 1;
    int OH = H >> 1;
    int total = N * Cout * OH * OW;
    int idx = blockIdx.x * blockDim.x + threadIdx.x;

    if (idx >= total) return;

    int ow = idx % OW;
    int t = idx / OW;
    int oh = t % OH;
    t /= OH;
    int co = t % Cout;
    int n = t / Cout;

    float acc_pool = 0.0f;

    #pragma unroll
    for (int dy = 0; dy < 2; ++dy) {
        #pragma unroll
        for (int dx = 0; dx < 2; ++dx) {
            int ih = (oh << 1) + dy;
            int iw = (ow << 1) + dx;
            float sum = 0.0f;

            for (int ci = 0; ci < Cin; ++ci) {
                int xidx = ((n * Cin + ci) * H + ih) * W + iw;
                float v = x[xidx];
                float invstd = rsqrtf(running_var[ci] + eps);
                v = (v - running_mean[ci]) * invstd * bn_w[ci] + bn_b[ci];
                v = v > 0.0f ? v : 0.0f;
                sum += v * conv_w[co * Cin + ci];
            }
            acc_pool += sum;
        }
    }

    out[idx] = acc_pool * 0.25f;
}

torch::Tensor fused_transition_eval_cuda(
    torch::Tensor x,
    torch::Tensor bn_w,
    torch::Tensor bn_b,
    torch::Tensor running_mean,
    torch::Tensor running_var,
    torch::Tensor conv_w,
    double eps
) {
    int N = x.size(0);
    int Cin = x.size(1);
    int H = x.size(2);
    int W = x.size(3);
    int Cout = conv_w.size(0);
    auto out = torch::empty({N, Cout, H / 2, W / 2}, x.options());

    int total = N * Cout * (H / 2) * (W / 2);
    int threads = 256;
    int blocks = (total + threads - 1) / threads;

    fused_transition_eval_kernel<<<blocks, threads>>>(
        x.data_ptr<float>(),
        bn_w.data_ptr<float>(),
        bn_b.data_ptr<float>(),
        running_mean.data_ptr<float>(),
        running_var.data_ptr<float>(),
        conv_w.data_ptr<float>(),
        out.data_ptr<float>(),
        N, Cin, H, W, Cout,
        (float)eps
    );

    return out;
}
"""

cpp_sources = r"""
torch::Tensor fused_transition_eval_cuda(
    torch::Tensor x,
    torch::Tensor bn_w,
    torch::Tensor bn_b,
    torch::Tensor running_mean,
    torch::Tensor running_var,
    torch::Tensor conv_w,
    double eps
);
"""

fused_transition_ext = load_inline(
    name="fused_transition_bn_relu_conv_pool_ext",
    cpp_sources=cpp_sources,
    cuda_sources=cuda_sources,
    functions=["fused_transition_eval_cuda"],
    verbose=False,
)


class ModelNew(nn.Module):
    def __init__(self, num_input_features: int, num_output_features: int):
        super(ModelNew, self).__init__()
        self.transition = nn.Sequential(
            nn.BatchNorm2d(num_input_features),
            nn.ReLU(inplace=True),
            nn.Conv2d(num_input_features, num_output_features, kernel_size=1, bias=False),
            nn.AvgPool2d(kernel_size=2, stride=2),
        )

    def forward(self, x):
        bn = self.transition[0]
        conv = self.transition[2]
        return fused_transition_ext.fused_transition_eval_cuda(
            x,
            bn.weight,
            bn.bias,
            bn.running_mean,
            bn.running_var,
            conv.weight,
            bn.eps,
        )
