import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

conv_transpose_tanh_cpp = """
torch::Tensor conv_transpose2d_bias_tanh_cuda(torch::Tensor x, torch::Tensor weight, torch::Tensor conv_bias, torch::Tensor sub_bias);
"""

conv_transpose_tanh_cuda = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math.h>

__global__ void conv_transpose2d_bias_tanh_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ conv_bias,
    const float* __restrict__ sub_bias,
    float* __restrict__ out,
    long total,
    int N,
    int C_in,
    int H,
    int W,
    int C_out,
    int H_out,
    int W_out
) {
    long idx = (long)blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= total) return;

    int ow = idx % W_out;
    long t = idx / W_out;
    int oh = t % H_out;
    t /= H_out;
    int oc = t % C_out;
    int n = t / C_out;

    float acc = conv_bias[oc] - sub_bias[oc];

    #pragma unroll
    for (int kh = 0; kh < 4; ++kh) {
        int ih_num = oh + 1 - kh;
        if ((ih_num & 1) != 0) continue;
        int ih = ih_num >> 1;
        if (ih < 0 || ih >= H) continue;

        #pragma unroll
        for (int kw = 0; kw < 4; ++kw) {
            int iw_num = ow + 1 - kw;
            if ((iw_num & 1) != 0) continue;
            int iw = iw_num >> 1;
            if (iw < 0 || iw >= W) continue;

            long x_base = ((long)n * C_in * H * W) + ((long)ih * W + iw);
            long w_base = ((long)oc * 4 * 4) + ((long)kh * 4 + kw);

            for (int ic = 0; ic < C_in; ++ic) {
                float xv = x[x_base + (long)ic * H * W];
                float wv = w[((long)ic * C_out * 4 * 4) + w_base];
                acc += xv * wv;
            }
        }
    }

    out[idx] = tanhf(acc);
}

torch::Tensor conv_transpose2d_bias_tanh_cuda(
    torch::Tensor x,
    torch::Tensor weight,
    torch::Tensor conv_bias,
    torch::Tensor sub_bias
) {
    int N = x.size(0);
    int C_in = x.size(1);
    int H = x.size(2);
    int W = x.size(3);
    int C_out = weight.size(1);
    int H_out = H * 2;
    int W_out = W * 2;

    auto out = torch::empty({N, C_out, H_out, W_out}, x.options());
    long total = (long)N * C_out * H_out * W_out;

    const int threads = 256;
    const int blocks = (int)((total + threads - 1) / threads);

    conv_transpose2d_bias_tanh_kernel<<<blocks, threads>>>(
        x.data_ptr<float>(),
        weight.data_ptr<float>(),
        conv_bias.data_ptr<float>(),
        sub_bias.data_ptr<float>(),
        out.data_ptr<float>(),
        total,
        N,
        C_in,
        H,
        W,
        C_out,
        H_out,
        W_out
    );

    return out;
}
"""

conv_transpose_tanh_ext = load_inline(
    name="conv_transpose2d_bias_tanh_ext",
    cpp_sources=conv_transpose_tanh_cpp,
    cuda_sources=conv_transpose_tanh_cuda,
    functions=["conv_transpose2d_bias_tanh_cuda"],
    verbose=False,
)


class ModelNew(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, bias_shape, stride=2, padding=1, output_padding=1):
        super(ModelNew, self).__init__()
        self.weight = nn.Parameter(torch.empty(in_channels, out_channels, kernel_size, kernel_size))
        self.conv_bias = nn.Parameter(torch.empty(out_channels))
        self.bias = nn.Parameter(torch.randn(bias_shape))
        nn.init.kaiming_uniform_(self.weight, a=5 ** 0.5)
        fan_in = in_channels * kernel_size * kernel_size
        bound = fan_in ** -0.5
        nn.init.uniform_(self.conv_bias, -bound, bound)
        self.op = conv_transpose_tanh_ext

    def forward(self, x):
        return self.op.conv_transpose2d_bias_tanh_cuda(
            x.contiguous(),
            self.weight.contiguous(),
            self.conv_bias.contiguous(),
            self.bias.contiguous(),
        )
