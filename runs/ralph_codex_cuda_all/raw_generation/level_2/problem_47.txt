import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_sources = r"""
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math.h>

__global__ void mish_tanh_kernel(const float* __restrict__ x, float* __restrict__ y, long n) {
    long idx = (long)blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        float v = x[idx];
        float sp;
        if (v > 20.0f) {
            sp = v;
        } else if (v < -20.0f) {
            sp = expf(v);
        } else {
            sp = log1pf(expf(v));
        }
        float mish = v * tanhf(sp);
        y[idx] = tanhf(mish);
    }
}

torch::Tensor mish_tanh_cuda(torch::Tensor x) {
    auto y = torch::empty_like(x);
    long n = x.numel();
    const int threads = 256;
    const int blocks = (n + threads - 1) / threads;
    mish_tanh_kernel<<<blocks, threads>>>(x.data_ptr<float>(), y.data_ptr<float>(), n);
    return y;
}
"""

cpp_sources = r"""
torch::Tensor mish_tanh_cuda(torch::Tensor x);
"""

mish_tanh_ext = load_inline(
    name="mish_tanh_ext",
    cpp_sources=cpp_sources,
    cuda_sources=cuda_sources,
    functions=["mish_tanh_cuda"],
    verbose=False,
    extra_cflags=["-O3"],
    extra_cuda_cflags=["-O3", "--use_fast_math"],
)


class ModelNew(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, stride=1, padding=0):
        super(ModelNew, self).__init__()
        self.conv = nn.Conv3d(
            in_channels,
            out_channels,
            kernel_size,
            stride=stride,
            padding=padding,
        )
        self.mish_tanh_ext = mish_tanh_ext

    def forward(self, x):
        x = self.conv(x)
        return self.mish_tanh_ext.mish_tanh_cuda(x)
