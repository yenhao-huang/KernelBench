import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cpp_sources = """
torch::Tensor fused_conv_tanh_avgpool_cuda(torch::Tensor x, torch::Tensor w, torch::Tensor b, double sub1, double sub2, int pool);
"""

cuda_sources = """
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void fused_conv_tanh_avgpool_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ b,
    float* __restrict__ y,
    int N, int Cin, int H, int W, int Cout, int K, int pool,
    int Hc, int Wc, int Ho, int Wo,
    float sub1, float sub2
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * Cout * Ho * Wo;
    if (idx >= total) return;

    int ow = idx % Wo;
    int t = idx / Wo;
    int oh = t % Ho;
    t /= Ho;
    int co = t % Cout;
    int n = t / Cout;

    float acc_pool = 0.0f;
    int base_ih0 = oh * pool;
    int base_iw0 = ow * pool;

    for (int ph = 0; ph < pool; ++ph) {
        int hc = base_ih0 + ph;
        for (int pw = 0; pw < pool; ++pw) {
            int wc = base_iw0 + pw;
            float sum = b[co];

            for (int ci = 0; ci < Cin; ++ci) {
                int x_base = ((n * Cin + ci) * H + hc) * W + wc;
                int w_base = ((co * Cin + ci) * K) * K;
                #pragma unroll
                for (int kh = 0; kh < 3; ++kh) {
                    #pragma unroll
                    for (int kw = 0; kw < 3; ++kw) {
                        sum += __ldg(x + x_base + kh * W + kw) * __ldg(w + w_base + kh * K + kw);
                    }
                }
            }

            acc_pool += tanhf(sum - sub1) - sub2;
        }
    }

    y[idx] = acc_pool / float(pool * pool);
}

torch::Tensor fused_conv_tanh_avgpool_cuda(torch::Tensor x, torch::Tensor w, torch::Tensor b, double sub1, double sub2, int pool) {
    int N = x.size(0);
    int Cin = x.size(1);
    int H = x.size(2);
    int W = x.size(3);
    int Cout = w.size(0);
    int K = w.size(2);

    int Hc = H - K + 1;
    int Wc = W - K + 1;
    int Ho = Hc / pool;
    int Wo = Wc / pool;

    auto y = torch::empty({N, Cout, Ho, Wo}, x.options());

    int total = N * Cout * Ho * Wo;
    int threads = 128;
    int blocks = (total + threads - 1) / threads;

    fused_conv_tanh_avgpool_kernel<<<blocks, threads>>>(
        x.data_ptr<float>(),
        w.data_ptr<float>(),
        b.data_ptr<float>(),
        y.data_ptr<float>(),
        N, Cin, H, W, Cout, K, pool, Hc, Wc, Ho, Wo,
        static_cast<float>(sub1),
        static_cast<float>(sub2)
    );

    return y;
}
"""

_fused_ops = load_inline(
    name="fused_conv_tanh_avgpool_ext",
    cpp_sources=cpp_sources,
    cuda_sources=cuda_sources,
    functions=["fused_conv_tanh_avgpool_cuda"],
    verbose=False,
)


class ModelNew(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool):
        super().__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.subtract1_value = subtract1_value
        self.subtract2_value = subtract2_value
        self.kernel_size_pool = kernel_size_pool
        self.ops = _fused_ops

    def forward(self, x):
        return self.ops.fused_conv_tanh_avgpool_cuda(
            x.contiguous(),
            self.conv.weight.contiguous(),
            self.conv.bias.contiguous(),
            self.subtract1_value,
            self.subtract2_value,
            self.kernel_size_pool,
        )
