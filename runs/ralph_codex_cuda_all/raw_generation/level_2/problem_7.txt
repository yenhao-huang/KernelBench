import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_sources = r"""
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math.h>

__global__ void fused_acts_bias_kernel(
    const float* __restrict__ x,
    const float* __restrict__ bias,
    float* __restrict__ out,
    long total,
    int C,
    int D,
    int H,
    int W
) {
    long idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= total) return;

    float v = x[idx];
    v = v > 0.0f ? v : 0.0f;

    const float inv_sqrt2 = 0.7071067811865475f;
    float gelu = 0.5f * v * (1.0f + erff(v * inv_sqrt2));
    float sig = 1.0f / (1.0f + expf(-gelu));

    long spatial = (long)D * H * W;
    int c = (idx / spatial) % C;
    out[idx] = sig + bias[c];
}

torch::Tensor fused_acts_bias_cuda(torch::Tensor x, torch::Tensor bias) {
    x = x.contiguous();
    auto out = torch::empty_like(x);

    int C = x.size(1);
    int D = x.size(2);
    int H = x.size(3);
    int W = x.size(4);
    long total = x.numel();

    const int threads = 256;
    const int blocks = (total + threads - 1) / threads;

    fused_acts_bias_kernel<<<blocks, threads>>>(
        x.data_ptr<float>(),
        bias.data_ptr<float>(),
        out.data_ptr<float>(),
        total,
        C,
        D,
        H,
        W
    );

    return out;
}
"""

cpp_sources = r"""
torch::Tensor fused_acts_bias_cuda(torch::Tensor x, torch::Tensor bias);
"""

fused_acts_bias = load_inline(
    name="fused_acts_bias_ext",
    cpp_sources=cpp_sources,
    cuda_sources=cuda_sources,
    functions=["fused_acts_bias_cuda"],
    verbose=False,
)


class ModelNew(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, bias_shape):
        super(ModelNew, self).__init__()
        self.conv = nn.Conv3d(in_channels, out_channels, kernel_size)
        self.bias = nn.Parameter(torch.randn(bias_shape))
        self.fused_acts_bias = fused_acts_bias

    def forward(self, x):
        x = self.conv(x)
        return self.fused_acts_bias.fused_acts_bias_cuda(x, self.bias)
