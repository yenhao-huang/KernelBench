import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_sources = r"""
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <vector>
#include <float.h>

__device__ __forceinline__ float lrelu(float x) {
    return x >= 0.0f ? x : 0.2f * x;
}

__global__ void fused_lrelu_mul_lrelu_pool3d_kernel(
    const float* __restrict__ x,
    const float* __restrict__ multiplier,
    float* __restrict__ out,
    int total,
    int C,
    int D2,
    int H2,
    int W2,
    int D,
    int H,
    int W
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= total) return;

    int ow = idx % W2;
    int t = idx / W2;
    int oh = t % H2;
    t /= H2;
    int od = t % D2;
    t /= D2;
    int c = t % C;
    int n = t / C;

    float m = multiplier[c];
    float best = -FLT_MAX;

    int base_n = n * C * D * H * W;
    int base_c = c * D * H * W;
    int id0 = od * 2;
    int ih0 = oh * 2;
    int iw0 = ow * 2;

    #pragma unroll
    for (int dz = 0; dz < 2; ++dz) {
        #pragma unroll
        for (int dy = 0; dy < 2; ++dy) {
            #pragma unroll
            for (int dx = 0; dx < 2; ++dx) {
                int in_idx = base_n + base_c + (id0 + dz) * H * W + (ih0 + dy) * W + (iw0 + dx);
                float v = x[in_idx];
                v = lrelu(v);
                v = v * m;
                v = lrelu(v);
                best = fmaxf(best, v);
            }
        }
    }

    out[idx] = best;
}

torch::Tensor fused_lrelu_mul_lrelu_pool3d_cuda(torch::Tensor x, torch::Tensor multiplier) {
    int N = x.size(0);
    int C = x.size(1);
    int D = x.size(2);
    int H = x.size(3);
    int W = x.size(4);

    int D2 = D / 2;
    int H2 = H / 2;
    int W2 = W / 2;

    auto out = torch::empty({N, C, D2, H2, W2}, x.options());
    int total = N * C * D2 * H2 * W2;

    const int threads = 256;
    const int blocks = (total + threads - 1) / threads;

    fused_lrelu_mul_lrelu_pool3d_kernel<<<blocks, threads>>>(
        x.data_ptr<float>(),
        multiplier.data_ptr<float>(),
        out.data_ptr<float>(),
        total,
        C,
        D2,
        H2,
        W2,
        D,
        H,
        W
    );

    return out;
}
"""

cpp_sources = "torch::Tensor fused_lrelu_mul_lrelu_pool3d_cuda(torch::Tensor x, torch::Tensor multiplier);"

fused_post = load_inline(
    name="fused_lrelu_mul_lrelu_pool3d_ext",
    cpp_sources=cpp_sources,
    cuda_sources=cuda_sources,
    functions=["fused_lrelu_mul_lrelu_pool3d_cuda"],
    verbose=False,
)


class ModelNew(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, multiplier_shape):
        super(ModelNew, self).__init__()
        self.conv_transpose = nn.ConvTranspose3d(
            in_channels,
            out_channels,
            kernel_size,
            stride=stride,
            padding=padding,
            output_padding=output_padding,
        )
        self.multiplier = nn.Parameter(torch.randn(multiplier_shape))

    def forward(self, x):
        x = self.conv_transpose(x)
        return fused_post.fused_lrelu_mul_lrelu_pool3d_cuda(x, self.multiplier)
