import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_sources = r"""
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cfloat>
#include <math.h>

__global__ void spatial_sum_kernel(const float* __restrict__ x,
                                   float* __restrict__ sums,
                                   int B, int C, int H, int W) {
    int bc = blockIdx.x;
    int b = bc / C;
    int c = bc - b * C;
    int HW = H * W;
    const float* ptr = x + ((b * C + c) * HW);

    float acc = 0.0f;
    for (int i = threadIdx.x; i < HW; i += blockDim.x) {
        acc += ptr[i];
    }

    __shared__ float sh[256];
    sh[threadIdx.x] = acc;
    __syncthreads();

    for (int stride = 128; stride > 0; stride >>= 1) {
        if (threadIdx.x < stride) {
            sh[threadIdx.x] += sh[threadIdx.x + stride];
        }
        __syncthreads();
    }

    if (threadIdx.x == 0) {
        sums[bc] = sh[0];
    }
}

__global__ void fused_finish_kernel(const float* __restrict__ sums,
                                    const float* __restrict__ weight,
                                    const float* __restrict__ conv_bias,
                                    const float* __restrict__ add_bias,
                                    float* __restrict__ out,
                                    int B, int IC, int OC, int H, int W, int KH, int KW) {
    int b = blockIdx.x;
    int OH = H + KH - 1;
    int OW = W + KW - 1;
    float inv_area = 1.0f / ((float)OH * (float)OW);

    float maxv = -FLT_MAX;
    for (int oc = threadIdx.x; oc < OC; oc += blockDim.x) {
        float v = conv_bias[oc];
        for (int ic = 0; ic < IC; ++ic) {
            float wsum = 0.0f;
            int base = ((ic * OC + oc) * KH) * KW;
            for (int k = 0; k < KH * KW; ++k) {
                wsum += weight[base + k];
            }
            v += sums[b * IC + ic] * wsum * inv_area;
        }
        v += add_bias[oc];
        maxv = fmaxf(maxv, v);
    }

    __shared__ float sh[256];
    sh[threadIdx.x] = maxv;
    __syncthreads();

    for (int stride = 128; stride > 0; stride >>= 1) {
        if (threadIdx.x < stride) {
            sh[threadIdx.x] = fmaxf(sh[threadIdx.x], sh[threadIdx.x + stride]);
        }
        __syncthreads();
    }

    float m = sh[0];
    float sumexp = 0.0f;
    for (int oc = threadIdx.x; oc < OC; oc += blockDim.x) {
        float v = conv_bias[oc];
        for (int ic = 0; ic < IC; ++ic) {
            float wsum = 0.0f;
            int base = ((ic * OC + oc) * KH) * KW;
            for (int k = 0; k < KH * KW; ++k) {
                wsum += weight[base + k];
            }
            v += sums[b * IC + ic] * wsum * inv_area;
        }
        v += add_bias[oc];
        sumexp += expf(v - m);
    }

    sh[threadIdx.x] = sumexp;
    __syncthreads();

    for (int stride = 128; stride > 0; stride >>= 1) {
        if (threadIdx.x < stride) {
            sh[threadIdx.x] += sh[threadIdx.x + stride];
        }
        __syncthreads();
    }

    if (threadIdx.x == 0) {
        out[b] = (logf(sh[0]) + m) * 10.0f;
    }
}

torch::Tensor fused_convtranspose_pool_lse_cuda(torch::Tensor x,
                                                torch::Tensor weight,
                                                torch::Tensor conv_bias,
                                                torch::Tensor add_bias) {
    int B = x.size(0);
    int IC = x.size(1);
    int H = x.size(2);
    int W = x.size(3);
    int OC = weight.size(1);
    int KH = weight.size(2);
    int KW = weight.size(3);

    auto sums = torch::empty({B, IC}, x.options());
    auto out = torch::empty({B, 1}, x.options());

    spatial_sum_kernel<<<B * IC, 256>>>(x.data_ptr<float>(), sums.data_ptr<float>(), B, IC, H, W);
    fused_finish_kernel<<<B, 256>>>(sums.data_ptr<float>(),
                                    weight.data_ptr<float>(),
                                    conv_bias.data_ptr<float>(),
                                    add_bias.data_ptr<float>(),
                                    out.data_ptr<float>(),
                                    B, IC, OC, H, W, KH, KW);
    return out;
}
"""

cpp_sources = "torch::Tensor fused_convtranspose_pool_lse_cuda(torch::Tensor x, torch::Tensor weight, torch::Tensor conv_bias, torch::Tensor add_bias);"

_fused_mod = load_inline(
    name="fused_convtranspose_pool_lse_ext",
    cpp_sources=cpp_sources,
    cuda_sources=cuda_sources,
    functions=["fused_convtranspose_pool_lse_cuda"],
    verbose=False,
    extra_cuda_cflags=["-O3", "--use_fast_math"],
)


class ModelNew(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, bias_shape):
        super(ModelNew, self).__init__()
        self.conv_transpose = nn.ConvTranspose2d(in_channels, out_channels, kernel_size)
        self.bias = nn.Parameter(torch.randn(bias_shape))

    def forward(self, x):
        return _fused_mod.fused_convtranspose_pool_lse_cuda(
            x,
            self.conv_transpose.weight,
            self.conv_transpose.bias,
            self.bias,
        )
