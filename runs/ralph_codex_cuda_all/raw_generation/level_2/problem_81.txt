import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cpp_sources = """
torch::Tensor fused_linear_act_cuda(torch::Tensor x, torch::Tensor weight, torch::Tensor bias);
torch::Tensor fused_linear_act_nobias_cuda(torch::Tensor x, torch::Tensor weight);
"""

cuda_sources = """
#include <torch/extension.h>
#include <cuda_runtime.h>

#define TILE 16

__device__ __forceinline__ float fused_activation(float v) {
    float s = 1.0f / (1.0f + __expf(-v));
    v = 0.5f * v * s;
    v = fminf(1.0f, fmaxf(-1.0f, v));
    v = tanhf(v);
    return fminf(1.0f, fmaxf(-1.0f, v));
}

__global__ void fused_linear_act_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ b,
    float* __restrict__ out,
    int M,
    int K,
    int N,
    int has_bias
) {
    __shared__ float xs[TILE][TILE];
    __shared__ float ws[TILE][TILE + 1];

    int row = blockIdx.y * TILE + threadIdx.y;
    int col = blockIdx.x * TILE + threadIdx.x;

    float acc = 0.0f;

    for (int t = 0; t < K; t += TILE) {
        int x_col = t + threadIdx.x;
        int w_col = t + threadIdx.y;

        xs[threadIdx.y][threadIdx.x] = (row < M && x_col < K) ? x[row * K + x_col] : 0.0f;
        ws[threadIdx.y][threadIdx.x] = (col < N && w_col < K) ? w[col * K + w_col] : 0.0f;

        __syncthreads();

        #pragma unroll
        for (int i = 0; i < TILE; ++i) {
            acc += xs[threadIdx.y][i] * ws[i][threadIdx.x];
        }

        __syncthreads();
    }

    if (row < M && col < N) {
        if (has_bias) {
            acc += b[col];
        }
        out[row * N + col] = fused_activation(acc);
    }
}

torch::Tensor fused_linear_act_cuda(torch::Tensor x, torch::Tensor weight, torch::Tensor bias) {
    int M = (int)x.size(0);
    int K = (int)x.size(1);
    int N = (int)weight.size(0);

    auto out = torch::empty({M, N}, x.options());

    dim3 block(TILE, TILE);
    dim3 grid((N + TILE - 1) / TILE, (M + TILE - 1) / TILE);

    fused_linear_act_kernel<<<grid, block>>>(
        x.data_ptr<float>(),
        weight.data_ptr<float>(),
        bias.data_ptr<float>(),
        out.data_ptr<float>(),
        M, K, N, 1
    );

    return out;
}

torch::Tensor fused_linear_act_nobias_cuda(torch::Tensor x, torch::Tensor weight) {
    int M = (int)x.size(0);
    int K = (int)x.size(1);
    int N = (int)weight.size(0);

    auto out = torch::empty({M, N}, x.options());

    dim3 block(TILE, TILE);
    dim3 grid((N + TILE - 1) / TILE, (M + TILE - 1) / TILE);

    fused_linear_act_kernel<<<grid, block>>>(
        x.data_ptr<float>(),
        weight.data_ptr<float>(),
        nullptr,
        out.data_ptr<float>(),
        M, K, N, 0
    );

    return out;
}
"""

_fused_linear_act = load_inline(
    name="fused_linear_swish_div_clamp_tanh_clamp",
    cpp_sources=cpp_sources,
    cuda_sources=cuda_sources,
    functions=["fused_linear_act_cuda", "fused_linear_act_nobias_cuda"],
    verbose=False,
    extra_cuda_cflags=["-O3", "--use_fast_math"],
)


class ModelNew(nn.Module):
    def __init__(self, in_features, out_features, bias=True):
        super(ModelNew, self).__init__()
        self.gemm = nn.Linear(in_features, out_features, bias=bias)
        self.fused = _fused_linear_act

    def forward(self, x):
        if self.gemm.bias is None:
            return self.fused.fused_linear_act_nobias_cuda(x, self.gemm.weight)
        return self.fused.fused_linear_act_cuda(x, self.gemm.weight, self.gemm.bias)
