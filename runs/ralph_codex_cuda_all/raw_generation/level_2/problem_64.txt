import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cpp_sources = """
torch::Tensor fused_linear_lse_acts_cuda(torch::Tensor x, torch::Tensor weight, torch::Tensor bias);
"""

cuda_sources = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math_constants.h>

#define TILE 16

__global__ void matmul_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ b,
    float* __restrict__ y,
    int B, int K, int N, bool has_bias
) {
    __shared__ float xs[TILE][TILE];
    __shared__ float ws[TILE][TILE + 1];

    int row = blockIdx.y * TILE + threadIdx.y;
    int col = blockIdx.x * TILE + threadIdx.x;

    float acc = 0.0f;

    for (int t = 0; t < K; t += TILE) {
        int x_col = t + threadIdx.x;
        int w_col = t + threadIdx.y;

        xs[threadIdx.y][threadIdx.x] = (row < B && x_col < K) ? x[row * K + x_col] : 0.0f;
        ws[threadIdx.y][threadIdx.x] = (col < N && w_col < K) ? w[col * K + w_col] : 0.0f;

        __syncthreads();

        #pragma unroll
        for (int i = 0; i < TILE; ++i) {
            acc += xs[threadIdx.y][i] * ws[i][threadIdx.x];
        }

        __syncthreads();
    }

    if (row < B && col < N) {
        y[row * N + col] = acc + (has_bias ? b[col] : 0.0f);
    }
}

__device__ __forceinline__ float gelu_exact(float x) {
    const float inv_sqrt2 = 0.70710678118654752440f;
    return 0.5f * x * (1.0f + erff(x * inv_sqrt2));
}

__global__ void lse_acts_kernel(
    const float* __restrict__ y,
    float* __restrict__ out,
    int B, int N
) {
    int row = blockIdx.x;
    int tid = threadIdx.x;

    float local_max = -CUDART_INF_F;
    for (int col = tid; col < N; col += blockDim.x) {
        local_max = fmaxf(local_max, y[row * N + col]);
    }

    __shared__ float smem[256];
    smem[tid] = local_max;
    __syncthreads();

    for (int stride = blockDim.x >> 1; stride > 0; stride >>= 1) {
        if (tid < stride) {
            smem[tid] = fmaxf(smem[tid], smem[tid + stride]);
        }
        __syncthreads();
    }

    float maxv = smem[0];
    float local_sum = 0.0f;

    for (int col = tid; col < N; col += blockDim.x) {
        local_sum += expf(y[row * N + col] - maxv);
    }

    smem[tid] = local_sum;
    __syncthreads();

    for (int stride = blockDim.x >> 1; stride > 0; stride >>= 1) {
        if (tid < stride) {
            smem[tid] += smem[tid + stride];
        }
        __syncthreads();
    }

    if (tid == 0) {
        float v = logf(smem[0]) + maxv;
        v = v > 0.0f ? v : 0.01f * v;
        v = v > 0.0f ? v : 0.01f * v;
        v = gelu_exact(v);
        v = gelu_exact(v);
        out[row] = v;
    }
}

torch::Tensor fused_linear_lse_acts_cuda(torch::Tensor x, torch::Tensor weight, torch::Tensor bias) {
    int B = (int)x.size(0);
    int K = (int)x.size(1);
    int N = (int)weight.size(0);
    bool has_bias = bias.numel() != 0;

    auto y = torch::empty({B, N}, x.options());
    auto out = torch::empty({B, 1}, x.options());

    dim3 block(TILE, TILE);
    dim3 grid((N + TILE - 1) / TILE, (B + TILE - 1) / TILE);

    matmul_kernel<<<grid, block>>>(
        x.data_ptr<float>(),
        weight.data_ptr<float>(),
        has_bias ? bias.data_ptr<float>() : nullptr,
        y.data_ptr<float>(),
        B, K, N, has_bias
    );

    lse_acts_kernel<<<B, 256>>>(y.data_ptr<float>(), out.data_ptr<float>(), B, N);

    return out;
}
"""

fused_linear_lse_acts = load_inline(
    name="fused_linear_lse_acts_ext",
    cpp_sources=cpp_sources,
    cuda_sources=cuda_sources,
    functions=["fused_linear_lse_acts_cuda"],
    verbose=False,
    extra_cuda_cflags=["-O3", "--use_fast_math"],
)


class ModelNew(nn.Module):
    def __init__(self, in_features, out_features, bias=True):
        super().__init__()
        self.linear = nn.Linear(in_features, out_features, bias=bias)

    def forward(self, x):
        bias = self.linear.bias
        if bias is None:
            bias = x.new_empty((0,))
        return fused_linear_lse_acts.fused_linear_lse_acts_cuda(x, self.linear.weight, bias)
