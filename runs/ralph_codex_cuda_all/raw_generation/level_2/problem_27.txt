import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_sources = r"""
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math.h>

__device__ __forceinline__ float hswish_f32(float x) {
    float r = fminf(fmaxf(x + 3.0f, 0.0f), 6.0f);
    return x * r * 0.16666666666666666f;
}

__global__ void conv_hswish_channel_stats_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ bias,
    float* __restrict__ channel_sum,
    float* __restrict__ channel_sumsq,
    int B, int Cin, int Cout,
    int Din, int Hin, int Win,
    int Kd, int Kh, int Kw,
    int Dout, int Hout, int Wout,
    int has_bias
) {
    int bc = blockIdx.x;
    int b = bc / Cout;
    int co = bc - b * Cout;
    int S = Dout * Hout * Wout;

    float local_sum = 0.0f;
    float local_sumsq = 0.0f;

    for (int idx = threadIdx.x; idx < S; idx += blockDim.x) {
        int t = idx;
        int ow = t % Wout;
        t /= Wout;
        int oh = t % Hout;
        int od = t / Hout;

        float acc = has_bias ? bias[co] : 0.0f;

        for (int ci = 0; ci < Cin; ++ci) {
            for (int kd = 0; kd < Kd; ++kd) {
                int id = od + kd;
                for (int kh = 0; kh < Kh; ++kh) {
                    int ih = oh + kh;
                    const float* x_base = x + (((b * Cin + ci) * Din + id) * Hin + ih) * Win + ow;
                    const float* w_base = w + (((co * Cin + ci) * Kd + kd) * Kh + kh) * Kw;
                    for (int kw = 0; kw < Kw; ++kw) {
                        acc += x_base[kw] * w_base[kw];
                    }
                }
            }
        }

        float y = hswish_f32(acc);
        local_sum += y;
        local_sumsq += y * y;
    }

    __shared__ float smem_sum[256];
    __shared__ float smem_sumsq[256];

    smem_sum[threadIdx.x] = local_sum;
    smem_sumsq[threadIdx.x] = local_sumsq;
    __syncthreads();

    for (int stride = blockDim.x >> 1; stride > 0; stride >>= 1) {
        if (threadIdx.x < stride) {
            smem_sum[threadIdx.x] += smem_sum[threadIdx.x + stride];
            smem_sumsq[threadIdx.x] += smem_sumsq[threadIdx.x + stride];
        }
        __syncthreads();
    }

    if (threadIdx.x == 0) {
        channel_sum[bc] = smem_sum[0];
        channel_sumsq[bc] = smem_sumsq[0];
    }
}

__global__ void groupnorm_spatial_mean_kernel(
    const float* __restrict__ channel_sum,
    const float* __restrict__ channel_sumsq,
    const float* __restrict__ gamma,
    const float* __restrict__ beta,
    float* __restrict__ out,
    int B, int Cout, int groups, int S, float eps
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = B * Cout;
    if (idx >= total) return;

    int b = idx / Cout;
    int c = idx - b * Cout;
    int group_size = Cout / groups;
    int g0 = (c / group_size) * group_size;

    float gsum = 0.0f;
    float gsumsq = 0.0f;
    for (int i = 0; i < group_size; ++i) {
        int off = b * Cout + g0 + i;
        gsum += channel_sum[off];
        gsumsq += channel_sumsq[off];
    }

    float n = (float)(group_size * S);
    float mean = gsum / n;
    float var = gsumsq / n - mean * mean;
    var = fmaxf(var, 0.0f);

    float ch_mean = channel_sum[idx] / (float)S;
    out[idx] = (ch_mean - mean) * rsqrtf(var + eps) * gamma[c] + beta[c];
}

torch::Tensor fused_conv_hswish_groupnorm_mean_cuda(
    torch::Tensor x,
    torch::Tensor weight,
    torch::Tensor bias,
    torch::Tensor gamma,
    torch::Tensor beta,
    int64_t groups,
    double eps
) {
    int B = (int)x.size(0);
    int Cin = (int)x.size(1);
    int Din = (int)x.size(2);
    int Hin = (int)x.size(3);
    int Win = (int)x.size(4);

    int Cout = (int)weight.size(0);
    int Kd = (int)weight.size(2);
    int Kh = (int)weight.size(3);
    int Kw = (int)weight.size(4);

    int Dout = Din - Kd + 1;
    int Hout = Hin - Kh + 1;
    int Wout = Win - Kw + 1;
    int S = Dout * Hout * Wout;

    auto opts = x.options();
    auto channel_sum = torch::empty({B, Cout}, opts);
    auto channel_sumsq = torch::empty({B, Cout}, opts);
    auto out = torch::empty({B, Cout}, opts);

    const int threads = 256;
    conv_hswish_channel_stats_kernel<<<B * Cout, threads>>>(
        x.data_ptr<float>(),
        weight.data_ptr<float>(),
        bias.defined() ? bias.data_ptr<float>() : nullptr,
        channel_sum.data_ptr<float>(),
        channel_sumsq.data_ptr<float>(),
        B, Cin, Cout, Din, Hin, Win, Kd, Kh, Kw, Dout, Hout, Wout,
        bias.defined() ? 1 : 0
    );

    int total = B * Cout;
    groupnorm_spatial_mean_kernel<<<(total + threads - 1) / threads, threads>>>(
        channel_sum.data_ptr<float>(),
        channel_sumsq.data_ptr<float>(),
        gamma.data_ptr<float>(),
        beta.data_ptr<float>(),
        out.data_ptr<float>(),
        B, Cout, (int)groups, S, (float)eps
    );

    return out;
}
"""

cpp_sources = r"""
torch::Tensor fused_conv_hswish_groupnorm_mean_cuda(
    torch::Tensor x,
    torch::Tensor weight,
    torch::Tensor bias,
    torch::Tensor gamma,
    torch::Tensor beta,
    int64_t groups,
    double eps
);
"""

fused_ops = load_inline(
    name="fused_conv_hswish_groupnorm_mean_ext",
    cpp_sources=cpp_sources,
    cuda_sources=cuda_sources,
    functions=["fused_conv_hswish_groupnorm_mean_cuda"],
    verbose=False,
    extra_cflags=["-O3"],
    extra_cuda_cflags=["-O3", "--use_fast_math"],
)


class ModelNew(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, num_groups=4, bias=True):
        super().__init__()
        self.conv = nn.Conv3d(in_channels, out_channels, kernel_size, bias=bias)
        self.group_norm = nn.GroupNorm(num_groups, out_channels)

    def forward(self, x):
        bias = self.conv.bias
        if bias is None:
            bias = torch.Tensor()
        return fused_ops.fused_conv_hswish_groupnorm_mean_cuda(
            x,
            self.conv.weight,
            bias,
            self.group_norm.weight,
            self.group_norm.bias,
            self.group_norm.num_groups,
            self.group_norm.eps,
        )
