import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_sources = r"""
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math.h>

__global__ void conv3x3_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ b,
    float* __restrict__ y,
    int N, int Cin, int H, int W, int Cout, int Ho, int Wo
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * Cout * Ho * Wo;
    if (idx >= total) return;

    int ow = idx % Wo;
    int t = idx / Wo;
    int oh = t % Ho;
    t /= Ho;
    int co = t % Cout;
    int n = t / Cout;

    float acc = b[co];
    for (int ci = 0; ci < Cin; ++ci) {
        const float* xp = x + ((n * Cin + ci) * H + oh) * W + ow;
        const float* wp = w + ((co * Cin + ci) * 3) * 3;
        acc += xp[0] * wp[0] + xp[1] * wp[1] + xp[2] * wp[2];
        acc += xp[W] * wp[3] + xp[W + 1] * wp[4] + xp[W + 2] * wp[5];
        acc += xp[2 * W] * wp[6] + xp[2 * W + 1] * wp[7] + xp[2 * W + 2] * wp[8];
    }
    y[idx] = acc;
}

__global__ void group_stats_kernel(
    const float* __restrict__ y,
    float* __restrict__ mean,
    float* __restrict__ invstd,
    int N, int C, int Ho, int Wo, int G
) {
    int ng = blockIdx.x;
    int n = ng / G;
    int g = ng % G;
    int cpg = C / G;
    int elems = cpg * Ho * Wo;

    float sum = 0.0f;
    float sumsq = 0.0f;

    for (int i = threadIdx.x; i < elems; i += blockDim.x) {
        int s = i % (Ho * Wo);
        int lc = i / (Ho * Wo);
        int c = g * cpg + lc;
        float v = y[((n * C + c) * Ho * Wo) + s];
        sum += v;
        sumsq += v * v;
    }

    __shared__ float sh_sum[256];
    __shared__ float sh_sumsq[256];
    sh_sum[threadIdx.x] = sum;
    sh_sumsq[threadIdx.x] = sumsq;
    __syncthreads();

    for (int stride = blockDim.x >> 1; stride > 0; stride >>= 1) {
        if (threadIdx.x < stride) {
            sh_sum[threadIdx.x] += sh_sum[threadIdx.x + stride];
            sh_sumsq[threadIdx.x] += sh_sumsq[threadIdx.x + stride];
        }
        __syncthreads();
    }

    if (threadIdx.x == 0) {
        float m = sh_sum[0] / elems;
        float var = sh_sumsq[0] / elems - m * m;
        mean[ng] = m;
        invstd[ng] = rsqrtf(var + 1.0e-5f);
    }
}

__global__ void norm_scale_pool_clamp_kernel(
    const float* __restrict__ y,
    const float* __restrict__ mean,
    const float* __restrict__ invstd,
    const float* __restrict__ gn_w,
    const float* __restrict__ gn_b,
    const float* __restrict__ scale,
    float* __restrict__ out,
    int N, int C, int Ho, int Wo, int G, int PoH, int PoW,
    float clamp_min, float clamp_max
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * C * PoH * PoW;
    if (idx >= total) return;

    int pw = idx % PoW;
    int t = idx / PoW;
    int ph = t % PoH;
    t /= PoH;
    int c = t % C;
    int n = t / C;

    int g = c / (C / G);
    float m = mean[n * G + g];
    float r = invstd[n * G + g];
    float gw = gn_w[c];
    float gb = gn_b[c];
    float sc = scale[c];

    float mx = -3.402823466e+38f;
    int h0 = ph * 4;
    int w0 = pw * 4;

    #pragma unroll
    for (int kh = 0; kh < 4; ++kh) {
        #pragma unroll
        for (int kw = 0; kw < 4; ++kw) {
            float v = y[((n * C + c) * Ho + h0 + kh) * Wo + w0 + kw];
            v = ((v - m) * r * gw + gb) * sc;
            mx = fmaxf(mx, v);
        }
    }

    mx = fminf(fmaxf(mx, clamp_min), clamp_max);
    out[idx] = mx;
}

torch::Tensor fused_forward_cuda(
    torch::Tensor x,
    torch::Tensor conv_w,
    torch::Tensor conv_b,
    torch::Tensor gn_w,
    torch::Tensor gn_b,
    torch::Tensor scale,
    int groups,
    float clamp_min,
    float clamp_max
) {
    int N = x.size(0);
    int Cin = x.size(1);
    int H = x.size(2);
    int W = x.size(3);
    int Cout = conv_w.size(0);
    int Ho = H - 2;
    int Wo = W - 2;
    int PoH = Ho / 4;
    int PoW = Wo / 4;

    auto y = torch::empty({N, Cout, Ho, Wo}, x.options());
    auto mean = torch::empty({N, groups}, x.options());
    auto invstd = torch::empty({N, groups}, x.options());
    auto out = torch::empty({N, Cout, PoH, PoW}, x.options());

    int conv_total = N * Cout * Ho * Wo;
    conv3x3_kernel<<<(conv_total + 255) / 256, 256>>>(
        x.data_ptr<float>(), conv_w.data_ptr<float>(), conv_b.data_ptr<float>(),
        y.data_ptr<float>(), N, Cin, H, W, Cout, Ho, Wo
    );

    group_stats_kernel<<<N * groups, 256>>>(
        y.data_ptr<float>(), mean.data_ptr<float>(), invstd.data_ptr<float>(),
        N, Cout, Ho, Wo, groups
    );

    int out_total = N * Cout * PoH * PoW;
    norm_scale_pool_clamp_kernel<<<(out_total + 255) / 256, 256>>>(
        y.data_ptr<float>(), mean.data_ptr<float>(), invstd.data_ptr<float>(),
        gn_w.data_ptr<float>(), gn_b.data_ptr<float>(), scale.data_ptr<float>(),
        out.data_ptr<float>(), N, Cout, Ho, Wo, groups, PoH, PoW, clamp_min, clamp_max
    );

    return out;
}
"""

cpp_sources = "torch::Tensor fused_forward_cuda(torch::Tensor x, torch::Tensor conv_w, torch::Tensor conv_b, torch::Tensor gn_w, torch::Tensor gn_b, torch::Tensor scale, int groups, float clamp_min, float clamp_max);"

fused_mod = load_inline(
    name="kb_conv_gn_scale_pool_clamp_fused",
    cpp_sources=cpp_sources,
    cuda_sources=cuda_sources,
    functions=["fused_forward_cuda"],
    verbose=False,
    extra_cflags=["-O3"],
    extra_cuda_cflags=["-O3", "--use_fast_math"],
)


class ModelNew(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, num_groups, scale_shape, maxpool_kernel_size, clamp_min, clamp_max):
        super(ModelNew, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.group_norm = nn.GroupNorm(num_groups, out_channels)
        self.scale = nn.Parameter(torch.ones(scale_shape))
        self.num_groups = num_groups
        self.clamp_min = float(clamp_min)
        self.clamp_max = float(clamp_max)
        self.fused_mod = fused_mod

    def forward(self, x):
        return self.fused_mod.fused_forward_cuda(
            x,
            self.conv.weight,
            self.conv.bias,
            self.group_norm.weight,
            self.group_norm.bias,
            self.scale,
            self.num_groups,
            self.clamp_min,
            self.clamp_max,
        )
