import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_sources = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda_runtime.h>
#include <vector>

__global__ void conv_transpose3d_clamp_div_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ b,
    float* __restrict__ y,
    long long total,
    int N, int Cin, int Din, int Hin, int Win,
    int Cout, int Dout, int Hout, int Wout,
    int Kd, int Kh, int Kw,
    int stride, int padding,
    float min_value, float divisor
) {
    long long idx = blockIdx.x * (long long)blockDim.x + threadIdx.x;
    if (idx >= total) return;

    int ow = idx % Wout;
    long long t = idx / Wout;
    int oh = t % Hout;
    t /= Hout;
    int od = t % Dout;
    t /= Dout;
    int co = t % Cout;
    int n = t / Cout;

    float acc = b ? b[co] : 0.0f;

    #pragma unroll
    for (int kd = 0; kd < 3; ++kd) {
        int td = od + padding - kd;
        if (td < 0 || (td % stride) != 0) continue;
        int id = td / stride;
        if (id < 0 || id >= Din) continue;

        #pragma unroll
        for (int kh = 0; kh < 3; ++kh) {
            int th = oh + padding - kh;
            if (th < 0 || (th % stride) != 0) continue;
            int ih = th / stride;
            if (ih < 0 || ih >= Hin) continue;

            #pragma unroll
            for (int kw = 0; kw < 3; ++kw) {
                int tw = ow + padding - kw;
                if (tw < 0 || (tw % stride) != 0) continue;
                int iw = tw / stride;
                if (iw < 0 || iw >= Win) continue;

                for (int ci = 0; ci < Cin; ++ci) {
                    long long xoff = (((((long long)n * Cin + ci) * Din + id) * Hin + ih) * Win + iw);
                    long long woff = (((((long long)ci * Cout + co) * Kd + kd) * Kh + kh) * Kw + kw);
                    acc += x[xoff] * w[woff];
                }
            }
        }
    }

    acc = acc < min_value ? min_value : acc;
    y[idx] = acc / divisor;
}

torch::Tensor conv_transpose3d_clamp_div_cuda(
    torch::Tensor x,
    torch::Tensor weight,
    torch::Tensor bias,
    int stride,
    int padding,
    double min_value,
    double divisor
) {
    int N = x.size(0);
    int Cin = x.size(1);
    int Din = x.size(2);
    int Hin = x.size(3);
    int Win = x.size(4);

    int Cout = weight.size(1);
    int Kd = weight.size(2);
    int Kh = weight.size(3);
    int Kw = weight.size(4);

    int Dout = (Din - 1) * stride - 2 * padding + Kd;
    int Hout = (Hin - 1) * stride - 2 * padding + Kh;
    int Wout = (Win - 1) * stride - 2 * padding + Kw;

    auto y = torch::empty({N, Cout, Dout, Hout, Wout}, x.options());
    long long total = y.numel();

    const int threads = 256;
    const int blocks = (int)((total + threads - 1) / threads);

    conv_transpose3d_clamp_div_kernel<<<blocks, threads, 0, at::cuda::getCurrentCUDAStream()>>>(
        x.data_ptr<float>(),
        weight.data_ptr<float>(),
        bias.defined() ? bias.data_ptr<float>() : nullptr,
        y.data_ptr<float>(),
        total,
        N, Cin, Din, Hin, Win,
        Cout, Dout, Hout, Wout,
        Kd, Kh, Kw,
        stride, padding,
        (float)min_value, (float)divisor
    );

    return y;
}
"""

cpp_sources = r"""
torch::Tensor conv_transpose3d_clamp_div_cuda(
    torch::Tensor x,
    torch::Tensor weight,
    torch::Tensor bias,
    int stride,
    int padding,
    double min_value,
    double divisor
);
"""

conv_transpose3d_fused = load_inline(
    name="conv_transpose3d_fused_clamp_div_kernelbench",
    cpp_sources=cpp_sources,
    cuda_sources=cuda_sources,
    functions=["conv_transpose3d_clamp_div_cuda"],
    verbose=False,
    extra_cflags=["-O3"],
    extra_cuda_cflags=["-O3", "--use_fast_math"],
)


class ModelNew(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, min_value, divisor):
        super(ModelNew, self).__init__()
        self.conv_transpose = nn.ConvTranspose3d(
            in_channels,
            out_channels,
            kernel_size,
            stride=stride,
            padding=padding,
        )
        self.stride = stride if isinstance(stride, int) else stride[0]
        self.padding = padding if isinstance(padding, int) else padding[0]
        self.min_value = min_value
        self.divisor = divisor
        self.op = conv_transpose3d_fused

    def forward(self, x):
        return self.op.conv_transpose3d_clamp_div_cuda(
            x.contiguous(),
            self.conv_transpose.weight.contiguous(),
            self.conv_transpose.bias.contiguous(),
            self.stride,
            self.padding,
            self.min_value,
            self.divisor,
        )
