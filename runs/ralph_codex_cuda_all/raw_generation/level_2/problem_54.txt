import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_sources = r"""
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math.h>

#define TILE_M 16
#define TILE_N 16
#define TILE_K 16

__device__ __forceinline__ float gelu_exact(float x) {
    return 0.5f * x * (1.0f + erff(x * 0.70710678118654752440f));
}

__global__ void conv2d_mul_lrelu_gelu_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ bias,
    const float* __restrict__ mult,
    float* __restrict__ out,
    int N, int C, int H, int W,
    int O, int KH, int KW,
    int OH, int OW,
    int total_cols
) {
    __shared__ float sw[TILE_M][TILE_K + 1];
    __shared__ float sx[TILE_K][TILE_N + 1];

    int oc = blockIdx.y * TILE_M + threadIdx.y;
    int col = blockIdx.x * TILE_N + threadIdx.x;

    float acc = 0.0f;
    if (oc < O) {
        acc = bias[oc];
    }

    int n = 0, p = 0, oh = 0, ow = 0;
    if (col < total_cols) {
        n = col / (OH * OW);
        p = col - n * OH * OW;
        oh = p / OW;
        ow = p - oh * OW;
    }

    int K = C * KH * KW;

    for (int kt = 0; kt < K; kt += TILE_K) {
        int kx = kt + threadIdx.x;
        int ky = kt + threadIdx.y;

        if (oc < O && kx < K) {
            sw[threadIdx.y][threadIdx.x] = w[oc * K + kx];
        } else {
            sw[threadIdx.y][threadIdx.x] = 0.0f;
        }

        if (threadIdx.y < TILE_K) {
            if (col < total_cols && ky < K) {
                int tmp = ky;
                int kw = tmp % KW;
                tmp /= KW;
                int kh = tmp % KH;
                int ic = tmp / KH;
                sx[threadIdx.y][threadIdx.x] = x[((n * C + ic) * H + (oh + kh)) * W + (ow + kw)];
            } else {
                sx[threadIdx.y][threadIdx.x] = 0.0f;
            }
        }

        __syncthreads();

        #pragma unroll
        for (int kk = 0; kk < TILE_K; ++kk) {
            acc += sw[threadIdx.y][kk] * sx[kk][threadIdx.x];
        }

        __syncthreads();
    }

    if (oc < O && col < total_cols) {
        float v = acc * mult[oc];
        v = v > 0.0f ? v : 0.01f * v;
        v = gelu_exact(v);
        out[(n * O + oc) * OH * OW + p] = v;
    }
}

torch::Tensor fused_conv2d_mul_lrelu_gelu_cuda(
    torch::Tensor x,
    torch::Tensor weight,
    torch::Tensor bias,
    torch::Tensor mult
) {
    int N = x.size(0);
    int C = x.size(1);
    int H = x.size(2);
    int W = x.size(3);
    int O = weight.size(0);
    int KH = weight.size(2);
    int KW = weight.size(3);
    int OH = H - KH + 1;
    int OW = W - KW + 1;
    int total_cols = N * OH * OW;

    auto out = torch::empty({N, O, OH, OW}, x.options());

    dim3 block(TILE_N, TILE_M);
    dim3 grid((total_cols + TILE_N - 1) / TILE_N, (O + TILE_M - 1) / TILE_M);

    conv2d_mul_lrelu_gelu_kernel<<<grid, block>>>(
        x.data_ptr<float>(),
        weight.data_ptr<float>(),
        bias.data_ptr<float>(),
        mult.data_ptr<float>(),
        out.data_ptr<float>(),
        N, C, H, W, O, KH, KW, OH, OW, total_cols
    );

    return out;
}
"""

cpp_sources = r"""
torch::Tensor fused_conv2d_mul_lrelu_gelu_cuda(
    torch::Tensor x,
    torch::Tensor weight,
    torch::Tensor bias,
    torch::Tensor mult
);
"""

fused_ops = load_inline(
    name="fused_conv2d_mul_lrelu_gelu_ext",
    cpp_sources=cpp_sources,
    cuda_sources=cuda_sources,
    functions=["fused_conv2d_mul_lrelu_gelu_cuda"],
    verbose=False,
    extra_cuda_cflags=["-O3", "--use_fast_math"],
    extra_cflags=["-O3"],
)


class ModelNew(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, multiplier_shape):
        super(ModelNew, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.multiplier = nn.Parameter(torch.randn(multiplier_shape))

    def forward(self, x):
        return fused_ops.fused_conv2d_mul_lrelu_gelu_cuda(
            x.contiguous(),
            self.conv.weight.contiguous(),
            self.conv.bias.contiguous(),
            self.multiplier.contiguous(),
        )
