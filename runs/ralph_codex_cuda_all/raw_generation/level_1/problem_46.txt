import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

avgpool3d_cpp_source = """
#include <torch/extension.h>
torch::Tensor avgpool3d_cuda(torch::Tensor x, int kernel_size, int stride, int padding);
"""

avgpool3d_cuda_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void avgpool3d_k3s2p1_kernel(
    const float* __restrict__ x,
    float* __restrict__ y,
    long total,
    int N, int C, int D, int H, int W,
    int OD, int OH, int OW
) {
    long idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= total) return;

    int ow = idx % OW;
    long t = idx / OW;
    int oh = t % OH;
    t /= OH;
    int od = t % OD;
    t /= OD;
    int c = t % C;
    int n = t / C;

    int id0 = od * 2 - 1;
    int ih0 = oh * 2 - 1;
    int iw0 = ow * 2 - 1;

    float sum = 0.0f;
    long base_nc = ((long)n * C + c) * D * H * W;

    #pragma unroll
    for (int kd = 0; kd < 3; ++kd) {
        int id = id0 + kd;
        if ((unsigned)id < (unsigned)D) {
            #pragma unroll
            for (int kh = 0; kh < 3; ++kh) {
                int ih = ih0 + kh;
                if ((unsigned)ih < (unsigned)H) {
                    long base = base_nc + ((long)id * H + ih) * W;
                    int iw = iw0;
                    if ((unsigned)iw < (unsigned)W) sum += __ldg(x + base + iw);
                    iw = iw0 + 1;
                    if ((unsigned)iw < (unsigned)W) sum += __ldg(x + base + iw);
                    iw = iw0 + 2;
                    if ((unsigned)iw < (unsigned)W) sum += __ldg(x + base + iw);
                }
            }
        }
    }

    y[idx] = sum * (1.0f / 27.0f);
}

__global__ void avgpool3d_generic_kernel(
    const float* __restrict__ x,
    float* __restrict__ y,
    long total,
    int N, int C, int D, int H, int W,
    int OD, int OH, int OW,
    int K, int S, int P
) {
    long idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= total) return;

    int ow = idx % OW;
    long t = idx / OW;
    int oh = t % OH;
    t /= OH;
    int od = t % OD;
    t /= OD;
    int c = t % C;
    int n = t / C;

    int id0 = od * S - P;
    int ih0 = oh * S - P;
    int iw0 = ow * S - P;

    float sum = 0.0f;
    long base_nc = ((long)n * C + c) * D * H * W;

    for (int kd = 0; kd < K; ++kd) {
        int id = id0 + kd;
        if ((unsigned)id >= (unsigned)D) continue;
        for (int kh = 0; kh < K; ++kh) {
            int ih = ih0 + kh;
            if ((unsigned)ih >= (unsigned)H) continue;
            long base = base_nc + ((long)id * H + ih) * W;
            for (int kw = 0; kw < K; ++kw) {
                int iw = iw0 + kw;
                if ((unsigned)iw < (unsigned)W) {
                    sum += __ldg(x + base + iw);
                }
            }
        }
    }

    y[idx] = sum / (float)(K * K * K);
}

torch::Tensor avgpool3d_cuda(torch::Tensor x, int kernel_size, int stride, int padding) {
    int N = (int)x.size(0);
    int C = (int)x.size(1);
    int D = (int)x.size(2);
    int H = (int)x.size(3);
    int W = (int)x.size(4);

    int OD = (D + 2 * padding - kernel_size) / stride + 1;
    int OH = (H + 2 * padding - kernel_size) / stride + 1;
    int OW = (W + 2 * padding - kernel_size) / stride + 1;

    auto y = torch::empty({N, C, OD, OH, OW}, x.options());
    long total = (long)N * C * OD * OH * OW;

    const int threads = 256;
    const int blocks = (int)((total + threads - 1) / threads);

    if (kernel_size == 3 && stride == 2 && padding == 1) {
        avgpool3d_k3s2p1_kernel<<<blocks, threads>>>(
            x.data_ptr<float>(), y.data_ptr<float>(),
            total, N, C, D, H, W, OD, OH, OW
        );
    } else {
        avgpool3d_generic_kernel<<<blocks, threads>>>(
            x.data_ptr<float>(), y.data_ptr<float>(),
            total, N, C, D, H, W, OD, OH, OW,
            kernel_size, stride, padding
        );
    }

    return y;
}
"""

avgpool3d_ext = load_inline(
    name="avgpool3d_inline_ext",
    cpp_sources=avgpool3d_cpp_source,
    cuda_sources=avgpool3d_cuda_source,
    functions=["avgpool3d_cuda"],
    verbose=False,
    extra_cflags=["-O3"],
    extra_cuda_cflags=["-O3", "--use_fast_math"],
)


class ModelNew(nn.Module):
    def __init__(self, kernel_size: int, stride: int = None, padding: int = 0):
        super(ModelNew, self).__init__()
        self.kernel_size = int(kernel_size)
        self.stride = int(kernel_size if stride is None else stride)
        self.padding = int(padding)
        self.avgpool3d_ext = avgpool3d_ext

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return self.avgpool3d_ext.avgpool3d_cuda(
            x, self.kernel_size, self.stride, self.padding
        )
