```python
import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_sources = r"""
#include <torch/extension.h>
#include <cuda_runtime.h>

#define TILE 16

__global__ void lower_tri_matmul_kernel(const float* __restrict__ A,
                                        const float* __restrict__ B,
                                        float* __restrict__ C,
                                        int N) {
    __shared__ float As[TILE][TILE + 1];
    __shared__ float Bs[TILE][TILE + 1];

    int tx = threadIdx.x;
    int ty = threadIdx.y;
    int block_col = blockIdx.x;
    int block_row = blockIdx.y;

    int row = block_row * TILE + ty;
    int col = block_col * TILE + tx;

    if (block_row < block_col) {
        if (row < N && col < N) C[row * N + col] = 0.0f;
        return;
    }

    float acc = 0.0f;

    for (int kt = block_col; kt <= block_row; ++kt) {
        int ak = kt * TILE + tx;
        int bk = kt * TILE + ty;

        float av = 0.0f;
        float bv = 0.0f;

        if (row < N && ak < N && row >= ak) {
            av = A[row * N + ak];
        }
        if (bk < N && col < N && bk >= col) {
            bv = B[bk * N + col];
        }

        As[ty][tx] = av;
        Bs[ty][tx] = bv;
        __syncthreads();

        #pragma unroll
        for (int k = 0; k < TILE; ++k) {
            acc += As[ty][k] * Bs[k][tx];
        }

        __syncthreads();
    }

    if (row < N && col < N) {
        C[row * N + col] = (row >= col) ? acc : 0.0f;
    }
}

torch::Tensor lower_tri_matmul_cuda(torch::Tensor A, torch::Tensor B) {
    int N = (int)A.size(0);
    auto C = torch::empty_like(A);

    dim3 block(TILE, TILE);
    dim3 grid((N + TILE - 1) / TILE, (N + TILE - 1) / TILE);

    lower_tri_matmul_kernel<<<grid, block>>>(A.data_ptr<float>(), B.data_ptr<float>(), C.data_ptr<float>(), N);

    return C;
}
"""

cpp_sources = "torch::Tensor lower_tri_matmul_cuda(torch::Tensor A, torch::Tensor B);"

lower_tri_matmul = load_inline(
    name="lower_tri_matmul_inline",
    cpp_sources=cpp_sources,
    cuda_sources=cuda_sources,
    functions=["lower_tri_matmul_cuda"],
    verbose=False,
    extra_cflags=["-O3"],
    extra_cuda_cflags=["-O3"],
)


class ModelNew(nn.Module):
    def __init__(self):
        super(ModelNew, self).__init__()
        self.op = lower_tri_matmul

    def forward(self, A, B):
        return self.op.lower_tri_matmul_cuda(A, B)
```
