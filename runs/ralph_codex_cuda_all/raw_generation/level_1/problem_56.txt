import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

conv2d_cuda_source = r"""
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void conv2d_direct_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ b,
    float* __restrict__ y,
    int N, int C, int H, int W,
    int OC, int KH, int KW,
    int OH, int OW,
    int stride_h, int stride_w,
    int pad_h, int pad_w,
    int dil_h, int dil_w,
    int has_bias
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * OC * OH * OW;
    if (idx >= total) return;

    int ow = idx % OW;
    int t = idx / OW;
    int oh = t % OH;
    t /= OH;
    int oc = t % OC;
    int n = t / OC;

    float acc = has_bias ? b[oc] : 0.0f;

    int in_h_base = oh * stride_h - pad_h;
    int in_w_base = ow * stride_w - pad_w;

    for (int c = 0; c < C; ++c) {
        const float* x_c = x + ((n * C + c) * H * W);
        const float* w_c = w + ((oc * C + c) * KH * KW);

        #pragma unroll
        for (int kh = 0; kh < 5; ++kh) {
            int ih = in_h_base + kh * dil_h;
            if (ih < 0 || ih >= H) continue;

            #pragma unroll
            for (int kw = 0; kw < 7; ++kw) {
                int iw = in_w_base + kw * dil_w;
                if (iw >= 0 && iw < W) {
                    acc += x_c[ih * W + iw] * w_c[kh * KW + kw];
                }
            }
        }
    }

    y[idx] = acc;
}

torch::Tensor conv2d_direct_cuda(
    torch::Tensor x,
    torch::Tensor weight,
    torch::Tensor bias,
    int stride_h, int stride_w,
    int pad_h, int pad_w,
    int dil_h, int dil_w
) {
    int N = x.size(0);
    int C = x.size(1);
    int H = x.size(2);
    int W = x.size(3);
    int OC = weight.size(0);
    int KH = weight.size(2);
    int KW = weight.size(3);

    int OH = (H + 2 * pad_h - dil_h * (KH - 1) - 1) / stride_h + 1;
    int OW = (W + 2 * pad_w - dil_w * (KW - 1) - 1) / stride_w + 1;

    auto y = torch::empty({N, OC, OH, OW}, x.options());

    int total = N * OC * OH * OW;
    int threads = 256;
    int blocks = (total + threads - 1) / threads;
    int has_bias = bias.defined() && bias.numel() > 0;

    conv2d_direct_kernel<<<blocks, threads>>>(
        x.data_ptr<float>(),
        weight.data_ptr<float>(),
        has_bias ? bias.data_ptr<float>() : nullptr,
        y.data_ptr<float>(),
        N, C, H, W, OC, KH, KW, OH, OW,
        stride_h, stride_w, pad_h, pad_w, dil_h, dil_w, has_bias
    );

    return y;
}
"""

conv2d_cpp_source = r"""
torch::Tensor conv2d_direct_cuda(
    torch::Tensor x,
    torch::Tensor weight,
    torch::Tensor bias,
    int stride_h, int stride_w,
    int pad_h, int pad_w,
    int dil_h, int dil_w
);
"""

conv2d_ext = load_inline(
    name="conv2d_direct_5x7_fp32_ext",
    cpp_sources=conv2d_cpp_source,
    cuda_sources=conv2d_cuda_source,
    functions=["conv2d_direct_cuda"],
    verbose=False,
    extra_cuda_cflags=["-O3", "--use_fast_math"],
)


class ModelNew(nn.Module):
    def __init__(
        self,
        in_channels: int,
        out_channels: int,
        kernel_size: tuple,
        stride: tuple = (1, 1),
        padding: tuple = (0, 0),
        dilation: tuple = (1, 1),
        groups: int = 1,
        bias: bool = False,
    ):
        super(ModelNew, self).__init__()
        self.conv2d = nn.Conv2d(
            in_channels,
            out_channels,
            kernel_size,
            stride=stride,
            padding=padding,
            dilation=dilation,
            groups=groups,
            bias=bias,
        )
        self.stride = stride
        self.padding = padding
        self.dilation = dilation
        self.groups = groups

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        bias = self.conv2d.bias
        if bias is None:
            bias = torch.empty(0, device=x.device, dtype=x.dtype)
        return conv2d_ext.conv2d_direct_cuda(
            x.contiguous(),
            self.conv2d.weight.contiguous(),
            bias,
            self.stride[0],
            self.stride[1],
            self.padding[0],
            self.padding[1],
            self.dilation[0],
            self.dilation[1],
        )
