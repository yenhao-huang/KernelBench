```python
import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_sources = r"""
#include <torch/extension.h>
#include <cuda_runtime.h>

#define BM 16
#define BN 16
#define BK 32

__global__ void at_bt_matmul_kernel(const float* __restrict__ A,
                                    const float* __restrict__ B,
                                    float* __restrict__ C,
                                    int M, int N, int K) {
    __shared__ float As[BK][BM + 1];
    __shared__ float Bs[BK][BN + 1];

    int tx = threadIdx.x;
    int ty = threadIdx.y;

    int m = blockIdx.y * BM + tx;
    int n = blockIdx.x * BN + ty;

    float acc = 0.0f;

    for (int kb = 0; kb < K; kb += BK) {
        for (int off = ty; off < BK; off += BN) {
            int k = kb + off;
            As[off][tx] = (m < M && k < K) ? A[k * M + m] : 0.0f;
        }

        for (int off = tx; off < BK; off += BM) {
            int k = kb + off;
            Bs[off][ty] = (n < N && k < K) ? B[n * K + k] : 0.0f;
        }

        __syncthreads();

        #pragma unroll
        for (int k = 0; k < BK; ++k) {
            acc += As[k][tx] * Bs[k][ty];
        }

        __syncthreads();
    }

    if (m < M && n < N) {
        C[m * N + n] = acc;
    }
}

torch::Tensor at_bt_matmul_cuda(torch::Tensor A, torch::Tensor B) {
    int K = A.size(0);
    int M = A.size(1);
    int N = B.size(0);

    auto C = torch::empty({M, N}, A.options());

    dim3 block(BM, BN);
    dim3 grid((N + BN - 1) / BN, (M + BM - 1) / BM);

    at_bt_matmul_kernel<<<grid, block>>>(
        A.data_ptr<float>(),
        B.data_ptr<float>(),
        C.data_ptr<float>(),
        M, N, K
    );

    return C;
}
"""

cpp_sources = r"""
torch::Tensor at_bt_matmul_cuda(torch::Tensor A, torch::Tensor B);
"""

at_bt_matmul = load_inline(
    name="at_bt_matmul_inline",
    cpp_sources=cpp_sources,
    cuda_sources=cuda_sources,
    functions=["at_bt_matmul_cuda"],
    verbose=False,
    extra_cflags=["-O3"],
    extra_cuda_cflags=["-O3", "--use_fast_math"],
)


class ModelNew(nn.Module):
    def __init__(self):
        super().__init__()
        self.op = at_bt_matmul

    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        return self.op.at_bt_matmul_cuda(A, B)
```
