```python
import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

sigmoid_cpp_source = """
torch::Tensor sigmoid_cuda(torch::Tensor x);
"""

sigmoid_cuda_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <stdint.h>

__device__ __forceinline__ float sigmoid_f32(float v) {
    return 1.0f / (1.0f + __expf(-v));
}

__global__ void sigmoid_scalar_kernel(const float* __restrict__ x, float* __restrict__ out, int64_t n) {
    int64_t idx = (int64_t)blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        out[idx] = sigmoid_f32(x[idx]);
    }
}

__global__ void sigmoid_vec4_kernel(const float4* __restrict__ x, float4* __restrict__ out, int64_t n4) {
    int64_t idx = (int64_t)blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n4) {
        float4 v = x[idx];
        v.x = sigmoid_f32(v.x);
        v.y = sigmoid_f32(v.y);
        v.z = sigmoid_f32(v.z);
        v.w = sigmoid_f32(v.w);
        out[idx] = v;
    }
}

torch::Tensor sigmoid_cuda(torch::Tensor x) {
    auto out = torch::empty_like(x);
    int64_t n = x.numel();

    const int threads = 256;
    uintptr_t xp = reinterpret_cast<uintptr_t>(x.data_ptr<float>());
    uintptr_t op = reinterpret_cast<uintptr_t>(out.data_ptr<float>());

    if ((n % 4 == 0) && ((xp & 15) == 0) && ((op & 15) == 0)) {
        int64_t n4 = n / 4;
        int blocks = (int)((n4 + threads - 1) / threads);
        sigmoid_vec4_kernel<<<blocks, threads>>>(
            reinterpret_cast<const float4*>(x.data_ptr<float>()),
            reinterpret_cast<float4*>(out.data_ptr<float>()),
            n4
        );
    } else {
        int blocks = (int)((n + threads - 1) / threads);
        sigmoid_scalar_kernel<<<blocks, threads>>>(x.data_ptr<float>(), out.data_ptr<float>(), n);
    }

    return out;
}
"""

sigmoid_ext = load_inline(
    name="custom_sigmoid_f32_ext",
    cpp_sources=sigmoid_cpp_source,
    cuda_sources=sigmoid_cuda_source,
    functions=["sigmoid_cuda"],
    verbose=False,
    extra_cflags=["-O3"],
    extra_cuda_cflags=["-O3", "--use_fast_math"],
)


class ModelNew(nn.Module):
    def __init__(self):
        super(ModelNew, self).__init__()
        self.sigmoid_ext = sigmoid_ext

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return self.sigmoid_ext.sigmoid_cuda(x)
```
