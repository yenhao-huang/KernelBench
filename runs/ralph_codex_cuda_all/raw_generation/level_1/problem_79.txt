import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_sources = r"""
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void conv_transpose1d_fp32_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ bias,
    float* __restrict__ out,
    int B, int IC, int OC, int L, int K,
    int stride, int padding, int dilation,
    int outL,
    bool has_bias
) {
    int y = blockIdx.x * blockDim.x + threadIdx.x;
    int bo = blockIdx.y;
    if (y >= outL) return;

    int b = bo / OC;
    int oc = bo - b * OC;

    float acc = has_bias ? bias[oc] : 0.0f;

    #pragma unroll
    for (int k = 0; k < 16; ++k) {
        if (k >= K) break;
        int t = y + padding - k * dilation;
        if (t % stride != 0) continue;
        int ix = t / stride;
        if ((unsigned)ix >= (unsigned)L) continue;

        const float* x_ptr = x + (b * IC * L) + ix;
        const float* w_ptr = w + (k) + oc * K;
        for (int ic = 0; ic < IC; ++ic) {
            acc += x_ptr[ic * L] * w_ptr[ic * OC * K];
        }
    }

    out[(b * OC + oc) * outL + y] = acc;
}

torch::Tensor conv_transpose1d_fp32_cuda(
    torch::Tensor x,
    torch::Tensor w,
    torch::Tensor bias,
    int64_t stride,
    int64_t padding,
    int64_t dilation,
    bool has_bias
) {
    int B = (int)x.size(0);
    int IC = (int)x.size(1);
    int L = (int)x.size(2);
    int OC = (int)w.size(1);
    int K = (int)w.size(2);
    int outL = (L - 1) * (int)stride - 2 * (int)padding + (int)dilation * (K - 1) + 1;

    auto out = torch::empty({B, OC, outL}, x.options());

    const int threads = 256;
    dim3 block(threads);
    dim3 grid((outL + threads - 1) / threads, B * OC);

    conv_transpose1d_fp32_kernel<<<grid, block>>>(
        x.data_ptr<float>(),
        w.data_ptr<float>(),
        has_bias ? bias.data_ptr<float>() : nullptr,
        out.data_ptr<float>(),
        B, IC, OC, L, K,
        (int)stride, (int)padding, (int)dilation,
        outL,
        has_bias
    );

    return out;
}
"""

cpp_sources = r"""
torch::Tensor conv_transpose1d_fp32_cuda(
    torch::Tensor x,
    torch::Tensor w,
    torch::Tensor bias,
    int64_t stride,
    int64_t padding,
    int64_t dilation,
    bool has_bias
);
"""

conv_transpose1d_ext = load_inline(
    name="conv_transpose1d_fp32_ext",
    cpp_sources=cpp_sources,
    cuda_sources=cuda_sources,
    functions=["conv_transpose1d_fp32_cuda"],
    verbose=False,
    extra_cflags=["-O3"],
    extra_cuda_cflags=["-O3"],
)


class ModelNew(nn.Module):
    def __init__(
        self,
        in_channels: int,
        out_channels: int,
        kernel_size: int,
        stride: int = 1,
        padding: int = 0,
        dilation: int = 1,
        bias: bool = False,
    ):
        super().__init__()
        self.conv1d_transpose = nn.ConvTranspose1d(
            in_channels,
            out_channels,
            kernel_size,
            stride=stride,
            padding=padding,
            dilation=dilation,
            bias=bias,
        )
        self.stride = stride
        self.padding = padding
        self.dilation = dilation
        self.has_bias = bias
        self.register_buffer("_empty_bias", torch.empty(0, dtype=torch.float32), persistent=False)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        bias = self.conv1d_transpose.bias if self.conv1d_transpose.bias is not None else self._empty_bias
        return conv_transpose1d_ext.conv_transpose1d_fp32_cuda(
            x,
            self.conv1d_transpose.weight,
            bias,
            self.stride,
            self.padding,
            self.dilation,
            self.has_bias,
        )
