```python
import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

conv3d_cpp_source = """
torch::Tensor conv3d_forward_cuda(torch::Tensor x, torch::Tensor w,
                                  int stride, int padding, int dilation, int groups);
torch::Tensor conv3d_bias_forward_cuda(torch::Tensor x, torch::Tensor w, torch::Tensor b,
                                       int stride, int padding, int dilation, int groups);
"""

conv3d_cuda_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void conv3d_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ b,
    float* __restrict__ y,
    int total,
    int N, int C, int D, int H, int Wd,
    int OC, int KD, int KH, int KW,
    int OD, int OH, int OW,
    int stride, int padding, int dilation, int groups,
    int has_bias
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= total) return;

    int ow = idx % OW;
    int t = idx / OW;
    int oh = t % OH;
    t /= OH;
    int od = t % OD;
    t /= OD;
    int oc = t % OC;
    int n = t / OC;

    int in_per_group = C / groups;
    int out_per_group = OC / groups;
    int g = oc / out_per_group;
    int ic_base = g * in_per_group;

    float acc = has_bias ? b[oc] : 0.0f;

    for (int icg = 0; icg < in_per_group; ++icg) {
        int ic = ic_base + icg;
        for (int kz = 0; kz < KD; ++kz) {
            int iz = od * stride - padding + kz * dilation;
            if ((unsigned)iz >= (unsigned)D) continue;
            for (int ky = 0; ky < KH; ++ky) {
                int iy = oh * stride - padding + ky * dilation;
                if ((unsigned)iy >= (unsigned)H) continue;
                for (int kx = 0; kx < KW; ++kx) {
                    int ix = ow * stride - padding + kx * dilation;
                    if ((unsigned)ix >= (unsigned)Wd) continue;

                    int x_idx = (((n * C + ic) * D + iz) * H + iy) * Wd + ix;
                    int w_idx = (((oc * in_per_group + icg) * KD + kz) * KH + ky) * KW + kx;
                    acc += x[x_idx] * w[w_idx];
                }
            }
        }
    }

    y[idx] = acc;
}

torch::Tensor conv3d_impl(torch::Tensor x, torch::Tensor w, torch::Tensor b,
                          int stride, int padding, int dilation, int groups, int has_bias) {
    int N = x.size(0);
    int C = x.size(1);
    int D = x.size(2);
    int H = x.size(3);
    int Wd = x.size(4);

    int OC = w.size(0);
    int KD = w.size(2);
    int KH = w.size(3);
    int KW = w.size(4);

    int OD = (D + 2 * padding - dilation * (KD - 1) - 1) / stride + 1;
    int OH = (H + 2 * padding - dilation * (KH - 1) - 1) / stride + 1;
    int OW = (Wd + 2 * padding - dilation * (KW - 1) - 1) / stride + 1;

    auto y = torch::empty({N, OC, OD, OH, OW}, x.options());
    int total = N * OC * OD * OH * OW;

    int threads = 256;
    int blocks = (total + threads - 1) / threads;

    const float* bptr = has_bias ? b.data_ptr<float>() : nullptr;
    conv3d_kernel<<<blocks, threads>>>(
        x.data_ptr<float>(), w.data_ptr<float>(), bptr, y.data_ptr<float>(),
        total, N, C, D, H, Wd, OC, KD, KH, KW, OD, OH, OW,
        stride, padding, dilation, groups, has_bias
    );

    return y;
}

torch::Tensor conv3d_forward_cuda(torch::Tensor x, torch::Tensor w,
                                  int stride, int padding, int dilation, int groups) {
    return conv3d_impl(x, w, x, stride, padding, dilation, groups, 0);
}

torch::Tensor conv3d_bias_forward_cuda(torch::Tensor x, torch::Tensor w, torch::Tensor b,
                                       int stride, int padding, int dilation, int groups) {
    return conv3d_impl(x, w, b, stride, padding, dilation, groups, 1);
}
"""

conv3d_ext = load_inline(
    name="kb_direct_conv3d_fp32",
    cpp_sources=conv3d_cpp_source,
    cuda_sources=conv3d_cuda_source,
    functions=["conv3d_forward_cuda", "conv3d_bias_forward_cuda"],
    verbose=False,
)


class ModelNew(nn.Module):
    def __init__(
        self,
        in_channels: int,
        out_channels: int,
        kernel_size: int,
        stride: int = 1,
        padding: int = 0,
        dilation: int = 1,
        groups: int = 1,
        bias: bool = False,
    ):
        super().__init__()
        self.conv3d = nn.Conv3d(
            in_channels,
            out_channels,
            (kernel_size, kernel_size, kernel_size),
            stride=stride,
            padding=padding,
            dilation=dilation,
            groups=groups,
            bias=bias,
        )
        self.stride = stride
        self.padding = padding
        self.dilation = dilation
        self.groups = groups

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        if self.conv3d.bias is None:
            return conv3d_ext.conv3d_forward_cuda(
                x.contiguous(),
                self.conv3d.weight.contiguous(),
                self.stride,
                self.padding,
                self.dilation,
                self.groups,
            )
        return conv3d_ext.conv3d_bias_forward_cuda(
            x.contiguous(),
            self.conv3d.weight.contiguous(),
            self.conv3d.bias.contiguous(),
            self.stride,
            self.padding,
            self.dilation,
            self.groups,
        )
```
