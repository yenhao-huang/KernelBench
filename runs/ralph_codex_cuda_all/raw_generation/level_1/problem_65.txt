import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

conv_transpose2d_cpp_source = """
torch::Tensor conv_transpose2d_forward_cuda(
    torch::Tensor x,
    torch::Tensor weight,
    c10::optional<torch::Tensor> bias,
    int stride_h,
    int stride_w,
    int pad_h,
    int pad_w,
    int out_pad_h,
    int out_pad_w,
    int groups);
"""

conv_transpose2d_cuda_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <c10/cuda/CUDAGuard.h>

__global__ void conv_transpose2d_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ b,
    float* __restrict__ y,
    int N,
    int Cin,
    int H,
    int W,
    int Cout,
    int KH,
    int KW,
    int OH,
    int OW,
    int stride_h,
    int stride_w,
    int pad_h,
    int pad_w,
    int groups,
    int has_bias) {
    long long idx = (long long)blockIdx.x * blockDim.x + threadIdx.x;
    long long total = (long long)N * Cout * OH * OW;

    if (idx >= total) return;

    int ow = idx % OW;
    int oh = (idx / OW) % OH;
    int oc = (idx / (OW * OH)) % Cout;
    int n = idx / ((long long)OW * OH * Cout);

    int cout_per_group = Cout / groups;
    int cin_per_group = Cin / groups;
    int g = oc / cout_per_group;
    int ic_begin = g * cin_per_group;
    int ic_end = ic_begin + cin_per_group;
    int ocg = oc - g * cout_per_group;

    float acc = has_bias ? b[oc] : 0.0f;

    for (int ic = ic_begin; ic < ic_end; ++ic) {
        const float* x_base = x + ((long long)n * Cin + ic) * H * W;
        const float* w_base = w + ((long long)ic * cout_per_group + ocg) * KH * KW;

        #pragma unroll
        for (int kh = 0; kh < 3; ++kh) {
            int ih_num = oh + pad_h - kh;
            if (ih_num % stride_h != 0) continue;
            int ih = ih_num / stride_h;
            if ((unsigned)ih >= (unsigned)H) continue;

            #pragma unroll
            for (int kw = 0; kw < 7; ++kw) {
                int iw_num = ow + pad_w - kw;
                if (iw_num % stride_w != 0) continue;
                int iw = iw_num / stride_w;
                if ((unsigned)iw < (unsigned)W) {
                    acc += x_base[ih * W + iw] * w_base[kh * KW + kw];
                }
            }
        }
    }

    y[idx] = acc;
}

torch::Tensor conv_transpose2d_forward_cuda(
    torch::Tensor x,
    torch::Tensor weight,
    c10::optional<torch::Tensor> bias,
    int stride_h,
    int stride_w,
    int pad_h,
    int pad_w,
    int out_pad_h,
    int out_pad_w,
    int groups) {
    const at::cuda::OptionalCUDAGuard device_guard(device_of(x));

    int N = x.size(0);
    int Cin = x.size(1);
    int H = x.size(2);
    int W = x.size(3);
    int KH = weight.size(2);
    int KW = weight.size(3);
    int Cout = weight.size(1) * groups;

    int OH = (H - 1) * stride_h - 2 * pad_h + KH + out_pad_h;
    int OW = (W - 1) * stride_w - 2 * pad_w + KW + out_pad_w;

    auto y = torch::empty({N, Cout, OH, OW}, x.options());

    const float* bias_ptr = nullptr;
    int has_bias = 0;
    if (bias.has_value() && bias.value().defined()) {
        bias_ptr = bias.value().data_ptr<float>();
        has_bias = 1;
    }

    long long total = (long long)N * Cout * OH * OW;
    const int threads = 256;
    const int blocks = (int)((total + threads - 1) / threads);

    conv_transpose2d_kernel<<<blocks, threads>>>(
        x.data_ptr<float>(),
        weight.data_ptr<float>(),
        bias_ptr,
        y.data_ptr<float>(),
        N, Cin, H, W, Cout, KH, KW, OH, OW,
        stride_h, stride_w, pad_h, pad_w, groups, has_bias);

    return y;
}
"""

_conv_transpose2d_ext = load_inline(
    name="conv_transpose2d_kernelbench_fp32",
    cpp_sources=conv_transpose2d_cpp_source,
    cuda_sources=conv_transpose2d_cuda_source,
    functions=["conv_transpose2d_forward_cuda"],
    verbose=False,
    extra_cuda_cflags=["-O3", "--use_fast_math"],
)


def _pair(v):
    if isinstance(v, tuple):
        return v
    return (v, v)


class ModelNew(nn.Module):
    def __init__(
        self,
        in_channels: int,
        out_channels: int,
        kernel_size: tuple,
        stride: int = 1,
        padding: int = 0,
        output_padding: int = 0,
        groups: int = 1,
        bias: bool = False,
    ):
        super(ModelNew, self).__init__()
        self.conv_transpose2d = nn.ConvTranspose2d(
            in_channels,
            out_channels,
            kernel_size,
            stride=stride,
            padding=padding,
            output_padding=output_padding,
            groups=groups,
            bias=bias,
        )
        self.stride = _pair(stride)
        self.padding = _pair(padding)
        self.output_padding = _pair(output_padding)
        self.groups = groups

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return _conv_transpose2d_ext.conv_transpose2d_forward_cuda(
            x,
            self.conv_transpose2d.weight,
            self.conv_transpose2d.bias,
            self.stride[0],
            self.stride[1],
            self.padding[0],
            self.padding[1],
            self.output_padding[0],
            self.output_padding[1],
            self.groups,
        )
