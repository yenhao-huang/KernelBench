import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_sources = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda_runtime.h>

#define TILE 16

__global__ void tensor_matmul_3d_kernel(
    const float* __restrict__ A,
    const float* __restrict__ B,
    float* __restrict__ C,
    int rows,
    int K,
    int L
) {
    __shared__ float As[TILE][TILE + 1];
    __shared__ float Bs[TILE][TILE + 1];

    int tx = threadIdx.x;
    int ty = threadIdx.y;

    int row = blockIdx.y * TILE + ty;
    int col = blockIdx.x * TILE + tx;

    float acc = 0.0f;

    for (int t = 0; t < K; t += TILE) {
        int a_col = t + tx;
        int b_row = t + ty;

        As[ty][tx] = (row < rows && a_col < K) ? A[row * K + a_col] : 0.0f;
        Bs[ty][tx] = (b_row < K && col < L) ? B[b_row * L + col] : 0.0f;

        __syncthreads();

        #pragma unroll
        for (int i = 0; i < TILE; ++i) {
            acc += As[ty][i] * Bs[i][tx];
        }

        __syncthreads();
    }

    if (row < rows && col < L) {
        C[row * L + col] = acc;
    }
}

torch::Tensor tensor_matmul_3d_cuda(torch::Tensor A, torch::Tensor B) {
    int N = A.size(0);
    int M = A.size(1);
    int K = A.size(2);
    int L = B.size(1);
    int rows = N * M;

    auto C = torch::empty({N, M, L}, A.options());

    dim3 block(TILE, TILE);
    dim3 grid((L + TILE - 1) / TILE, (rows + TILE - 1) / TILE);

    tensor_matmul_3d_kernel<<<grid, block, 0, at::cuda::getCurrentCUDAStream()>>>(
        A.data_ptr<float>(),
        B.data_ptr<float>(),
        C.data_ptr<float>(),
        rows,
        K,
        L
    );

    return C;
}
"""

cpp_sources = r"""
torch::Tensor tensor_matmul_3d_cuda(torch::Tensor A, torch::Tensor B);
"""

tensor_matmul_3d = load_inline(
    name="tensor_matmul_3d_inline",
    cpp_sources=cpp_sources,
    cuda_sources=cuda_sources,
    functions=["tensor_matmul_3d_cuda"],
    verbose=False,
)


class ModelNew(nn.Module):
    def __init__(self):
        super(ModelNew, self).__init__()
        self.op = tensor_matmul_3d

    def forward(self, A, B):
        return self.op.tensor_matmul_3d_cuda(A, B)
