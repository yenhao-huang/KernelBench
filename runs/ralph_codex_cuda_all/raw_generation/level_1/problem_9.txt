import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cpp_sources = """
torch::Tensor tall_skinny_matmul_cuda(torch::Tensor A, torch::Tensor B);
"""

cuda_sources = r"""
#include <torch/extension.h>
#include <cuda_runtime.h>

template<int BM, int BN, int BK>
__global__ void matmul_k32_kernel(const float* __restrict__ A,
                                  const float* __restrict__ B,
                                  float* __restrict__ C,
                                  int M, int N, int K) {
    __shared__ float As[BM][BK + 1];
    __shared__ float Bs[BK][BN + 1];

    const int tx = threadIdx.x;
    const int ty = threadIdx.y;
    const int row = blockIdx.y * BM + ty;
    const int col = blockIdx.x * BN + tx;

    for (int k = tx; k < K; k += BN) {
        As[ty][k] = (row < M) ? A[row * K + k] : 0.0f;
    }

    for (int k = ty; k < K; k += BM) {
        Bs[k][tx] = (col < N) ? B[k * N + col] : 0.0f;
    }

    __syncthreads();

    float acc = 0.0f;
    #pragma unroll
    for (int k = 0; k < 32; ++k) {
        if (k < K) {
            acc += As[ty][k] * Bs[k][tx];
        }
    }

    if (row < M && col < N) {
        C[row * N + col] = acc;
    }
}

__global__ void matmul_generic_kernel(const float* __restrict__ A,
                                      const float* __restrict__ B,
                                      float* __restrict__ C,
                                      int M, int N, int K) {
    const int row = blockIdx.y * blockDim.y + threadIdx.y;
    const int col = blockIdx.x * blockDim.x + threadIdx.x;

    if (row < M && col < N) {
        float acc = 0.0f;
        for (int k = 0; k < K; ++k) {
            acc += A[row * K + k] * B[k * N + col];
        }
        C[row * N + col] = acc;
    }
}

torch::Tensor tall_skinny_matmul_cuda(torch::Tensor A, torch::Tensor B) {
    const int M = static_cast<int>(A.size(0));
    const int K = static_cast<int>(A.size(1));
    const int N = static_cast<int>(B.size(1));

    auto C = torch::empty({M, N}, A.options());

    if (K == 32) {
        constexpr int BM = 16;
        constexpr int BN = 16;
        constexpr int BK = 32;
        dim3 block(BN, BM);
        dim3 grid((N + BN - 1) / BN, (M + BM - 1) / BM);
        matmul_k32_kernel<BM, BN, BK><<<grid, block>>>(
            A.data_ptr<float>(), B.data_ptr<float>(), C.data_ptr<float>(), M, N, K
        );
    } else {
        dim3 block(16, 16);
        dim3 grid((N + 15) / 16, (M + 15) / 16);
        matmul_generic_kernel<<<grid, block>>>(
            A.data_ptr<float>(), B.data_ptr<float>(), C.data_ptr<float>(), M, N, K
        );
    }

    return C;
}
"""

matmul_ext = load_inline(
    name="kb_tall_skinny_matmul_fp32",
    cpp_sources=cpp_sources,
    cuda_sources=cuda_sources,
    functions=["tall_skinny_matmul_cuda"],
    verbose=False,
    extra_cflags=["-O3"],
    extra_cuda_cflags=["-O3", "--use_fast_math"],
)


class ModelNew(nn.Module):
    def __init__(self):
        super(ModelNew, self).__init__()
        self.matmul_ext = matmul_ext

    def forward(self, A, B):
        return self.matmul_ext.tall_skinny_matmul_cuda(A, B)
