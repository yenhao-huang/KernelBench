import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

maxpool3d_cpp_source = """
torch::Tensor maxpool3d_forward_cuda(torch::Tensor x, int kernel_size, int stride, int padding, int dilation, bool ceil_mode);
"""

maxpool3d_cuda_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <float.h>

static inline int64_t pool_out_dim(int64_t input, int kernel, int stride, int padding, int dilation, bool ceil_mode) {
    int64_t effective = dilation * (kernel - 1) + 1;
    int64_t out;
    if (ceil_mode) {
        out = (input + 2 * padding - effective + stride - 1) / stride + 1;
        if ((out - 1) * stride >= input + padding) {
            --out;
        }
    } else {
        out = (input + 2 * padding - effective) / stride + 1;
    }
    return out > 0 ? out : 0;
}

__global__ void maxpool3d_kernel(
    const float* __restrict__ x,
    float* __restrict__ out,
    int total,
    int N,
    int C,
    int D,
    int H,
    int W,
    int OD,
    int OH,
    int OW,
    int kernel_size,
    int stride,
    int padding,
    int dilation
) {
    int linear = blockIdx.x * blockDim.x + threadIdx.x;
    if (linear >= total) return;

    int ow = linear % OW;
    int t = linear / OW;
    int oh = t % OH;
    t /= OH;
    int od = t % OD;
    t /= OD;
    int c = t % C;
    int n = t / C;

    int d_start = od * stride - padding;
    int h_start = oh * stride - padding;
    int w_start = ow * stride - padding;

    float max_val = -FLT_MAX;

    #pragma unroll
    for (int kd = 0; kd < 3; ++kd) {
        int id = d_start + kd * dilation;
        if (kd >= kernel_size || id < 0 || id >= D) continue;

        #pragma unroll
        for (int kh = 0; kh < 3; ++kh) {
            int ih = h_start + kh * dilation;
            if (kh >= kernel_size || ih < 0 || ih >= H) continue;

            #pragma unroll
            for (int kw = 0; kw < 3; ++kw) {
                int iw = w_start + kw * dilation;
                if (kw >= kernel_size || iw < 0 || iw >= W) continue;

                int64_t idx = (((int64_t)n * C + c) * D + id) * H * W + ih * W + iw;
                float v = x[idx];
                max_val = v > max_val ? v : max_val;
            }
        }
    }

    out[linear] = max_val;
}

torch::Tensor maxpool3d_forward_cuda(torch::Tensor x, int kernel_size, int stride, int padding, int dilation, bool ceil_mode) {
    int64_t N = x.size(0);
    int64_t C = x.size(1);
    int64_t D = x.size(2);
    int64_t H = x.size(3);
    int64_t W = x.size(4);

    int64_t OD = pool_out_dim(D, kernel_size, stride, padding, dilation, ceil_mode);
    int64_t OH = pool_out_dim(H, kernel_size, stride, padding, dilation, ceil_mode);
    int64_t OW = pool_out_dim(W, kernel_size, stride, padding, dilation, ceil_mode);

    auto out = torch::empty({N, C, OD, OH, OW}, x.options());
    int64_t total64 = N * C * OD * OH * OW;
    int total = (int)total64;

    const int threads = 256;
    const int blocks = (total + threads - 1) / threads;

    maxpool3d_kernel<<<blocks, threads>>>(
        x.data_ptr<float>(),
        out.data_ptr<float>(),
        total,
        (int)N, (int)C, (int)D, (int)H, (int)W,
        (int)OD, (int)OH, (int)OW,
        kernel_size, stride, padding, dilation
    );

    return out;
}
"""

maxpool3d_ext = load_inline(
    name="maxpool3d_custom_ext",
    cpp_sources=maxpool3d_cpp_source,
    cuda_sources=maxpool3d_cuda_source,
    functions=["maxpool3d_forward_cuda"],
    verbose=False,
    extra_cuda_cflags=["-O3"],
)


class ModelNew(nn.Module):
    def __init__(
        self,
        kernel_size: int,
        stride: int = None,
        padding: int = 0,
        dilation: int = 1,
        return_indices: bool = False,
        ceil_mode: bool = False,
    ):
        super().__init__()
        self.kernel_size = int(kernel_size)
        self.stride = int(stride) if stride is not None else int(kernel_size)
        self.padding = int(padding)
        self.dilation = int(dilation)
        self.return_indices = bool(return_indices)
        self.ceil_mode = bool(ceil_mode)
        self.maxpool3d_ext = maxpool3d_ext

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return self.maxpool3d_ext.maxpool3d_forward_cuda(
            x,
            self.kernel_size,
            self.stride,
            self.padding,
            self.dilation,
            self.ceil_mode,
        )
