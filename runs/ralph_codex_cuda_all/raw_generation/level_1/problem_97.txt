import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

sdpa_cpp_source = """
torch::Tensor sdpa_forward_cuda(torch::Tensor Q, torch::Tensor K, torch::Tensor V);
"""

sdpa_cuda_source = r"""
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math.h>

#define BLOCK_SIZE 256
#define MAX_S 512

__device__ __forceinline__ float warp_reduce_sum(float v) {
    for (int offset = 16; offset > 0; offset >>= 1) {
        v += __shfl_down_sync(0xffffffff, v, offset);
    }
    return v;
}

__device__ __forceinline__ float warp_reduce_max(float v) {
    for (int offset = 16; offset > 0; offset >>= 1) {
        v = fmaxf(v, __shfl_down_sync(0xffffffff, v, offset));
    }
    return v;
}

__device__ __forceinline__ float block_reduce_sum(float v) {
    __shared__ float shared[32];
    int lane = threadIdx.x & 31;
    int wid = threadIdx.x >> 5;

    v = warp_reduce_sum(v);
    if (lane == 0) shared[wid] = v;
    __syncthreads();

    v = (threadIdx.x < (blockDim.x >> 5)) ? shared[lane] : 0.0f;
    if (wid == 0) v = warp_reduce_sum(v);
    return v;
}

__device__ __forceinline__ float block_reduce_max(float v) {
    __shared__ float shared[32];
    int lane = threadIdx.x & 31;
    int wid = threadIdx.x >> 5;

    v = warp_reduce_max(v);
    if (lane == 0) shared[wid] = v;
    __syncthreads();

    v = (threadIdx.x < (blockDim.x >> 5)) ? shared[lane] : -3.402823466e38f;
    if (wid == 0) v = warp_reduce_max(v);
    return v;
}

__global__ void sdpa_kernel(
    const float* __restrict__ Q,
    const float* __restrict__ K,
    const float* __restrict__ V,
    float* __restrict__ O,
    int total_rows,
    int S,
    int D,
    float scale
) {
    __shared__ float scores[MAX_S];

    int row = blockIdx.x;
    int tid = threadIdx.x;

    if (row >= total_rows) return;

    int q_pos = row % S;
    int bh = row / S;

    const long long q_base = ((long long)bh * S + q_pos) * D;
    const long long kv_base = (long long)bh * S * D;
    const long long o_base = q_base;

    for (int k_pos = 0; k_pos < S; ++k_pos) {
        const long long k_base = kv_base + (long long)k_pos * D;
        float sum = 0.0f;

        for (int d = tid; d < D; d += BLOCK_SIZE) {
            sum += Q[q_base + d] * K[k_base + d];
        }

        sum = block_reduce_sum(sum);
        if (tid == 0) {
            scores[k_pos] = sum * scale;
        }
        __syncthreads();
    }

    float local_max = -3.402823466e38f;
    for (int i = tid; i < S; i += BLOCK_SIZE) {
        local_max = fmaxf(local_max, scores[i]);
    }

    float row_max = block_reduce_max(local_max);
    row_max = __shfl_sync(0xffffffff, row_max, 0);

    float local_sum = 0.0f;
    for (int i = tid; i < S; i += BLOCK_SIZE) {
        float e = expf(scores[i] - row_max);
        scores[i] = e;
        local_sum += e;
    }

    float denom = block_reduce_sum(local_sum);
    denom = __shfl_sync(0xffffffff, denom, 0);
    float inv_denom = 1.0f / denom;

    __syncthreads();

    for (int d = tid; d < D; d += BLOCK_SIZE) {
        float acc = 0.0f;
        for (int k_pos = 0; k_pos < S; ++k_pos) {
            const long long v_idx = kv_base + (long long)k_pos * D + d;
            acc += scores[k_pos] * inv_denom * V[v_idx];
        }
        O[o_base + d] = acc;
    }
}

torch::Tensor sdpa_forward_cuda(torch::Tensor Q, torch::Tensor K, torch::Tensor V) {
    auto O = torch::empty_like(Q);

    const int B = Q.size(0);
    const int H = Q.size(1);
    const int S = Q.size(2);
    const int D = Q.size(3);
    const int total_rows = B * H * S;
    const float scale = rsqrtf((float)D);

    sdpa_kernel<<<total_rows, BLOCK_SIZE>>>(
        Q.data_ptr<float>(),
        K.data_ptr<float>(),
        V.data_ptr<float>(),
        O.data_ptr<float>(),
        total_rows,
        S,
        D,
        scale
    );

    return O;
}
"""

sdpa_ext = load_inline(
    name="custom_sdpa_fp32_kernelbench",
    cpp_sources=sdpa_cpp_source,
    cuda_sources=sdpa_cuda_source,
    functions=["sdpa_forward_cuda"],
    verbose=False,
    extra_cflags=["-O3"],
    extra_cuda_cflags=["-O3", "--use_fast_math"],
)


class ModelNew(nn.Module):
    def __init__(self):
        super(ModelNew, self).__init__()
        self.sdpa_ext = sdpa_ext

    def forward(self, Q: torch.Tensor, K: torch.Tensor, V: torch.Tensor) -> torch.Tensor:
        return self.sdpa_ext.sdpa_forward_cuda(Q, K, V)
