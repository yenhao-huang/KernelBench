```python
import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

matmul_cpp_source = """
torch::Tensor matmul_cuda(torch::Tensor A, torch::Tensor B);
"""

matmul_cuda_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>

#define BM 64
#define BN 64
#define BK 16
#define TX 16
#define TY 16

__global__ void sgemm_64x64x16_kernel(
    const float* __restrict__ A,
    const float* __restrict__ B,
    float* __restrict__ C,
    int M,
    int K,
    int N
) {
    __shared__ float As[BM][BK];
    __shared__ float Bs[BK][BN];

    const int tx = threadIdx.x;
    const int ty = threadIdx.y;
    const int tid = ty * TX + tx;

    const int block_m = blockIdx.y * BM;
    const int block_n = blockIdx.x * BN;

    float acc[4][4];

    #pragma unroll
    for (int i = 0; i < 4; ++i) {
        #pragma unroll
        for (int j = 0; j < 4; ++j) {
            acc[i][j] = 0.0f;
        }
    }

    for (int kt = 0; kt < K; kt += BK) {
        for (int idx = tid; idx < BM * BK; idx += TX * TY) {
            int r = idx / BK;
            int c = idx - r * BK;
            int gr = block_m + r;
            int gc = kt + c;
            As[r][c] = (gr < M && gc < K) ? A[gr * K + gc] : 0.0f;
        }

        for (int idx = tid; idx < BK * BN; idx += TX * TY) {
            int r = idx / BN;
            int c = idx - r * BN;
            int gr = kt + r;
            int gc = block_n + c;
            Bs[r][c] = (gr < K && gc < N) ? B[gr * N + gc] : 0.0f;
        }

        __syncthreads();

        #pragma unroll
        for (int k = 0; k < BK; ++k) {
            float av[4];
            float bv[4];

            #pragma unroll
            for (int i = 0; i < 4; ++i) {
                av[i] = As[ty + i * TY][k];
            }

            #pragma unroll
            for (int j = 0; j < 4; ++j) {
                bv[j] = Bs[k][tx + j * TX];
            }

            #pragma unroll
            for (int i = 0; i < 4; ++i) {
                #pragma unroll
                for (int j = 0; j < 4; ++j) {
                    acc[i][j] += av[i] * bv[j];
                }
            }
        }

        __syncthreads();
    }

    #pragma unroll
    for (int i = 0; i < 4; ++i) {
        int row = block_m + ty + i * TY;
        if (row < M) {
            #pragma unroll
            for (int j = 0; j < 4; ++j) {
                int col = block_n + tx + j * TX;
                if (col < N) {
                    C[row * N + col] = acc[i][j];
                }
            }
        }
    }
}

torch::Tensor matmul_cuda(torch::Tensor A, torch::Tensor B) {
    const int M = (int)A.size(0);
    const int K = (int)A.size(1);
    const int N = (int)B.size(1);

    auto C = torch::empty({M, N}, A.options());

    dim3 block(TX, TY);
    dim3 grid((N + BN - 1) / BN, (M + BM - 1) / BM);

    sgemm_64x64x16_kernel<<<grid, block>>>(
        A.data_ptr<float>(),
        B.data_ptr<float>(),
        C.data_ptr<float>(),
        M,
        K,
        N
    );

    return C;
}
"""

matmul_ext = load_inline(
    name="custom_sgemm_64x64x16_fp32",
    cpp_sources=matmul_cpp_source,
    cuda_sources=matmul_cuda_source,
    functions=["matmul_cuda"],
    verbose=False,
    extra_cuda_cflags=["-O3", "--use_fast_math"],
)


class ModelNew(nn.Module):
    def __init__(self):
        super(ModelNew, self).__init__()
        self.matmul_ext = matmul_ext

    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        return self.matmul_ext.matmul_cuda(A, B)
```
