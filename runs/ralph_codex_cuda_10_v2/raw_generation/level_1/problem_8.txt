```python
import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

matmul_cpp_source = """
torch::Tensor matmul_cuda(torch::Tensor A, torch::Tensor B);
"""

matmul_cuda_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>

#define BM 64
#define BN 64
#define BK 16
#define TM 4
#define TN 4

__global__ void matmul_kernel(const float* __restrict__ A,
                              const float* __restrict__ B,
                              float* __restrict__ C,
                              int M, int K, int N) {
    __shared__ float As[BM][BK + 1];
    __shared__ float Bs[BK][BN + 1];

    int tx = threadIdx.x;
    int ty = threadIdx.y;
    int tid = ty * blockDim.x + tx;

    int block_row = blockIdx.y * BM;
    int block_col = blockIdx.x * BN;

    float acc[TM][TN];
    #pragma unroll
    for (int i = 0; i < TM; ++i) {
        #pragma unroll
        for (int j = 0; j < TN; ++j) {
            acc[i][j] = 0.0f;
        }
    }

    for (int kb = 0; kb < K; kb += BK) {
        for (int idx = tid; idx < BM * BK; idx += 256) {
            int r = idx / BK;
            int c = idx - r * BK;
            int gr = block_row + r;
            int gc = kb + c;
            As[r][c] = (gr < M && gc < K) ? A[gr * K + gc] : 0.0f;
        }

        for (int idx = tid; idx < BK * BN; idx += 256) {
            int r = idx / BN;
            int c = idx - r * BN;
            int gr = kb + r;
            int gc = block_col + c;
            Bs[r][c] = (gr < K && gc < N) ? B[gr * N + gc] : 0.0f;
        }

        __syncthreads();

        int rows[TM] = {ty, ty + 16, ty + 32, ty + 48};
        int cols[TN] = {tx, tx + 16, tx + 32, tx + 48};

        #pragma unroll
        for (int k = 0; k < BK; ++k) {
            float a[TM];
            float b[TN];

            #pragma unroll
            for (int i = 0; i < TM; ++i) {
                a[i] = As[rows[i]][k];
            }

            #pragma unroll
            for (int j = 0; j < TN; ++j) {
                b[j] = Bs[k][cols[j]];
            }

            #pragma unroll
            for (int i = 0; i < TM; ++i) {
                #pragma unroll
                for (int j = 0; j < TN; ++j) {
                    acc[i][j] += a[i] * b[j];
                }
            }
        }

        __syncthreads();
    }

    int rows[TM] = {ty, ty + 16, ty + 32, ty + 48};
    int cols[TN] = {tx, tx + 16, tx + 32, tx + 48};

    #pragma unroll
    for (int i = 0; i < TM; ++i) {
        int gr = block_row + rows[i];
        if (gr < M) {
            #pragma unroll
            for (int j = 0; j < TN; ++j) {
                int gc = block_col + cols[j];
                if (gc < N) {
                    C[gr * N + gc] = acc[i][j];
                }
            }
        }
    }
}

torch::Tensor matmul_cuda(torch::Tensor A, torch::Tensor B) {
    int M = A.size(0);
    int K = A.size(1);
    int N = B.size(1);

    auto C = torch::empty({M, N}, A.options());

    dim3 block(16, 16);
    dim3 grid((N + BN - 1) / BN, (M + BM - 1) / BM);

    matmul_kernel<<<grid, block>>>(A.data_ptr<float>(), B.data_ptr<float>(), C.data_ptr<float>(), M, K, N);

    return C;
}
"""

matmul_ext = load_inline(
    name="kernelbench_irregular_matmul_fp32",
    cpp_sources=matmul_cpp_source,
    cuda_sources=matmul_cuda_source,
    functions=["matmul_cuda"],
    verbose=False,
    extra_cflags=["-O3"],
    extra_cuda_cflags=["-O3", "--use_fast_math"],
)


class ModelNew(nn.Module):
    def __init__(self):
        super(ModelNew, self).__init__()
        self.matmul_ext = matmul_ext

    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        return self.matmul_ext.matmul_cuda(A, B)
```
