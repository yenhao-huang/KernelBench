```python
import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

matmul_cpp_source = """
torch::Tensor skinny_matmul_cuda(torch::Tensor A, torch::Tensor B);
"""

matmul_cuda_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>

#define BM 16
#define BN 16
#define BK 32

__global__ void skinny_matmul_kernel(
    const float* __restrict__ A,
    const float* __restrict__ B,
    float* __restrict__ C,
    int M,
    int K,
    int N
) {
    __shared__ float As[BM][BK];
    __shared__ float Bs[BK][BN];

    int tx = threadIdx.x;
    int ty = threadIdx.y;
    int row = blockIdx.y * BM + ty;
    int col = blockIdx.x * BN + tx;

    float acc = 0.0f;

    for (int k0 = 0; k0 < K; k0 += BK) {
        int a_k = k0 + tx;
        if (row < M && a_k < K) {
            As[ty][tx] = A[row * K + a_k];
        } else {
            As[ty][tx] = 0.0f;
        }

        for (int kk = ty; kk < BK; kk += BM) {
            int b_k = k0 + kk;
            if (b_k < K && col < N) {
                Bs[kk][tx] = B[b_k * N + col];
            } else {
                Bs[kk][tx] = 0.0f;
            }
        }

        __syncthreads();

        #pragma unroll
        for (int kk = 0; kk < BK; ++kk) {
            acc += As[ty][kk] * Bs[kk][tx];
        }

        __syncthreads();
    }

    if (row < M && col < N) {
        C[row * N + col] = acc;
    }
}

torch::Tensor skinny_matmul_cuda(torch::Tensor A, torch::Tensor B) {
    int M = (int)A.size(0);
    int K = (int)A.size(1);
    int N = (int)B.size(1);

    auto C = torch::empty({M, N}, A.options());

    dim3 block(BN, BM);
    dim3 grid((N + BN - 1) / BN, (M + BM - 1) / BM);

    skinny_matmul_kernel<<<grid, block>>>(
        A.data_ptr<float>(),
        B.data_ptr<float>(),
        C.data_ptr<float>(),
        M,
        K,
        N
    );

    return C;
}
"""

skinny_matmul = load_inline(
    name="skinny_matmul_inline_cuda",
    cpp_sources=matmul_cpp_source,
    cuda_sources=matmul_cuda_source,
    functions=["skinny_matmul_cuda"],
    extra_cflags=["-O3"],
    extra_cuda_cflags=["-O3", "--use_fast_math"],
    verbose=False,
)


class ModelNew(nn.Module):
    def __init__(self):
        super(ModelNew, self).__init__()
        self.matmul = skinny_matmul

    def forward(self, A, B):
        return self.matmul.skinny_matmul_cuda(A, B)
```
